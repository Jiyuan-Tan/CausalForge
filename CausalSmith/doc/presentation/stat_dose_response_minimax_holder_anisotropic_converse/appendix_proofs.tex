\begin{proof}[Proof of \cref{obj:thm:sharp-pointwise-lower-bound}]
\begin{enumerate}
\item The hypotheses of \cref{obj:lem:oracle-dose-regression-lower-all-beta} are exactly the hypotheses assumed here: the three smoothness parameters are positive, the regime constants satisfy the stated interior and boundedness conditions, and the strict-slack baseline condition holds for the same tuple \((d,\beta,s,M,c_0,\varepsilon_0,t_0)\). Applying that lemma gives a constant \(c_{\mathrm{or}}>0\) such that, for all sufficiently large \(n\),
\[
R_n\!\left(
    \mathcal P_{\alpha,\beta,s}(M,c_0,\varepsilon_0,t_0),
    t_0
  \right)
\ge
c_{\mathrm{or}}\,n^{-2\alpha/(2\alpha+1)} .
\]
Taking \(c=c_{\mathrm{or}}\) gives the asserted positive constant and the desired eventual lower bound. % lean: sharp_pointwise_lower_bound
\end{enumerate}
\end{proof}

\begin{proof}[Proof of \cref{obj:thm:sharp-minimax-smooth-covariate}]
\begin{enumerate}
\item By \cref{obj:thm:sharp-pointwise-lower-bound}, applied with the stated positivity, regime-constant, and strict-slack hypotheses, there is a constant \(c>0\) such that, for all sufficiently large \(n\),
\[
c\,n^{-2\alpha/(2\alpha+1)}
\le
R_n\!\left(
\mathcal P_{\alpha,\beta,s}(M,c_0,\varepsilon_0,t_0),t_0
\right).
\]
Fix this \(c\). % lean: sharp_pointwise_lower_bound

\item Intersect this eventual lower-bound statement with the eventual condition \(n\ge 1\). Thus, for all sufficiently large \(n\), both the displayed lower bound above holds and \(n\ge 1\). % lean: Filter.eventually_ge_atTop

\item For every such \(n\), the assumptions \(\alpha>0\), \(s>0\), \(d\le 4s\), and \(n\ge 1\) allow us to apply \cref{obj:lem:rho-oracle-regime-algebra}. Hence
\[
\rho_n
=
n^{-2\alpha/(2\alpha+1)}.
\]
Equivalently,
\[
n^{-2\alpha/(2\alpha+1)}=\rho_n.
\]
Combining this identity with the lower bound from Step 2 gives the desired eventual conjunction for the same constant \(c>0\). % lean: rho_oracle_regime_algebra
\end{enumerate}
\end{proof}

\begin{proof}[Proof of \cref{obj:thm:frontier-bracket-deficient}]
\begin{enumerate}
\item By \cref{obj:lem:oracle-dose-regression-lower-all-beta}, applied with the stated positivity, regime, and strict-slack baseline hypotheses, there is a constant \(c>0\) such that, for all sufficiently large \(n\),
\[
R_n\!\left(\mathcal P_{\alpha,\beta,s}(M,c_0,\varepsilon_0,t_0),t_0\right)
\ge
c\,n^{-2\alpha/(2\alpha+1)} .
\]
Fix this \(c\). % lean: oracle_dose_regression_lower_all_beta

\item Now restrict \(n\) to a sufficiently large tail on which the preceding lower bound holds and also \(n\ge 1\). This is only the usual intersection of two eventual conditions: the eventual lower-bound tail and the eventual tail \(\{n:n\ge 1\}\). % lean: Filter.eventually_ge_atTop

\item For such an \(n\), apply \cref{obj:lem:rho-deficient-regime-algebra} with \(\alpha>0\), \(s>0\), \(4s<d\), and \(n\ge 1\). It gives
\[
\rho_n
=
n^{-\,2/(1+d/(4s)+1/\alpha)}
\]
and
\[
\frac{2}{1+d/(4s)+1/\alpha}
<
\frac{2\alpha}{2\alpha+1}.
\]
% lean: rho_deficient_regime_algebra

\item Combining the lower bound from Step 1 with the two conclusions from Step 3 yields, on the same sufficiently large tail of sample sizes,
\[
c\,n^{-2\alpha/(2\alpha+1)}
\le
R_n\!\left(
\mathcal P_{\alpha,\beta,s}(M,c_0,\varepsilon_0,t_0),t_0
\right),
\]
together with the stated deficient-regime formula for \(\rho_n\) and the strict comparison of exponents. % lean: frontier_bracket_deficient
\end{enumerate}
\end{proof}
