\begin{proof}[Proof of \cref{obj:thm:chebyshev-minimax}]
\begin{enumerate}
\item By \cref{obj:lem:continuous-chebyshev-endpoint-bound}, choose constants
\(C_{\mathrm{ep}}(q_{\max})>0\) and \(c_{\mathrm{ep}}(q_{\max})>0\) such that, for
\(x_q=2/q-1\) and
\[
\lambda(q)=x_q+\sqrt{x_q^2-1},
\]
both the continuous endpoint upper bound and the Chebyshev endpoint lower bound hold uniformly
for \(0<q\le q_{\max}\). Set
\[
C_-(q_{\max})=c_{\mathrm{ep}}(q_{\max})^2 .
\]
Now fix \(c>1\). By \cref{obj:lem:oversampled-chebyshev-lobatto-norming}, choose
\(K(c)>0\), and set
\[
C_+(c,q_{\max})=\bigl(K(c)C_{\mathrm{ep}}(q_{\max})\bigr)^2 .
\]
% lean: chebyshev_minimax

\item Fix \(\beta,k,q\) satisfying the hypotheses. The low-budget cap gives
\(q\le q_{\max}<1\), hence \(q\le 1\). Since \(k\ge c\beta\), \(c>1\), and
\(\beta\ge1\), we have \(\beta\le k\), and in particular \(k\ge1\). Therefore
\cref{obj:lem:chebyshev-schedule-admissible} gives
\[
p^{\mathrm{Ch}}(k,q)\in S_{k,q}.
\]
Also,
\[
\lambda(q)=\frac{(1+\sqrt{1-q})^2}{q},
\]
by the elementary algebraic identity for the exterior Chebyshev parameter.
% lean: chebyshev_lambda_eq_rho_div_q

\item We first prove the lower bound for an arbitrary \(p\in S_{k,q}\). Since
\(\beta\le k\), \(\beta\ge1\), and \(p\in S_{k,q}\), \cref{obj:lem:unbiased-weight-set-nonempty}
makes \(W_\beta(p)\) nonempty; the strict ordering in \(S_{k,q}\) gives distinct nodes.
Thus \cref{obj:lem:amplification-dual-norm} applies.

Consider
\[
r(u)=T_\beta\!\left(\frac{2u}{q}-1\right).
\]
This polynomial has degree at most \(\beta\). For every rollout node \(p_j\), admissibility gives
\(0\le p_j\le q\), hence \(2p_j/q-1\in[-1,1]\), so \(|r(p_j)|\le1\). Therefore \(r\) is an
admissible dual test polynomial. The endpoint lower bound from
\cref{obj:lem:continuous-chebyshev-endpoint-bound} gives
\[
T_\beta(x_q)-1\ge c_{\mathrm{ep}}(q_{\max})\lambda(q)^\beta .
\]
Since \(x_q>1\), \(T_\beta(x_q)\ge0\), and \(|T_\beta(-1)|\le1\), the reverse triangle
inequality yields
\[
T_\beta(x_q)-1
\le |T_\beta(x_q)-T_\beta(-1)|
=|r(1)-r(0)|.
\]
Hence the dual supremum is at least \(c_{\mathrm{ep}}(q_{\max})\lambda(q)^\beta\), and after
squaring,
\[
A_\beta(p)\ge c_{\mathrm{ep}}(q_{\max})^2\lambda(q)^{2\beta}
=
C_-(q_{\max})
\left(\frac{(1+\sqrt{1-q})^2}{q}\right)^{2\beta}.
\]
% lean: chebyshev_amplification_lower

\item For the Chebyshev--Lobatto schedule, apply the same dual characterization. Let \(r\) be
any degree-at-most-\(\beta\) dual polynomial with
\(\max_j |r(p^{\mathrm{Ch}}_j(k,q))|\le1\), and put
\[
R(x)=r\!\left(\frac q2(x+1)\right).
\]
Then \(\deg(R)\le\beta\), and the Lobatto nodes
\(-\cos(\pi j/k)\) are mapped exactly to \(p^{\mathrm{Ch}}_j(k,q)\). Hence the discrete
Lobatto maximum of \(|R|\) is at most \(1\). The oversampled norming inequality gives
\[
|R(x)|\le K(c)\qquad (x\in[-1,1]).
\]
Thus \(K(c)^{-1}R\) is bounded by \(1\) on \([-1,1]\). Applying the endpoint upper bound from
\cref{obj:lem:continuous-chebyshev-endpoint-bound} gives
\[
K(c)^{-1}|r(1)-r(0)|
\le C_{\mathrm{ep}}(q_{\max})\lambda(q)^\beta,
\]
because \(x_q\) maps to \(1\) and \(-1\) maps to \(0\). Taking the supremum over all such \(r\)
and squaring,
\[
A_\beta\!\left(p^{\mathrm{Ch}}(k,q)\right)
\le
\bigl(K(c)C_{\mathrm{ep}}(q_{\max})\bigr)^2\lambda(q)^{2\beta}
=
C_+(c,q_{\max})
\left(\frac{(1+\sqrt{1-q})^2}{q}\right)^{2\beta}.
\]
% lean: chebyshev_amplification_upper

\item The first two displayed inequalities are exactly the bounds just proved. For the minimax
lower bound, the lower estimate holds for every \(p\in S_{k,q}\), and the schedule class is
nonempty because \(p^{\mathrm{Ch}}(k,q)\in S_{k,q}\). Taking the infimum over \(S_{k,q}\) gives
\[
M_{\beta,k,q}
\ge
C_-(q_{\max})
\left(\frac{(1+\sqrt{1-q})^2}{q}\right)^{2\beta}.
\]
For the minimax upper bound, use the admissible schedule \(p^{\mathrm{Ch}}(k,q)\) in the
infimum and then apply the Chebyshev upper bound:
\[
M_{\beta,k,q}
\le
A_\beta\!\left(p^{\mathrm{Ch}}(k,q)\right)
\le
C_+(c,q_{\max})
\left(\frac{(1+\sqrt{1-q})^2}{q}\right)^{2\beta}.
\]
This proves the claimed two-sided minimax estimate.
% lean: minimaxAmplification_lower_of_forall
\end{enumerate}
\end{proof}
