\begin{assumptionv}{ass:static-rollout-consistency}[Static Rollout Consistency]
At each rollout measurement \(j\), the observed round mean is
\[
\bar Y_j
=
n^{-1}\sum_{i\in U_n} Y_i(Z_j).
\]
The potential outcomes \(Y_i(Z_j)\) depend on the contemporaneous assignment vector \(Z_j\), but not on earlier rollout steps.
\end{assumptionv}

\begin{assumptionv}{ass:beta-order-polynomial}[Polynomial Rollout Mean]
For every law \(P\), the rollout mean curve satisfies
\[
m_P(u)=\sum_{\ell=0}^{\beta} a_{P,\ell}u^\ell
\]
for all \(u\in[0,1]\).
\end{assumptionv}

\begin{assumptionv}{ass:round-mean-variance-envelope}[Round-Mean Variance Envelope]
For every law \(P\), every admissible schedule \(p\), and every rollout measurement \(j\),
\[
\operatorname{Var}_\pi(\bar Y_j)\le \frac{\sigma_0^2}{n}.
\]
\end{assumptionv}

\begin{assumptionv}{ass:low-budget-cap}[Low-Budget Cap]
The rollout budget and its cap satisfy
\[
0<q_{\max}\quad\text{and}\quad q\le q_{\max}<1.
\]
\end{assumptionv}

\begin{definitionv}{def:budgeted-schedule-class}[Budgeted schedule class \(S_{k,q}\)]
The budgeted schedule class is
\[
S_{k,q}
=
\left\{
p\in[0,1]^{k+1}:
p_0=0
\ \text{and}\
0<p_1<\cdots<p_k=q
\right\}.
\]
\end{definitionv}

\begin{definitionv}{def:rollout-law-class}[Rollout law class \(\mathcal P_\beta\)]
The rollout law class is
\[
\mathcal P_\beta
=
\left\{
P:
\begin{array}{l}
P \text{ satisfies \cref{obj:ass:static-rollout-consistency}, \cref{obj:ass:beta-order-polynomial}, and \cref{obj:ass:round-mean-variance-envelope}},\\
\text{and } \mathbb E_\pi[\bar Y_j]=m_P(p_j) \text{ for every } j=0,\ldots,k
\end{array}
\right\}.
\]
\end{definitionv}

\begin{definitionv}{def:unbiased-weight-set}[Unbiased weight set $W_\beta(p)$]
For a schedule \(p=(p_0,\ldots,p_k)\), the set of unbiased weights of order \(\beta\) is
\[
W_\beta(p)
=
\left\{
w\in\mathbb R^{k+1}:
\sum_{j=0}^k w_j p_j^0=0
\ \text{and}\
\sum_{j=0}^k w_j p_j^\ell=1
\ \text{for every } \ell=1,\ldots,\beta
\right\}.
\]
\end{definitionv}

\begin{definitionv}{def:chebyshev-schedule}[Chebyshev schedule \(p^{\mathrm{Ch}}(k,q)\)]
For \(j=0,\ldots,k\), the shifted Chebyshev-Lobatto node on \([0,q]\) is
\[
p^{\mathrm{Ch}}_j(k,q)
=
\frac{q}{2}\left\{1-\cos\left(\frac{\pi j}{k}\right)\right\}.
\]
The shifted Chebyshev-Lobatto schedule is
\[
p^{\mathrm{Ch}}(k,q)
=
\left(p^{\mathrm{Ch}}_0(k,q),\ldots,p^{\mathrm{Ch}}_k(k,q)\right).
\]
\end{definitionv}

\begin{definitionv}{def:amplification-criterion}[Amplification criterion \(A_\beta(p)\)]
For a schedule \(p\), the amplification criterion is
\[
A_\beta(p)
=
\inf_{w\in W_\beta(p)}
\left(\sum_{j=0}^k |w_j|\right)^2.
\]
The minimax amplification over budgeted schedules is
\[
M_{\beta,k,q}
=
\inf_{p\in S_{k,q}} A_\beta(p).
\]
\end{definitionv}

\begin{definitionv}{def:exact-nested-risk}[Exact nested risk \(R_{\mathrm{exact}}(\beta,k,q)\)]
The exact nested risk is
\[
R_{\mathrm{exact}}(\beta,k,q)
=
\inf_{p\in S_{k,q}}
\inf_{w\in W_\beta(p)}
\sup_{P\in\mathcal P_\beta}
w^\prime \Gamma_P(p) w,
\]
where \(\Gamma_P(p)\) is the design covariance matrix of the rollout round means under law \(P\) and schedule \(p\).
\end{definitionv}

\begin{lemmav}{prop:rollout-polynomial-identity}[Rollout Polynomial Identity]
For any finite assignment space $\Omega$, population size $n$, number of rollout intervals $k$, polynomial order $\beta$, contemporaneous assignment vectors $Z_j$, potential outcomes $Y_i(Z_j)$, round means $\bar Y_j$, rollout mean curve $m_P$, and polynomial coefficients $(a_{P,\ell})_{\ell\ge 0}$, suppose \cref{obj:ass:static-rollout-consistency} and \cref{obj:ass:beta-order-polynomial} hold. Then
\[
m_P(1)-m_P(0)=\sum_{\ell=1}^{\beta} a_{P,\ell}.
\]
Thus the endpoint treatment contrast satisfies
\[
\tau_P=m_P(1)-m_P(0)=\sum_{\ell=1}^{\beta} a_{P,\ell}.
\]
\end{lemmav}

\begin{theoremv}{thm:tv-envelope-design}[TV Envelope Design Bound]
Let \(n,k,\beta\) be integers, let \(q\in\mathbb R\), and let \(\sigma_0^2\in\mathbb R\). Fix a finite assignment space \(\Omega\), a finite design \(\pi\) on \(\Omega\), rollout assignment vectors \(Z_0,\ldots,Z_k\), round means \(\bar Y_0,\ldots,\bar Y_k\), a rollout mean curve \(m_P\), polynomial coefficients \(a_{P,\ell}\), and a schedule \(p=(p_0,\ldots,p_k)\). Suppose that
\begin{itemize}
\item \textbf{(Order.)} \(1\le \beta\le k\).
\item \textbf{(Envelope scale.)} \(\sigma_0^2\ge 0\).
\item \textbf{(Schedule.)} \(p\in S_{k,q}\), in the sense of \cref{obj:def:budgeted-schedule-class}.
\item \textbf{(Law class.)} The design, rollout variables, round means, mean curve, coefficients, envelope scale, and schedule satisfy the rollout law-class restrictions \(\mathcal P_\beta\) in \cref{obj:def:rollout-law-class}.
\end{itemize}
Then \(W_\beta(p)\), as defined in \cref{obj:def:unbiased-weight-set}, is nonempty. Moreover, for every \(w\in W_\beta(p)\),
\[
\mathbb E_\pi\!\left[\sum_{j=0}^k w_j\bar Y_j\right]=m_P(1)-m_P(0)
\]
and
\[
\operatorname{Var}_\pi\!\left(\sum_{j=0}^k w_j\bar Y_j\right)
\le
\frac{\sigma_0^2}{n}\left(\sum_{j=0}^k |w_j|\right)^2 .
\]
For every such \(w\), there also exists a positive semidefinite \((k+1)\times(k+1)\) matrix \(\Gamma\) such that
\[
\Gamma_{jj}\le \frac{\sigma_0^2}{n}\quad\text{for every }j=0,\ldots,k,
\]
and
\[
\sum_{i=0}^k\sum_{j=0}^k w_i\Gamma_{ij}w_j
=
\frac{\sigma_0^2}{n}\left(\sum_{j=0}^k |w_j|\right)^2 .
\]
Consequently, with \(A_\beta(\cdot)\) and \(M_{\beta,k,q}\) as in \cref{obj:def:amplification-criterion},
\[
\inf\left\{
v:\ \exists w\in W_\beta(p)\text{ with }
v=\frac{\sigma_0^2}{n}\left(\sum_{j=0}^k |w_j|\right)^2
\right\}
=
\frac{\sigma_0^2}{n}A_\beta(p),
\]
and
\[
\inf\left\{
v:\ \exists p'\in S_{k,q}\text{ with }
v=\frac{\sigma_0^2}{n}A_\beta(p')
\right\}
=
\frac{\sigma_0^2}{n}M_{\beta,k,q}.
\]
Finally, if \(p^\star\in S_{k,q}\) satisfies
\[
A_\beta(p^\star)=M_{\beta,k,q},
\]
then for every \(p'\in S_{k,q}\),
\[
\frac{\sigma_0^2}{n}A_\beta(p^\star)
\le
\frac{\sigma_0^2}{n}A_\beta(p').
\]
\end{theoremv}

\begin{lemmav}{prop:equal-spacing-benchmark}[Equal-Spacing Amplification Bound]
For every integer $\beta\ge 1$ and every $q\in(0,1]$, let
\[
p^{\mathrm{eq}}(\beta,q)_j=\frac{qj}{\beta},
\qquad j=0,\ldots,\beta .
\]
Then $p^{\mathrm{eq}}(\beta,q)_j\in[0,1]$ for every $j=0,\ldots,\beta$, and, with $A_\beta$ as in \cref{obj:def:amplification-criterion},
\[
A_\beta\!\left(p^{\mathrm{eq}}(\beta,q)\right)
\le
9\left(\frac{\beta}{q}\right)^{2\beta}.
\]
\end{lemmav}

\begin{lemmav}{prop:no-extrapolation-boundary}[Full-Budget Endpoint Rule]
For every polynomial order \(\beta\ge 1\), take \(k=\beta\) and use the equal-spacing schedule \(p^{\mathrm{eq}}(\beta,1)\). Define the weight vector \(w=(w_0,\ldots,w_\beta)\) by
\[
w_0=-1,\qquad w_\beta=1,\qquad w_j=0 \quad\text{for } 1\le j\le \beta-1 .
\]
Then \(w\in W_\beta(p^{\mathrm{eq}}(\beta,1))\), where \(W_\beta(\cdot)\) is the unbiased weight set in \cref{obj:def:unbiased-weight-set}. Consequently, for the amplification criterion in \cref{obj:def:amplification-criterion},
\[
A_\beta\!\left(p^{\mathrm{eq}}(\beta,1)\right)\le 4 .
\]
\end{lemmav}

\begin{theoremv}{thm:chebyshev-minimax}[Chebyshev Minimax Amplification]
Fix a budget cap \(q_{\max}\in(0,1)\). There exists a constant
\(C_-(q_{\max})>0\) such that, for every oversampling ratio \(c>1\), there
exists a constant \(C_+(c,q_{\max})>0\) with the following property.

For every pair of integers \(\beta,k\), every rollout budget \(q\), and every
schedule \(p\), suppose that
\begin{itemize}
\item \textbf{(Order and oversampling.)} \(\beta\ge 1\) and \(k\ge c\beta\).
\item \textbf{(Low-budget regime.)} \(q>0\) and the low-budget cap condition of
\cref{obj:ass:low-budget-cap} holds for \((q,q_{\max})\).
\item \textbf{(Admissible schedule.)} When \(p\) is used below, \(p\in S_{k,q}\)
as in \cref{obj:def:budgeted-schedule-class}.
\end{itemize}
Then, with \(A_\beta(\cdot)\) and \(M_{\beta,k,q}\) as in
\cref{obj:def:amplification-criterion} and
\(p^{\mathrm{Ch}}(k,q)\) as in \cref{obj:def:chebyshev-schedule},
\[
A_\beta(p)
\ge
C_-(q_{\max})
\left(\frac{(1+\sqrt{1-q})^2}{q}\right)^{2\beta},
\]
and
\[
A_\beta\!\left(p^{\mathrm{Ch}}(k,q)\right)
\le
C_+(c,q_{\max})
\left(\frac{(1+\sqrt{1-q})^2}{q}\right)^{2\beta}.
\]
Moreover, the minimax amplification satisfies the matching two-sided bound
\[
C_-(q_{\max})
\left(\frac{(1+\sqrt{1-q})^2}{q}\right)^{2\beta}
\le
M_{\beta,k,q}
\le
C_+(c,q_{\max})
\left(\frac{(1+\sqrt{1-q})^2}{q}\right)^{2\beta}.
\]
\end{theoremv}

\begin{remarkv}{oeq:exact-nested-minimax}[Exact nested optimality]
Fix a finite assignment space \(\Omega\) with rollout randomization \(\pi\), population size \(n\), and variance-envelope scale \(\sigma_0^2\). Fix a budget cap \(q_{\max}\) with \(0<q_{\max}<1\), let \(c>1\), let \(\beta\ge 1\), set \(k=\lceil c\beta\rceil\), and take \(q\) with \(0<q\le q_{\max}\). Under these hypotheses the shifted Chebyshev-Lobatto schedule \(p^{\mathrm{Ch}}(k,q)\) is admissible, so that
\[
p^{\mathrm{Ch}}(k,q)\in S_{k,q}.
\]
For the exact finite-population nested-rollout minimax problem
\[
R_{\mathrm{exact}}(\beta,k,q)
=
\inf_{p\in S_{k,q}}
\inf_{w\in W_\beta(p)}
\sup_{P\in\mathcal P_\beta}
w^\prime \Gamma_P(p) w,
\]
where \(\Gamma_P(p)\) is the true design covariance matrix of the monotone Bernoulli rollout and \(\mathcal P_\beta\) includes the static-rollout, \(\beta\)-polynomial, variance-envelope, and mean-curve-pinning restrictions, the open question is whether the fixed Chebyshev-Lobatto exact risk attains the nested infimum:
\[
\inf_{w\in W_\beta(p^{\mathrm{Ch}}(k,q))}
\sup_{P\in\mathcal P_\beta}
w^\prime \Gamma_P(p^{\mathrm{Ch}}(k,q)) w
=
R_{\mathrm{exact}}(\beta,k,q).
\]
This statement is only the exact-optimality question for \(p^{\mathrm{Ch}}(k,q)\); it does not assert the separate rate-feasibility bound.
\end{remarkv}

\begin{lemmav}{lem:amplification-dual-norm}[Dual Norm Amplification]
Let \(\beta,k\in\mathbb N\) and let \(p=(p_0,\ldots,p_k)\) be a real-valued schedule whose nodes are distinct. Suppose that the unbiased weight set \(W_\beta(p)\) in \cref{obj:def:unbiased-weight-set} is nonempty. Then
\[
\inf_{w\in W_\beta(p)} \sum_{j=0}^k |w_j|
=
\sup\left\{
|r(1)-r(0)|:
r\in\mathbb R[x],\ \deg(r)\le \beta,\ \max_{0\le j\le k}|r(p_j)|\le 1
\right\}.
\]
Consequently, for the amplification criterion \(A_\beta(p)\) in \cref{obj:def:amplification-criterion},
\[
A_\beta(p)
=
\left(
\sup\left\{
|r(1)-r(0)|:
r\in\mathbb R[x],\ \deg(r)\le \beta,\ \max_{0\le j\le k}|r(p_j)|\le 1
\right\}
\right)^2 .
\]
\end{lemmav}

\begin{lemmav}{lem:chebyshev-exterior-extremal}[Exterior Chebyshev Extremal Bound]
Let $\beta$ be a positive integer. For every real polynomial $P$ and every exterior point $x_0$, suppose that
\begin{itemize}
\item \textbf{(Degree bound.)} $\deg(P)\le \beta$.
\item \textbf{(Unit interval bound.)} For every $x\in[-1,1]$, $|P(x)|\le 1$.
\item \textbf{(Exterior point.)} $x_0>1$.
\end{itemize}
Then
\[
|P(x_0)|\le T_\beta(x_0),
\]
where $T_\beta$ is the Chebyshev polynomial of the first kind of degree $\beta$.
\end{lemmav}

\begin{lemmav}{lem:oversampled-chebyshev-lobatto-norming}[Oversampled Lobatto Norming]
Suppose the Ehlich--Zeller Chebyshev--Lobatto mesh inequality holds: for every pair of nonnegative integers $\beta,k$ and every real polynomial $R$ with $\deg(R)\le \beta$ and $\beta<k$,
\[
 |R(x)|
 \le
 \frac{1}{\cos(\pi\beta/(2k))}
 \max_{0\le j\le k}\left|R\!\left(-\cos\!\left(\frac{\pi j}{k}\right)\right)\right|
\]
for every $x\in[-1,1]$.  Then, for every real oversampling ratio $c$ with $c>1$, there exists a real constant $K(c)>0$ such that the following holds.  For every pair of nonnegative integers $\beta,k$ and every real polynomial $R$ satisfying
\begin{itemize}
\item \textbf{(Positive order.)} $\beta\ge 1$,
\item \textbf{(Oversampling.)} $k\ge c\beta$,
\item \textbf{(Degree bound.)} $\deg(R)\le \beta$,
\end{itemize}
one has, for every $x\in[-1,1]$,
\[
 |R(x)|
 \le
 K(c)
 \max_{0\le j\le k}\left|R\!\left(-\cos\!\left(\frac{\pi j}{k}\right)\right)\right|.
\]
\end{lemmav}

\begin{lemmav}{lem:continuous-chebyshev-endpoint-bound}[Continuous Chebyshev Endpoint Bound]
Fix \(q_{\max}\) with \(0<q_{\max}<1\). Then there exist constants \(C(q_{\max})>0\) and \(c(q_{\max})>0\) such that, for every \(\beta\in\mathbb N\) and every \(q\in\mathbb R\), if
\begin{itemize}
\item \textbf{(Degree.)} \(1\le \beta\),
\item \textbf{(Budget.)} \(0<q\le q_{\max}\),
\item \textbf{(Exterior notation.)} \(x_q=2/q-1\) and \(\lambda(q)=x_q+\sqrt{x_q^2-1}\),
\end{itemize}
then both of the following hold:
\[
\text{for every real polynomial }R\text{ with }\deg(R)\le \beta
\text{ and }\sup_{x\in[-1,1]}|R(x)|\le 1,\qquad
|R(x_q)-R(-1)|\le C(q_{\max})\,\lambda(q)^\beta,
\]
and
\[
T_\beta(x_q)-1\ge c(q_{\max})\,\lambda(q)^\beta .
\]
\end{lemmav}

\begin{lemmav}{lem:chebyshev-schedule-admissible}[Chebyshev Schedule Admissibility]
For every integer $k \ge 1$ and every $q \in (0,1]$, the shifted Chebyshev--Lobatto schedule $p^{\mathrm{Ch}}(k,q)$ of \cref{obj:def:chebyshev-schedule} is an element of the budgeted schedule class $S_{k,q}$ of \cref{obj:def:budgeted-schedule-class}. Equivalently, its coordinates all lie in $[0,1]$, it starts at $0$, is strictly increasing across the rollout measurement indices, and ends at $q$.
\end{lemmav}

\begin{lemmav}{lem:unbiased-weight-set-nonempty}[Nonempty Unbiased Weights]
Let \(q\in\mathbb R\), let \(\beta\) and \(k\) be integers, and let \(p=(p_0,\ldots,p_k)\) be a rollout schedule. Suppose that
\begin{itemize}
\item \textbf{(Order.)} \(\beta\ge 1\).
\item \textbf{(Enough nodes.)} \(k\ge \beta\).
\item \textbf{(Budgeted schedule.)} \(p\in S_{k,q}\), with \(S_{k,q}\) as in \cref{obj:def:budgeted-schedule-class}.
\end{itemize}
Then the unbiased weight set \(W_\beta(p)\) of \cref{obj:def:unbiased-weight-set} is nonempty. Equivalently, there exists a weight vector \(w=(w_0,\ldots,w_k)\) such that \(w\in W_\beta(p)\).
\end{lemmav}

\begin{lemmav}{lem:variance-envelope-sharpness}[Sharp Variance Envelope]
For any population size $n$, final rollout index $k$, finite assignment space $\Omega$ with rollout randomization $\pi$, rollout-round statistics $X_0,\ldots,X_k$, variance scale $\sigma_0^2$, and weight vector $w=(w_0,\ldots,w_k)$, suppose that
\begin{itemize}
\item \textbf{(Nonnegative scale.)} $\sigma_0^2\ge 0$.
\item \textbf{(Round variance envelope.)} \cref{obj:ass:round-mean-variance-envelope} holds for $X_0,\ldots,X_k$ with scale $\sigma_0^2/n$.
\end{itemize}
Then
\[
\operatorname{Var}_\pi\!\left(\sum_{j=0}^k w_j X_j\right)
\le
\frac{\sigma_0^2}{n}\left(\sum_{j=0}^k |w_j|\right)^2 .
\]
Moreover, there exists a positive semidefinite matrix
$\Gamma\in\mathbb R^{(k+1)\times(k+1)}$ such that
\[
\Gamma_{jj}\le \frac{\sigma_0^2}{n}\quad\text{for every }j=0,\ldots,k,
\]
and
\[
\sum_{i=0}^k\sum_{j=0}^k w_i\,\Gamma_{ij}\,w_j
=
\frac{\sigma_0^2}{n}\left(\sum_{j=0}^k |w_j|\right)^2 .
\]
\end{lemmav}

\begin{lemmav}{lem:exact-risk-envelope-upper}[Exact Risk Envelope Bound]
Let $\Omega$ be a finite assignment space equipped with rollout randomization $\pi$. For any population size $n$, any integers $k$ and $\beta$, any real numbers $q$ and $\sigma_0^2$, and any schedule $p=(p_0,\ldots,p_k)$, suppose that
\begin{itemize}
\item \textbf{(Order.)} $\beta\ge 1$.
\item \textbf{(Number of rollout intervals.)} $k\ge \beta$.
\item \textbf{(Budget.)} $0<q\le 1$.
\item \textbf{(Variance scale.)} $\sigma_0^2\ge 0$.
\item \textbf{(Budgeted schedule.)} $p\in S_{k,q}$, as in \cref{obj:def:budgeted-schedule-class}.
\end{itemize}
Then
\[
\inf_{w\in W_\beta(p)}
\sup_{P\in\mathcal P_\beta}
w^\prime \Gamma_P(p) w
\le
\frac{\sigma_0^2}{n}\,A_\beta(p),
\]
with $W_\beta(p)$ as in \cref{obj:def:unbiased-weight-set}, $\mathcal P_\beta$ as in \cref{obj:def:rollout-law-class}, and $A_\beta(p)$ as in \cref{obj:def:amplification-criterion}. Consequently,
\[
R_{\mathrm{exact}}(\beta,k,q)
\le
\frac{\sigma_0^2}{n}\,M_{\beta,k,q},
\]
where $R_{\mathrm{exact}}(\beta,k,q)$ is as in \cref{obj:def:exact-nested-risk} and $M_{\beta,k,q}$ is as in \cref{obj:def:amplification-criterion}.
\end{lemmav}

\begin{lemmav}{lem:exact-chebyshev-rate-feasible}[Feasible Chebyshev Rate]
Fix an oversampling constant \(c\) and a budget cap \(q_{\max}\) such that \(c>1\) and \(0<q_{\max}<1\). Then there exists a constant \(C_+(c,q_{\max})>0\) such that, for every population size \(n\), polynomial order \(\beta\), number of rollout intervals \(k\), finite assignment space \(\Omega\), finite design \(\pi\) on \(\Omega\), rollout budget \(q\), and variance-envelope scale \(\sigma_0^2\), the following conditions imply the stated bound:
\begin{itemize}
\item \textbf{(Order.)} \(\beta\ge 1\).
\item \textbf{(Budget.)} \(q>0\) and \(q\) satisfies the low-budget cap with \(q_{\max}\) in \cref{obj:ass:low-budget-cap}.
\item \textbf{(Variance scale.)} \(\sigma_0^2\ge 0\).
\item \textbf{(Chebyshev size.)} \(k=\lceil c\beta\rceil\).
\item \textbf{(Schedule.)} \(p=p^{\mathrm{Ch}}(k,q)\) is the shifted Chebyshev-Lobatto schedule of \cref{obj:def:chebyshev-schedule}.
\end{itemize}
Under these conditions,
\[
\inf_{w\in W_\beta(p^{\mathrm{Ch}}(k,q))}
\sup_{P\in\mathcal P_\beta}
\operatorname{Var}_\pi\!\left(\sum_{j=0}^k w_j\bar Y_j\right)
\le
\frac{\sigma_0^2}{n}\,
C_+(c,q_{\max})
\left(\frac{(1+\sqrt{1-q})^2}{q}\right)^{2\beta},
\]
where \(W_\beta(p^{\mathrm{Ch}}(k,q))\) is the unbiased weight set of \cref{obj:def:unbiased-weight-set}, and the supremum ranges over rollout laws \(P\in\mathcal P_\beta\) satisfying \cref{obj:def:rollout-law-class} at the schedule \(p^{\mathrm{Ch}}(k,q)\) with variance-envelope scale \(\sigma_0^2\).
\end{lemmav}
