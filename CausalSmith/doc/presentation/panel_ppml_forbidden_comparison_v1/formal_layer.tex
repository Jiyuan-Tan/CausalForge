\begin{assumptionv}{ass:cohort-share-limit}[Cohort share limits]
For every \(N\ge |\mathcal C|\) and every \(g\in\mathcal C\),
\[
n_{gN}>0
\quad\text{and}\quad
\frac{n_{gN}}{N}\to \pi_g\in(0,1).
\]
\end{assumptionv}

\begin{assumptionv}{ass:unit-untreated-exponential-mean}[Untreated exponential mean]
For every unit \(i=1,\ldots,N\) and period \(t=1,\ldots,T\),
\[
E_{\mathcal P_N}\!\left[Y_{it}(0)\right]
=
b_{iN}\exp(\gamma_{t0}),
\]
with the normalization \(\gamma_{10}=0\).
\end{assumptionv}

\begin{assumptionv}{ass:within-cohort-baseline-limit}[Within-cohort baseline limits]
For every \(g\in\mathcal C\), the deterministic triangular-array average
\[
\bar b_{gN}
=
n_{gN}^{-1}\sum_{i\in I_{gN}} b_{iN}
\]
satisfies
\[
\bar b_{gN}\to \bar b_g\in(0,\infty).
\]
\end{assumptionv}

\begin{assumptionv}{ass:proportional-effects}[Proportional treatment effects]
For every unit \(i=1,\ldots,N\) and period \(t=1,\ldots,T\) with \(D_{it}=1\),
\[
E_{\mathcal P_N}\!\left[Y_{it}(1)\right]
=
E_{\mathcal P_N}\!\left[Y_{it}(0)\right]\exp(\delta_{G_i t}).
\]
\end{assumptionv}

\begin{assumptionv}{ass:collapsed-design-rank}[Collapsed design rank]
The collapsed second-moment matrix
\[
\sum_{g\in\mathcal C}\sum_{t=1}^T q_{gt} r_{gt} r_{gt}'
\]
is positive definite.
\end{assumptionv}

\begin{assumptionv}{ass:multicohort-frontier-scope}[Multicohort frontier scope]
The cohort set satisfies
\[
\{2,3,4,\infty\}\subseteq \mathcal C.
\]
\end{assumptionv}

\begin{assumptionv}{ass:strict-positive-effects}[Strict Positive Effects]
For every cohort-period cell \((g,t)\) with \(D_{gt}=1\),
\[
\delta_{gt}>0.
\]
\end{assumptionv}

\begin{definitionv}{def:unit-fe-population-projection}[Unit-FE Projection \(\beta_N^\star(\delta)\)]
Define \(\beta_N^\star(\delta)\) as the last coordinate of a maximizer over \(\theta_N\in\mathbb R^{N+T}\) of
\[
L_N(\theta_N;\delta)
=
(NT)^{-1}\sum_{i=1}^N\sum_{t=1}^T
\left\{
m_{itN}(\delta)\,v_{itN}'\theta_N-\exp(v_{itN}'\theta_N)
\right\}.
\]
\end{definitionv}

\begin{definitionv}{def:collapsed-population-projection}[Collapsed Projection \(\theta^\star(\delta)\)]
Define
\[
\theta^\star(\delta)\in
\arg\max_{\theta\in\mathbb R^{|\mathcal C|+T}} L(\theta;\delta),
\]
where
\[
L(\theta;\delta)
=
\sum_{g\in\mathcal C}\sum_{t=1}^T
q_{gt}
\left\{
m_{gt}(\delta)\,r_{gt}'\theta-\exp(r_{gt}'\theta)
\right\}.
\]
The coefficient \(\beta^\star(\delta)\) is the last coordinate of \(\theta^\star(\delta)\), and the fitted collapsed mean is
\[
\mu^\star_{gt}(\delta)=\exp\{r_{gt}'\theta^\star(\delta)\}.
\]
\end{definitionv}

\begin{definitionv}{def:sample-unit-fe-ppml}[Sample PPML Coefficient \(\hat\beta_N\)]
Let \(A_N\) be the set of admissible extended maximizers of
\[
\sum_{i=1}^N\sum_{t=1}^T
\left\{
Y_{it}\,v_{itN}'\theta_N-\exp(v_{itN}'\theta_N)
\right\}
\]
over unit-effect coordinates in \(\mathbb R\cup\{-\infty\}\), time-effect coordinates in \(\mathbb R\) satisfying the period-1 normalization, and \(\beta\in\mathbb R\), with the convention \(\exp(-\infty)=0\). Admissibility means that the period-1 time effect is normalized to zero and that a unit-effect coordinate may equal \(-\infty\) only for a unit whose whole observed outcome path satisfies \(Y_{it}=0\) for every \(t\); for such a unit the extended objective contribution is defined to be zero. If \(A_N\) contains an element whose squared Euclidean norm over the finite time-effect and \(\beta\) coordinates is no larger than that of every element of \(A_N\), then \(\hat\beta_N\) is the \(\beta\)-coordinate of such a selected element. If no such minimum-finite-coordinate-norm admissible extended maximizer exists, set \(\hat\beta_N=0\).
\end{definitionv}

\begin{definitionv}{def:weighted-fwl-residual}[Weighted Residual \(\widetilde W_{gt}(\delta)\)]
Define
\[
\rho^\star(\delta)
\in
\arg\min_{\rho\in\mathbb R^{|\mathcal C|+T-1}}
\sum_{g\in\mathcal C}\sum_{t=1}^T
q_{gt}\,\mu^\star_{gt}(\delta)\{D_{gt}-X_{gt}'\rho\}^2.
\]
The weighted Frisch--Waugh--Lovell residualized treatment is
\[
\widetilde W_{gt}(\delta)=D_{gt}-X_{gt}'\rho^\star(\delta).
\]
\end{definitionv}

\begin{definitionv}{def:four-cohort-witness}[Four-cohort witness $W_4$]
Let $W_4$ be the cofinal triangular subsequence indexed by $N=4m$, $m=1,2,\ldots$, with $T=4$ and
\[
\mathcal C=\{2,3,4,\infty\}.
\]
For each such $N$,
\[
n_{2N}=n_{3N}=n_{4N}=n_{\infty N}=m,
\]
$b_{iN}=1$ for every $i$, and $\gamma_{t0}=0$ for every $t$. The treatment effects satisfy
\[
\delta_{gt}=\log(101/100)
\]
for every treated cell $(g,t)\ne(2,4)$, while
\[
\delta_{2,4}=\log(4).
\]
\end{definitionv}

\begin{definitionv}{def:frontier-elimination-handle}[Frontier-elimination polynomial $\Phi(\delta)$]
Define
\[
\Phi(\delta)
=
M A
-
\sum_{g,t} D_{gt} R_g C_t,
\]
where
\[
h_{gt}
=
\pi_g B_{gt}\exp(D_{gt}\delta_{gt}),\qquad
R_g=\sum_t h_{gt},\qquad
C_t=\sum_g h_{gt},
\]
and
\[
M=\sum_g R_g,\qquad
A=\sum_{g,t}D_{gt}h_{gt}.
\]
The relation between $\Phi(\delta)$ and $\beta^\star(\delta)$ is stated separately.
\end{definitionv}

\begin{lemmav}{lem:unit-fe-collapse}[Unit FE Collapse]
Let $T$ be a number of periods and let $\mathcal C$ be a finite cohort set. Suppose:
\begin{itemize}
\item \textbf{(Horizon.)} $T\ge 4$.
\item \textbf{(Cohort support.)} The never-treated cohort belongs to $\mathcal C$, and every finite cohort in $\mathcal C$ is dated after the first period.
\item \textbf{(Array support.)} For every $N$ and every unit $i$, the cohort label $G_i$ belongs to $\mathcal C$.
\item \textbf{(Positive primitives.)} The unit baselines $b_{iN}$ and the limiting within-cohort baselines $\bar b_g$ are strictly positive, and each limiting cohort share satisfies $\pi_g\in(0,1)$.
\item \textbf{(Share limits.)} \cref{obj:ass:cohort-share-limit} holds for $G_i$ and $\pi_g$.
\item \textbf{(Untreated means.)} \cref{obj:ass:unit-untreated-exponential-mean} holds for $Y_{it}(d)$, $\mathcal P_N$, $b_{iN}$, and $\gamma_{t0}$.
\item \textbf{(Baseline limits.)} \cref{obj:ass:within-cohort-baseline-limit} holds for $G_i$, $b_{iN}$, and $\bar b_g$.
\item \textbf{(Proportional effects.)} \cref{obj:ass:proportional-effects} holds for $Y_{it}(d)$, $\mathcal P_N$, $G_i$, and $\delta_{gt}$.
\item \textbf{(Collapsed rank.)} \cref{obj:ass:collapsed-design-rank} holds for $\mathcal C$ and $\pi_g$.
\end{itemize}
Then, for every $N$ with $|\mathcal C|\le N$, the finite-array unit-FE population criterion $L_N(\theta_N;\delta)$ of \cref{obj:def:unit-fe-population-projection} has a unique global maximizer, the collapsed finite-array criterion $L_{Nc}(\theta;\delta)$ has a unique global maximizer, and
\[
\beta_N^\star(\delta)
=
\text{the treatment coordinate of the unique maximizer of }L_{Nc}(\theta;\delta).
\]
Moreover,
\[
\beta_N^\star(\delta)\longrightarrow \beta^\star(\delta)
\qquad\text{as }N\to\infty .
\]
Finally, for every $N$ with $|\mathcal C|\le N$ and every alternative positive baseline array $b'_{iN}$, if
\[
\bar b'_{gN}=\bar b_{gN}
\qquad\text{for every }g\in\mathcal C,
\]
then replacing $b_{iN}$ by $b'_{iN}$ leaves the finite-array unit-FE coefficient unchanged:
\[
\beta_N^\star(\delta;b')=\beta_N^\star(\delta;b).
\]
\end{lemmav}

\begin{lemmav}{lem:pseudo-true-ppml-projection}[Unique PPML Projection]
Fix a number of periods \(T\), a finite cohort set \(\mathcal C\), cohort shares \(\pi_g\in(0,1)\), positive limiting baseline means \(\bar b_g>0\), time components \(\gamma_{t0}\), and proportional-effect log multipliers \(\delta_{gt}\). Suppose the collapsed design rank condition in \cref{obj:ass:collapsed-design-rank} holds for \((T,\mathcal C,\pi)\). Let \(\theta^\star(\delta)\), \(L(\theta;\delta)\), \(m_{gt}(\delta)\), and \(\mu^\star_{gt}(\delta)\) be as in \cref{obj:def:collapsed-population-projection}. Then \(\theta^\star(\delta)\) is the unique global maximizer of \(L(\theta;\delta)\). Moreover, for every collapsed nuisance regressor index \(j\),
\[
\sum_{g\in\mathcal C}\sum_{t=1}^T
q_{gt} X_{gt,j}\{m_{gt}(\delta)-\mu^\star_{gt}(\delta)\}=0,
\]
and the treatment score also vanishes:
\[
\sum_{g\in\mathcal C}\sum_{t=1}^T
q_{gt} D_{gt}\{m_{gt}(\delta)-\mu^\star_{gt}(\delta)\}=0.
\]
\end{lemmav}

\begin{theoremv}{thm:sharp-ppml-forbidden-sign}[Sharp PPML Sign]
Let \(T\) be positive and let \(\mathcal C\) be a nonempty finite cohort set. Fix cohort shares \(\pi_g\in(0,1)\), baseline levels \(\bar b_g>0\), time effects \(\gamma_t\), and a proportional-effect array \(\delta\). Assume:
\begin{itemize}
\item \textbf{(Rank.)} The collapsed design satisfies \cref{obj:ass:collapsed-design-rank}.
\item \textbf{(Cell.)} The cohort \(k\) belongs to \(\mathcal C\), the period \(s\in\{1,\ldots,T\}\), and the cohort-period cell is treated: \(D_{ks}=1\).
\end{itemize}
Write
\[
\widetilde W_{ks}(\delta)
\]
for the pseudo-true \(q_{gt}\mu^\star_{gt}(\delta)\)-weighted FWL residual of the treatment indicator in cell \((k,s)\), and define
\[
\mathcal E(\delta)
=
\sum_{g\in\mathcal C}\sum_{t=1}^T
q_{gt}\,\mu^\star_{gt}(\delta)\,
\widetilde W_{gt}(\delta)^2 .
\]
Then \(\mathcal E(\delta)>0\), and the one-coordinate path
\[
x\mapsto \beta^\star\!\bigl(\delta \text{ with } \delta_{ks}\text{ replaced by }x\bigr)
\]
is differentiable at \(x=\delta_{ks}\) with derivative
\[
\frac{
q_{ks}\,B_{ks}\exp(\delta_{ks})\,\widetilde W_{ks}(\delta)
}{
\mathcal E(\delta)
}.
\]
Consequently,
\[
\frac{\partial \beta^\star(\delta)}{\partial \delta_{ks}}<0
\iff
\widetilde W_{ks}(\delta)<0,\qquad
\frac{\partial \beta^\star(\delta)}{\partial \delta_{ks}}=0
\iff
\widetilde W_{ks}(\delta)=0,
\]
and
\[
\frac{\partial \beta^\star(\delta)}{\partial \delta_{ks}}>0
\iff
\widetilde W_{ks}(\delta)>0.
\]
\end{theoremv}

\begin{theoremv}{prop:four-cohort-sign-reversal}[Four-Cohort Sign Reversal]
The four-cohort witness \(W_4\) is constructed with cohort support
\(\mathcal C=\{2,3,4,\infty\}\), equal shares \(\pi_g=1/4\), limiting baselines
\(\bar b_g=1\), untreated time component \(\gamma_{t0}=0\), and proportional-effect
log multipliers
\[
\delta_{gt}=
\begin{cases}
\log 4, & (g,t)=(2,4),\\
\log(101/100), & D_{gt}=1 \text{ and } (g,t)\neq(2,4),\\
0, & D_{gt}=0 .
\end{cases}
\]
Then:
\begin{itemize}
\item \textbf{(Region membership.)} The primitive \(W_4\) belongs to the sign-reversal region \(\mathcal R_4\) of \cref{obj:def:global-sign-reversal-region}.
\item \textbf{(Positive effects.)} Every treated supported cell has a strictly positive proportional-effect log multiplier: if \(g\in\mathcal C\), \(t\in\{1,2,3,4\}\), and \(D_{gt}=1\), then \(\delta_{gt}>0\).
\item \textbf{(Unique largest treated effect.)} The treated cell \((2,4)\) has the strictly largest proportional-effect log multiplier: if \(g\in\mathcal C\), \(t\in\{1,2,3,4\}\), \(D_{gt}=1\), and \((g,t)\neq(2,4)\), then \(\delta_{gt}<\delta_{2,4}\).
\item \textbf{(Weighted FWL residuals.)} At the no-effect vector, the weighted FWL residual \(\widetilde W_{gt}\) of \cref{obj:def:weighted-fwl-residual} equals
\[
\frac{1}{8}
\begin{pmatrix}
-3 & 3 & 1 & -1\\
-1 & -3 & 3 & 1\\
1 & -1 & -3 & 3\\
3 & 1 & -1 & -3
\end{pmatrix},
\]
with rows \(g=2,3,4,\infty\) and columns \(t=1,2,3,4\). In particular, \(\widetilde W_{2,4}=-1/8\).
\item \textbf{(Negative late-cell derivative at zero.)} Holding all other effects at zero,
\[
\left.\frac{d}{dx}\,
\beta^\star\!\left(\delta_{2,4}=x,\ \delta_{gt}=0\text{ for }(g,t)\neq(2,4)\right)
\right|_{x=0}
=-\frac{1}{10}.
\]
\item \textbf{(Local negative derivative under positive effects.)} There exists \(\varepsilon>0\) such that, for every effect vector \(\delta'\) satisfying \(|\delta'_{gt}|<\varepsilon\) for all \(g\in\mathcal C\) and \(t\in\{1,2,3,4\}\), if \(\delta'\) has strictly positive treated effects in the sense of \cref{obj:ass:strict-positive-effects}, then
\[
\left.
\frac{d}{dx}\,
\beta^\star\!\left(\delta' \text{ with } \delta'_{2,4}\text{ replaced by }x\right)
\right|_{x=\delta'_{2,4}}
<0 .
\]
\end{itemize}
\end{theoremv}

\begin{theoremv}{prop:homogeneous-effect-reduction}[Homogeneous Effect Reduction]
Fix $T$, a finite cohort set $\mathcal C$, sampling laws $\{\mathcal P_N\}_N$, potential outcomes $Y_{it}(d)$, positive unit baselines $b_{iN}$, cohort shares $\pi_g\in(0,1)$, positive limiting cohort baselines $\bar b_g$, untreated time components $\gamma_{t0}$, cohort-time log effects $\delta_{gt}$, and a scalar $\delta_0$. Suppose that
\begin{itemize}
\item \textbf{(Untreated means.)} \cref{obj:ass:unit-untreated-exponential-mean} holds.
\item \textbf{(Collapsed rank.)} \cref{obj:ass:collapsed-design-rank} holds for $\mathcal C$ and $\{\pi_g\}_g$.
\item \textbf{(Common treated-cell effect.)} For every $g\in\mathcal C$ and every period $t$, if $D_{gt}=1$, then $\delta_{gt}=\delta_0$.
\end{itemize}
Then
\[
\beta^\star(\delta)=\delta_0,
\]
and, for every $g\in\mathcal C$ and every period $t$,
\[
\mu^\star_{gt}(\delta)
=
B_{gt}\exp(D_{gt}\delta_0).
\]
\end{theoremv}

\begin{definitionv}{def:global-sign-reversal-region}[Sign-Reversal Region \(\mathcal R_T\)]
Define \(\mathcal R_T\) as the set of all triples
\[
\left((\pi_g)_{g\in\mathcal C},(B_{gt})_{g\in\mathcal C,\ 1\le t\le T},(\delta_{gt})_{(g,t):D_{gt}=1}\right)
\]
with \(T\ge 4\), indexed by a cohort support \(\mathcal C\) that contains \(\{2,3,4,\infty\}\) and contains no first-period finite cohort, such that
\[
0<\pi_g<1\quad(g\in\mathcal C),\qquad \sum_{g\in\mathcal C}\pi_g=1,
\]
there exist \(\bar b_g>0\) for \(g\in\mathcal C\) with \(B_{gt}=\bar b_g\exp(\gamma_{t0})\) for every supported cell \((g,t)\), \(\delta_{gt}>0\) for every treated supported cell \((g,t)\) with \(D_{gt}=1\), the collapsed design has full weighted rank in the sense that
\[
\forall a\ne 0,\qquad
0<\sum_{g\in\mathcal C}\sum_{t=1}^T \frac{\pi_g}{T}\,(r_{gt}'a)^2,
\]
and
\[
\beta^\star(\delta)<0.
\]
\end{definitionv}

\begin{theoremv}{thm:primitive-global-frontier}[Primitive Sign Frontier]
Fix a panel length \(T\), a finite cohort support \(\mathcal C\), triangular-array sampling laws \(\mathcal P_N\), potential outcomes \(Y_{it}(d)\), deterministic cohort labels \(G_i\), positive unit baselines \(b_{iN}\), limiting shares \(\pi_g\in(0,1)\), positive limiting cohort baselines \(\bar b_g\), untreated time effects \(\gamma_{t0}\), and log proportional effects \(\delta_{gt}\). Suppose:
\begin{itemize}
\item \textbf{(Horizon and support.)} \(T>0\), \(T\ge 4\), the never-treated cohort belongs to \(\mathcal C\), and every finite supported cohort has adoption date different from the first period.
\item \textbf{(Array support.)} For every \(N\) and every unit \(i\), \(G_i\in\mathcal C\).
\item \textbf{(Share limits.)} \cref{obj:ass:cohort-share-limit} holds for \((\mathcal C,G_i,\pi_g)\).
\item \textbf{(Untreated means.)} \cref{obj:ass:unit-untreated-exponential-mean} holds for \((\mathcal P_N,Y_{it}(d),b_{iN},\gamma_{t0})\).
\item \textbf{(Baseline limits.)} \cref{obj:ass:within-cohort-baseline-limit} holds for \((\mathcal C,G_i,b_{iN},\bar b_g)\).
\item \textbf{(Proportional effects.)} \cref{obj:ass:proportional-effects} holds for \((\mathcal P_N,Y_{it}(d),G_i,\delta_{gt})\).
\item \textbf{(Rank.)} \cref{obj:ass:collapsed-design-rank} holds for \((\mathcal C,\pi_g)\).
\item \textbf{(Frontier scope.)} \cref{obj:ass:multicohort-frontier-scope} holds for \((T,\mathcal C)\).
\item \textbf{(Positive effects.)} \cref{obj:ass:strict-positive-effects} holds for \((\mathcal C,\delta_{gt})\).
\end{itemize}
Define
\[
h_{gt}
=
\pi_g\,\bar b_g\exp(\gamma_{t0})\exp(D_{gt}\delta_{gt}),
\qquad
R_g=\sum_{t=1}^T h_{gt},
\qquad
C_t=\sum_{g\in\mathcal C} h_{gt},
\]
\[
M=\sum_{g\in\mathcal C}R_g,
\qquad
A=\sum_{g\in\mathcal C}\sum_{t=1}^T D_{gt}h_{gt},
\qquad
\Phi
=
MA-\sum_{g\in\mathcal C}\sum_{t=1}^T D_{gt}R_gC_t .
\]
Then
\[
\beta^\star(\delta)<0\iff \Phi<0,\qquad
\beta^\star(\delta)=0\iff \Phi=0,\qquad
0<\beta^\star(\delta)\iff 0<\Phi .
\]
Moreover, the primitive tuple induced by \((\pi_g,\bar b_g,\gamma_{t0},\delta_{gt})\) belongs to \(\mathcal R_T\) if and only if \(\Phi<0\).

For every \(x>1\) and \(y>1\), in the four-period support \(\{2,3,4,\infty\}\) with equal shares, unit limiting baselines, zero untreated time effects, multiplier \(x\) on treated cells other than \((2,4)\), and multiplier \(y\) at \((2,4)\),
\[
\Phi=\frac{5x^2+12x-2y-15}{16}.
\]
Also,
\[
\frac{5(101/100)^2+12(101/100)-15}{2}=\frac{4441}{4000},
\]
and the explicit witness \(W_4\) has
\[
\Phi=-\frac{11559}{32000}.
\]

Let
\[
m_{gt}=\bar b_g\exp(\gamma_{t0})\exp(D_{gt}\delta_{gt}),
\qquad
H=\{(g,t):g\in\mathcal C,\ D_{gt}=1\}.
\]
For every \((g,t)\in H\),
\[
B^{obs}_{gt}=B_{gt},\qquad
\tau_{gt}=\exp(\delta_{gt})-1,\qquad
\omega_{gt}>0.
\]
The counterfactual-share weights sum to one:
\[
\sum_{(g,t)\in H}\omega_{gt}=1.
\]
The counterfactual-share PTT satisfies
\[
PTT
=
\frac{\sum_{(g,t)\in H} q_{gt}m_{gt}}
{\sum_{(g,t)\in H} q_{gt}B^{obs}_{gt}}
-1.
\]
Finally,
\[
\Phi<0
\quad\Longrightarrow\quad
\beta^\star(\delta)<0<PTT .
\]
\end{theoremv}
