\begin{proof}[Proof of \cref{obj:thm:robust-corner-exactness}]
\begin{enumerate}
\item We first prove the finite-weight equivalence.  Write
\[
q=2(m-1),\qquad
c_x=q\left((a+b)+\frac r{a+b}\right),\qquad
c_y=2b+\frac r{2b},\qquad
c_z=2m .
\]
Suppose that some \(\kappa\ge 0\) makes \(I_n\) a relaxed minimizer over
\(\mathcal E_m^{\mathrm{blk}}\).  For an arbitrary
\((x,y,z_{\mathrm{sp}})\in T_m\), set
\[
u=1-x,\qquad v=\frac{z_{\mathrm{sp}}-y}{2m}.
\]
Using \cref{obj:lem:block-spectral-coordinates}, this inverse change of variables sends
\((x,y,z_{\mathrm{sp}})\) to a feasible block matrix \(X(u,v)\), and the matrix
objective equals the reduced objective at \((x,y,z_{\mathrm{sp}})\).  Since
\(I_n=X(0,0)\) corresponds to \((1,1,1)\), the assumed matrix minimality says that
\((1,1,1)\) minimizes
\[
c_xx+c_yy+c_zz_{\mathrm{sp}}
+\kappa\sqrt{q x^2+y^2+z_{\mathrm{sp}}^2}
\]
on \(T_m\).  The certificate-necessity part of
\cref{obj:lem:frobenius-center-certificate} therefore gives
\[
\frac{c_x}{q}=c_y,\qquad c_y=c_z .
\]
The first equality is
\[
a+b+\frac r{a+b}=2b+\frac r{2b},
\]
hence
\[
(a-b)\left(1-\frac{r}{2b(a+b)}\right)=0.
\]
By two-block homophily, \(a>b>0\), so
\[
r=2b(a+b).
\]
Substituting this into \(c_y=c_z\) gives
\[
2b+\frac{2b(a+b)}{2b}=2m,
\]
that is \(a+3b=2m\).  Thus the affine-balanced locus holds.  Conversely, if
\[
a+3b=2m,\qquad r=2b(a+b),
\]
then the same algebra gives \(c_x/q=c_y=c_z\).  Taking \(\kappa=1\), the strict
optimality part of \cref{obj:lem:frobenius-center-certificate}, transported through
\cref{obj:lem:block-spectral-coordinates}, gives \(I_n\in\mathcal E_m^{\mathrm{blk}}\)
and strict optimality against every \(X\ne I_n\).  For \(X=I_n\) the desired inequality
is equality, while for \(X\ne I_n\) it follows from strict optimality.  Hence a finite
nonnegative robustness weight exists exactly on the stated locus. % lean: robust_corner_exactness

\item On the affine-balanced locus, let \(\kappa>0\).  As above,
\[
\frac{c_x}{q}=c_y=c_z .
\]
The point \(I_n=X(0,0)\) has reduced coordinates \((1,1,1)\) by
\cref{obj:lem:block-spectral-coordinates}.  The Frobenius-center certificate
\cref{obj:lem:frobenius-center-certificate} gives strict reduced optimality of
\((1,1,1)\) on \(T_m\).  Translating back by
\cref{obj:lem:block-spectral-coordinates}, \(I_n\in\mathcal E_m^{\mathrm{blk}}\), and
for any \(X=X(u,v)\in\mathcal E_m^{\mathrm{blk}}\) with \(X\ne I_n\), the associated
reduced coordinates are not \((1,1,1)\); otherwise \(u=0\) and \(v=0\), hence
\(X=I_n\).  Therefore
\[
F_{r,\kappa}(I_n)<F_{r,\kappa}(X).
\] % lean: identity_strict_relaxed_min_of_locus

\item The iid law is invariant under sign flip, within-block permutations, and block
swap, so
\[
P_{\mathrm{iid}}\in\mathcal P_m^{\mathrm{sym}}.
\]
Its second moments are diagonal: the diagonal entries equal \(1\), and for \(i\ne j\)
the involution that flips the \(i\)-th sign pairs every assignment with one contributing
the negative product.  Thus all off-diagonal moments vanish and
\[
X(P_{\mathrm{iid}})=I_n .
\] % lean: iidDesign_mem_blockExchangeable, iidDesign_secondMoment

\item Now assume the affine-balanced locus fails.  If for some \(\kappa>0\) the matrix
\(I_n\) were a relaxed minimizer, then the same center-coefficient necessity argument
from the first step would give
\[
\frac{c_x}{q}=c_y,\qquad c_y=c_z,
\]
and the same algebra would force
\[
a+3b=2m,\qquad r=2b(a+b),
\]
contradicting the assumption.  Hence \(I_n\) is not a relaxed minimizer at any positive
finite robustness weight off the locus. % lean: robust_corner_exactness

\item Finally, let \((X_\kappa)_{\kappa\in\mathbb R}\) be any family of relaxed
minimizers for positive \(\kappa\).  Since \(r\ge0\) and \(a>b>0\), the reduced linear
coefficients \(c_x,c_y,c_z\) are nonnegative.  Comparing each minimizer with the feasible
point \(I_n\) gives
\[
F_{r,\kappa}(X_\kappa)\le F_{r,\kappa}(I_n).
\]
Writing \(X_\kappa=X(u_\kappa,v_\kappa)\), the reduced squared norm decomposes as
\[
q(1-u_\kappa)^2+
\bigl(1+(m-1)u_\kappa-mv_\kappa\bigr)^2+
\bigl(1+(m-1)u_\kappa+mv_\kappa\bigr)^2
=
q+2+N_{\mathrm{same}}u_\kappa^2+N_{\mathrm{cross}}v_\kappa^2,
\]
with positive constants \(N_{\mathrm{same}}\) and \(N_{\mathrm{cross}}\).  If a
same-block off-diagonal entry stayed at distance at least \(\varepsilon>0\) from zero
for arbitrarily large \(\kappa\), then \(|u_\kappa|\ge\varepsilon\), so the robustness
norm would exceed its value at \(I_n\) by a fixed amount \(\delta>0\).  The nonnegative
linear part then yields
\[
\kappa\bigl(\|I_n\|+\delta\bigr)\le F_{r,\kappa}(X_\kappa)
\le F_{r,\kappa}(I_n)=L+\kappa\|I_n\|,
\]
forcing \(\kappa\delta\le L\), impossible for all sufficiently large \(\kappa\).  The
same argument with \(N_{\mathrm{cross}}\) shows that the across-block parameter
\(v_\kappa\to0\).  Diagonal entries are identically \(1\), while off-diagonal entries
are either \(u_\kappa\) or \(v_\kappa\).  Hence every entry of \(X_\kappa\) converges to
the corresponding entry of \(I_n\) as \(\kappa\to\infty\). % lean: robust_minimizers_tendsto_identity_entries
\end{enumerate}
\end{proof}
