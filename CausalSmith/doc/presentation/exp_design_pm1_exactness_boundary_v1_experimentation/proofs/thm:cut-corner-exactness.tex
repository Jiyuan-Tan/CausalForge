\begin{proof}[Proof of \cref{obj:thm:cut-corner-exactness}]
\begin{enumerate}
\item
By the two-block homophily assumption, \(m\ge2\), \(0<b<a\). Hence
\[
q=2(m-1)>0,\qquad a-b>0,\qquad a+b>0,\qquad 2b>0 .
\]
Set
\[
A=2b(a+b)\left(1-\frac{\kappa}{a-b}\right),
\qquad
B=2b(2m-2b-\kappa).
\]
From \(r<\max\{0,\min(A,B)\}\) and \(0\le r\), we first get
\[
r<\min(A,B).
\]
Indeed, if \(\min(A,B)\le0\), then the outer maximum is \(0\), contradicting
\(0\le r<0\). Otherwise the outer maximum is \(\min(A,B)\). Thus
\[
r<A,\qquad r<B .
\]
% lean: cut_corner_exactness

\item
We now convert these two inequalities into the coefficient separations needed at the cut vertex.
From \(r<A\), division by the positive number \(2b(a+b)\) gives
\[
\frac{r}{2b(a+b)}
<
1-\frac{\kappa}{a-b}.
\]
Multiplying by \(a-b>0\) and rearranging gives
\[
\kappa
<
(a-b)\left(1-\frac{r}{2b(a+b)}\right).
\]
Using
\[
(a+b)+\frac{r}{a+b}
-\left(2b+\frac{r}{2b}+\kappa\right)
=
(a-b)\left(1-\frac{r}{2b(a+b)}\right)-\kappa
\]
and \(c_x/q=(a+b)+r/(a+b)\), \(c_y=2b+r/(2b)\), we obtain
\[
\frac{c_x}{q}>c_y+\kappa .
\]
Similarly, from \(r<B\), division by \(2b>0\) gives
\[
\frac{r}{2b}<2m-2b-\kappa,
\]
and hence
\[
2b+\frac{r}{2b}+\kappa<2m.
\]
Since \(c_z=2m\), this is
\[
c_z>c_y+\kappa .
\]
% lean: cut_corner_exactness

\item
The hypotheses of the cut-vertex certificate \cref{obj:lem:cut-vertex-certificate} are therefore satisfied: \(q>0\), \(\kappa\ge0\), and the two coefficient separations above hold. Hence
\[
(0,2m,0)\in T_m,
\]
and every other point of \(T_m\) has strictly larger reduced objective value.

By the spectral-coordinate \cref{obj:lem:block-spectral-coordinates}, the cut covariance is
\[
X_{\mathrm{cut}}=X(1,-1),
\]
and its spectral coordinates are
\[
(1-1,\;1+(m-1)\cdot1-m(-1),\;1+(m-1)\cdot1+m(-1))
=(0,2m,0).
\]
Thus \(X_{\mathrm{cut}}\in\mathcal E_m^{\mathrm{blk}}\), and
\[
F_{r,\kappa}(X_{\mathrm{cut}})
=
\phi_{r,\kappa}(0,2m,0).
\]
% lean: cut_vertex_certificate

\item
Let \(X\in\mathcal E_m^{\mathrm{blk}}\) and suppose \(X\ne X_{\mathrm{cut}}\). Write \(X=X(u,v)\). By \cref{obj:lem:block-spectral-coordinates}, the associated coordinates
\[
x=1-u,\qquad
y=1+(m-1)u-mv,\qquad
z_{\mathrm{sp}}=1+(m-1)u+mv
\]
belong to \(T_m\), and
\[
F_{r,\kappa}(X)=\phi_{r,\kappa}(x,y,z_{\mathrm{sp}}).
\]
These coordinates cannot equal \((0,2m,0)\): if \(x=0\), then \(u=1\), and if also
\(z_{\mathrm{sp}}=0\), then
\[
m(1+v)=0,
\]
so \(v=-1\) since \(m>0\), giving \(X=X_{\mathrm{cut}}\), a contradiction. The strict part of \cref{obj:lem:cut-vertex-certificate} therefore gives
\[
F_{r,\kappa}(X_{\mathrm{cut}})
=
\phi_{r,\kappa}(0,2m,0)
<
\phi_{r,\kappa}(x,y,z_{\mathrm{sp}})
=
F_{r,\kappa}(X).
\]
Thus \(X_{\mathrm{cut}}\) is the unique minimizer over \(\mathcal E_m^{\mathrm{blk}}\).
% lean: block_spectral_coordinates

\item
The cut design is the two-point cut law, so
\[
P_{\mathrm{cut}}\in\mathcal P_m^{\mathrm{sym}},
\qquad
X(P_{\mathrm{cut}})=X_{\mathrm{cut}}.
\]
Consequently \(X_{\mathrm{cut}}\in\mathcal C_m^{\pm}\). Moreover, every
\(X\in\mathcal C_m^{\pm}\) is obtained from a block-exchangeable design. Its second moment is block-symmetric, say \(X=X(u,v)\), and the forward reduced-slice characterization places its spectral coordinates in the same reduced triangle \(T_m\). By \cref{obj:lem:block-spectral-coordinates}, this implies
\[
\mathcal C_m^{\pm}\subseteq \mathcal E_m^{\mathrm{blk}}.
\]
Therefore, if \(X\in\mathcal C_m^{\pm}\) and \(X\ne X_{\mathrm{cut}}\), the strict relaxed optimality already proved yields
\[
F_{r,\kappa}(X_{\mathrm{cut}})<F_{r,\kappa}(X).
\]
Thus \(X_{\mathrm{cut}}\) is also the unique minimizer over \(\mathcal C_m^{\pm}\).
% lean: cutDesign_eq_cutVDesign

\item
It remains to identify the gap. From the strict inequalities above, with equality allowed only at
\(X_{\mathrm{cut}}\), we have the non-strict minimizing inequalities
\[
F_{r,\kappa}(X_{\mathrm{cut}})\le F_{r,\kappa}(X)
\quad\text{for all }X\in\mathcal E_m^{\mathrm{blk}},
\]
and
\[
F_{r,\kappa}(X_{\mathrm{cut}})\le F_{r,\kappa}(X)
\quad\text{for all }X\in\mathcal C_m^{\pm}.
\]
Since \(X_{\mathrm{cut}}\) belongs to both feasible classes, the infimum of each corresponding objective image is exactly \(F_{r,\kappa}(X_{\mathrm{cut}})\). Hence, by the definition of the implementability gap \cref{obj:def:implementability-gap},
\[
\Delta_m^{\pm}(r,\kappa)
=
F_{r,\kappa}(X_{\mathrm{cut}})
-
F_{r,\kappa}(X_{\mathrm{cut}})
=0.
\]
This proves all asserted conclusions.
% lean: sInf_image_eq_of_minimizer
\end{enumerate}
\end{proof}
