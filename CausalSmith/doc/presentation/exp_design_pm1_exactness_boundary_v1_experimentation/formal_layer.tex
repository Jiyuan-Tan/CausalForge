\begin{assumptionv}{ass:two-block-homophily}[Two-Block Homophily]
The block size and edge intensities satisfy
\[
m\ge 2
\qquad\text{and}\qquad
a>b>0.
\]
\end{assumptionv}

\begin{assumptionv}{ass:balanced-sign-design}[Balanced Sign Symmetry]
The assignment law is sign-symmetric: for every \(z\in\{-1,1\}^n\),
\[
P(Z=z)=P(Z=-z).
\]
\end{assumptionv}

\begin{assumptionv}{ass:low-scale-two-block}[Low-Scale Two-Block Regime]
The two-block edge intensities satisfy
\[
a+3b<2m.
\]
\end{assumptionv}

\begin{assumptionv}{ass:odd-community-size}[Odd Community Size]
The block size \(m\) is odd.
\end{assumptionv}

\begin{definitionv}{def:two-block-graph}[Two-Block Graph Weights \(W_{ij}\)]
The vertex set is the disjoint union \(A_m\cup B_m\), where
\[
|A_m|=|B_m|=m.
\]
The graph weights \(W_{ij}\) are given as follows: if \(i\ne j\) and \(i\) and \(j\) lie in the same block, then
\[
W_{ij}=\frac{a}{m};
\]
if \(i\) and \(j\) lie in opposite blocks, then
\[
W_{ij}=\frac{b}{m};
\]
and for every vertex \(i\),
\[
W_{ii}=0.
\]
\end{definitionv}

\begin{definitionv}{def:block-elliptope}[Block Elliptope \(\mathcal E_m^{\mathrm{blk}}\)]
Under \cref{obj:ass:two-block-homophily}, the block elliptope is
\[
\mathcal E_m^{\mathrm{blk}}
=
\left\{
X(u,v):
1-u\ge 0,\ 
1+(m-1)u-mv\ge 0,\ 
1+(m-1)u+mv\ge 0
\right\}.
\]
\end{definitionv}

\begin{definitionv}{def:balanced-design-class}[Balanced sign class $\mathcal P_m^{\mathrm{bal}}$]
Under \cref{obj:ass:balanced-sign-design}, the balanced sign class is
\[
\mathcal P_m^{\mathrm{bal}}
=
\left\{
P \text{ on } \{-1,1\}^n :
P(Z=z)=P(Z=-z)\ \text{for every } z\in\{-1,1\}^n
\right\}.
\]
\end{definitionv}

\begin{definitionv}{def:block-exchangeable-design-class}[Block-exchangeable class $\mathcal P_m^{\mathrm{sym}}$]
The block-exchangeable class is the set of all assignment laws
\[
P\in \mathcal P_m^{\mathrm{bal}}
\]
that are invariant under permutations within $A_m$, invariant under permutations within $B_m$, and invariant under swapping the two equal communities.
\end{definitionv}

\begin{definitionv}{def:implementable-covariance-class}[Implementable covariance class $\mathcal C_m^{\pm}$]
The implementable covariance class is
\[
\mathcal C_m^{\pm}
=
\left\{
X(P): P\in \mathcal P_m^{\mathrm{sym}}
\right\}.
\]
\end{definitionv}

\begin{definitionv}{def:design-objective}[Design objective $F_{r,\kappa}$]
For $r,\kappa\ge 0$, the design objective evaluated at a covariance matrix $X$ is
\[
F_{r,\kappa}(X)
=
\operatorname{tr}(L_m X)
+
r\,\operatorname{tr}(L_m^\dagger X)
+
\kappa \|X\|_{S_2}
+
\operatorname{tr}(J_n X).
\]
\end{definitionv}

\begin{definitionv}{def:implementability-gap}[Implementability loss $\Delta_m^{\pm}(r,\kappa)$]
The implementability loss is
\[
\Delta_m^{\pm}(r,\kappa)
=
\inf_{P\in \mathcal P_m^{\mathrm{sym}}}
F_{r,\kappa}(X(P))
-
\inf_{X\in \mathcal E_m^{\mathrm{blk}}}
F_{r,\kappa}(X).
\]
\end{definitionv}

\begin{definitionv}{def:block-sum-handle}[Block-sum law $\mathcal L_P(S_A,S_B)$]
For an assignment law $P$, the block-sum law is the law under $P$ of
\[
(S_A,S_B)
=
\left(
\sum_{i\in A_m} Z_i,\,
\sum_{i\in B_m} Z_i
\right).
\]
\end{definitionv}

\begin{lemmav}{prop:symmetry-reduction}[Symmetry Reduction Principle]
Assume \cref{obj:ass:two-block-homophily}. Let \(r,\kappa\in[0,\infty)\). Then:
\begin{itemize}
\item \textbf{(Relaxed matrices.)} For every positive semidefinite matrix \(X\in\mathbb R^{n\times n}\) with \(X_{ii}=1\) for every \(i\), let \(u\) and \(v\) be the averages of the off-diagonal within-community and across-community entries of \(X\), respectively. The block matrix \(X(u,v)\) of \cref{obj:def:block-elliptope} has feasible block-elliptope parameters, and
\[
F_{r,\kappa}\bigl(X(u,v)\bigr)\le F_{r,\kappa}(X),
\]
where \(F_{r,\kappa}\) is \cref{obj:def:design-objective}.
\item \textbf{(Assignment laws.)} For every assignment law \(P\in\mathcal P_m^{\mathrm{bal}}\) in the sense of \cref{obj:def:balanced-design-class}, its orbit average over the full two-block automorphism group belongs to \(\mathcal P_m^{\mathrm{sym}}\) in the sense of \cref{obj:def:block-exchangeable-design-class}, has zero one-point margins,
\[
\mathbb E[Z_i]=0\qquad\text{for every }i,
\]
and satisfies
\[
F_{r,\kappa}\bigl(X(P^{\mathrm{orb}})\bigr)\le F_{r,\kappa}\bigl(X(P)\bigr).
\]
\item \textbf{(Relaxed infimum.)}
\[
\inf_{\substack{X\succeq0\\ X_{ii}=1\ \forall i}} F_{r,\kappa}(X)
=
\inf_{X\in\mathcal E_m^{\mathrm{blk}}} F_{r,\kappa}(X).
\]
\item \textbf{(Implementable infimum.)}
\[
\inf_{P\in\mathcal P_m^{\mathrm{bal}}} F_{r,\kappa}\bigl(X(P)\bigr)
=
\inf_{P\in\mathcal P_m^{\mathrm{sym}}} F_{r,\kappa}\bigl(X(P)\bigr).
\]
\end{itemize}
\end{lemmav}

\begin{theoremv}{thm:cut-corner-exactness}[Cut-Corner Exactness]
Assume the two-block homophily condition in \cref{obj:ass:two-block-homophily}. Let \(r,\kappa\in\mathbb R\) satisfy
\begin{itemize}
\item \textbf{(Nonnegative tradeoff.)} \(0\le r\).
\item \textbf{(Nonnegative robustness.)} \(0\le \kappa\).
\item \textbf{(Cut region.)}
\[
r< r_{\mathrm{cut}}(m,a,b,\kappa)
:= \max\!\left\{0,\,
\min\!\left(
2b(a+b)\left(1-\frac{\kappa}{a-b}\right),
2b(2m-2b-\kappa)
\right)\right\}.
\]
\end{itemize}
Then \(X_{\mathrm{cut}}\in\mathcal E_m^{\mathrm{blk}}\), and for every
\(X\in\mathcal E_m^{\mathrm{blk}}\) with \(X\ne X_{\mathrm{cut}}\),
\[
F_{r,\kappa}(X_{\mathrm{cut}})<F_{r,\kappa}(X),
\]
where \(F_{r,\kappa}\) is the objective in \cref{obj:def:design-objective}. Moreover,
\[
\Delta_m^{\pm}(r,\kappa)=0,
\]
with \(\Delta_m^{\pm}\) as in \cref{obj:def:implementability-gap}. Finally,
\(X_{\mathrm{cut}}\in\mathcal C_m^{\pm}\), and for every
\(X\in\mathcal C_m^{\pm}\) with \(X\ne X_{\mathrm{cut}}\),
\[
F_{r,\kappa}(X_{\mathrm{cut}})<F_{r,\kappa}(X),
\]
where \(\mathcal C_m^{\pm}\) is the class in \cref{obj:def:implementable-covariance-class}; the cut design satisfies
\[
P_{\mathrm{cut}}\in\mathcal P_m^{\mathrm{sym}}
\qquad\text{and}\qquad
X(P_{\mathrm{cut}})=X_{\mathrm{cut}},
\]
with \(\mathcal P_m^{\mathrm{sym}}\) as in \cref{obj:def:block-exchangeable-design-class}.
\end{theoremv}

\begin{theoremv}{thm:robust-corner-exactness}[Robust Corner Exactness]
Let \(m\in\mathbb N\) and \(a,b,r\in\mathbb R\). Suppose that
\begin{itemize}
\item \textbf{(Two-block homophily.)} \cref{obj:ass:two-block-homophily} holds for \((m,a,b)\).
\item \textbf{(Nonnegative pseudoinverse weight.)} \(r\ge 0\).
\end{itemize}
For the objective \(F_{r,\kappa}\) of \cref{obj:def:design-objective}, the following statements hold.

First, there exists a finite nonnegative robustness weight \(\kappa\) such that \(I_n\in\mathcal E_m^{\mathrm{blk}}\) and
\[
F_{r,\kappa}(I_n)\le F_{r,\kappa}(X)
\qquad\text{for every }X\in\mathcal E_m^{\mathrm{blk}},
\]
if and only if
\[
a+3b=2m
\qquad\text{and}\qquad
r=2b(a+b).
\]

Second, on this affine-balanced locus, every positive robustness weight makes \(I_n\) the unique relaxed minimizer: if
\[
a+3b=2m
\qquad\text{and}\qquad
r=2b(a+b),
\]
then for every \(\kappa>0\),
\[
I_n\in\mathcal E_m^{\mathrm{blk}}
\]
and, for every \(X\in\mathcal E_m^{\mathrm{blk}}\) with \(X\ne I_n\),
\[
F_{r,\kappa}(I_n)<F_{r,\kappa}(X).
\]

Third, the iid design satisfies
\[
P_{\mathrm{iid}}\in\mathcal P_m^{\mathrm{sym}}
\]
for the block-exchangeable class of \cref{obj:def:block-exchangeable-design-class}, and
\[
X(P_{\mathrm{iid}})=I_n.
\]

Finally, off the affine-balanced locus, \(I_n\) is never a minimizer at any positive finite robustness weight, and relaxed minimizers converge entrywise to \(I_n\) as \(\kappa\to\infty\). Precisely, if
\[
\neg\bigl(a+3b=2m\ \text{and}\ r=2b(a+b)\bigr),
\]
then for every \(\kappa>0\) it is not the case that
\[
F_{r,\kappa}(I_n)\le F_{r,\kappa}(X)
\qquad\text{for every }X\in\mathcal E_m^{\mathrm{blk}}.
\]
Moreover, for any family \((X_\kappa)_{\kappa\in\mathbb R}\) such that, for every \(\kappa>0\),
\[
X_\kappa\in\mathcal E_m^{\mathrm{blk}}
\quad\text{and}\quad
F_{r,\kappa}(X_\kappa)\le F_{r,\kappa}(X)
\ \text{ for every }X\in\mathcal E_m^{\mathrm{blk}},
\]
each matrix entry converges to the corresponding entry of \(I_n\):
\[
(X_\kappa)_{ij}\longrightarrow (I_n)_{ij}
\qquad\text{as }\kappa\to\infty
\]
for every \(i,j\in\{1,\ldots,n\}\).
\end{theoremv}

\begin{theoremv}{thm:gap-window}[Positive-Gap Window]
Suppose \cref{obj:ass:two-block-homophily} and \cref{obj:ass:low-scale-two-block} hold for the two-block model with block size \(m\) and weights \(a,b\). Let
\[
q=2(m-1),\qquad
\kappa_{\mathrm{gap}}(m,a,b)
=
\frac{(2m-a-3b)(a-b)\sqrt q}{a+b},
\]
and
\[
r_{\mathrm{cut,gap}}(m,a,b,\kappa)
=
\max\left\{0,\;2b(a+b)\left(1-\frac{\kappa}{a-b}\right)\right\}.
\]
Define
\[
R_x^-(m,a,b,\kappa)
=
2b(a+b)\left(1+\frac{\kappa}{(a-b)\sqrt q}\right),
\qquad
R_x^+(m,a,b,\kappa)
=
(a+b)\left(2m-a-b-\frac{\kappa}{\sqrt q}\right),
\]
and
\[
r_{\mathrm{gap}}^-(m,a,b,\kappa)
=
\frac{R_x^-(m,a,b,\kappa)+R_x^+(m,a,b,\kappa)}{2},
\qquad
r_{\mathrm{gap}}^+(m,a,b,\kappa)
=
R_x^+(m,a,b,\kappa).
\]

Then the following two statements hold.
\begin{itemize}
\item \textbf{(Odd block size.)} If \cref{obj:ass:odd-community-size} holds, then for every \(\kappa\) satisfying
\[
0\le \kappa < \kappa_{\mathrm{gap}}(m,a,b),
\]
one has
\[
r_{\mathrm{cut,gap}}(m,a,b,\kappa)
<
r_{\mathrm{gap}}^-(m,a,b,\kappa)
<
r_{\mathrm{gap}}^+(m,a,b,\kappa).
\]
Moreover, for every
\[
r_{\mathrm{gap}}^-(m,a,b,\kappa)
<
r
<
r_{\mathrm{gap}}^+(m,a,b,\kappa),
\]
the spread covariance \(X_{\mathrm{spread}}\) belongs to \(\mathcal E_m^{\mathrm{blk}}\), and it is the unique strict minimizer of the design objective \(F_{r,\kappa}\) from \cref{obj:def:design-objective} over \(\mathcal E_m^{\mathrm{blk}}\): for every \(X\in \mathcal E_m^{\mathrm{blk}}\) with \(X\ne X_{\mathrm{spread}}\),
\[
F_{r,\kappa}(X_{\mathrm{spread}})<F_{r,\kappa}(X).
\]
At the same time,
\[
X_{\mathrm{spread}}\notin \mathcal C_m^{\pm},
\]
where \(\mathcal C_m^{\pm}\) is the implementable covariance class of \cref{obj:def:implementable-covariance-class}, and the implementability loss of \cref{obj:def:implementability-gap} satisfies
\[
\Delta_m^{\pm}(r,\kappa)>0.
\]

\item \textbf{(Even block size.)} If \(m\) is even, then every covariance in the block elliptope is implementable:
\[
\mathcal E_m^{\mathrm{blk}}\subseteq \mathcal C_m^{\pm}.
\]
\end{itemize}
\end{theoremv}

\begin{theoremv}{thm:sharp-rho-star}[Sharp Rounding Loss]
Under \cref{obj:ass:two-block-homophily}, fix $r,\kappa\in[0,\infty)$.  Let
\[
q=2(m-1),\qquad
c_x=q\Bigl((a+b)+\frac{r}{a+b}\Bigr),\qquad
c_y=2b+\frac{r}{2b},\qquad
c_z=2m,
\]
let
\[
T_m=\{(x,y,z_{\mathrm{sp}}):x,y,z_{\mathrm{sp}}\ge0,\ qx+y+z_{\mathrm{sp}}=2m\},
\]
let
\[
\phi_{r,\kappa}(x,y,z_{\mathrm{sp}})
=c_xx+c_yy+c_zz_{\mathrm{sp}}
+\kappa\sqrt{qx^2+y^2+z_{\mathrm{sp}}^2},
\]
and let $d_m=0$ if $m$ is even and $d_m=2/m$ if $m$ is odd.  Then:
\begin{itemize}
\item \textbf{(Sharp equality and range.)}
\[
\rho_\star(m,a,b,r,\kappa)=\Delta_m^{\pm}(r,\kappa),
\qquad
\rho_\star(m,a,b,r,\kappa)\ge0,
\qquad
\Delta_m^{\pm}(r,\kappa)\ge0,
\]
where $\Delta_m^{\pm}$ is the implementability gap of \cref{obj:def:implementability-gap}.

\item \textbf{(Zero loss.)}
\[
\rho_\star(m,a,b,r,\kappa)=0
\]
if and only if there exists $(x,y,z_{\mathrm{sp}})\in T_m$ such that
\[
\phi_{r,\kappa}(x,y,z_{\mathrm{sp}})
\le
\phi_{r,\kappa}(x',y',z'_{\mathrm{sp}})
\quad\text{for every }(x',y',z'_{\mathrm{sp}})\in T_m,
\]
and
\[
d_m\le y+z_{\mathrm{sp}}.
\]

\item \textbf{(Strict robustness.)}
If $\kappa>0$, then there is a unique relaxed minimizer of $\phi_{r,\kappa}$ over $T_m$.  Moreover,
\[
\rho_\star(m,a,b,r,\kappa)=0
\]
if and only if every relaxed minimizer $(x,y,z_{\mathrm{sp}})\in T_m$ satisfies
\[
d_m\le y+z_{\mathrm{sp}}.
\]

\item \textbf{(Linear case.)}
If $\kappa=0$, then for every $(x,y,z_{\mathrm{sp}})\in T_m$, the point $(x,y,z_{\mathrm{sp}})$ minimizes $\phi_{r,0}$ over $T_m$ if and only if
\[
x\ne0\Rightarrow \frac{c_x}{q}\le c_y\ \text{and}\ \frac{c_x}{q}\le c_z,
\]
\[
y\ne0\Rightarrow c_y\le \frac{c_x}{q}\ \text{and}\ c_y\le c_z,
\]
and
\[
z_{\mathrm{sp}}\ne0\Rightarrow c_z\le \frac{c_x}{q}\ \text{and}\ c_z\le c_y.
\]

\item \textbf{(Active-set formula.)}
If $\kappa>0$, then there exist an active support $S\subseteq\{x,y,z_{\mathrm{sp}}\}$ and a multiplier $\lambda\in\mathbb R$ such that $S$ and $\lambda$ are admissible for
\[
\alpha^\sharp=\Bigl(\frac{c_x}{q},c_y,c_z\Bigr),
\qquad
\beta^\sharp=\Bigl(\frac1q,1,1\Bigr),
\qquad
\kappa.
\]
Writing $t^{S,\lambda}=(t_x^{S,\lambda},t_y^{S,\lambda},t_z^{S,\lambda})$ for the associated active-set point on the simplex of mass $2m$, the reduced point
\[
\Bigl(\frac{t_x^{S,\lambda}}{q},t_y^{S,\lambda},t_z^{S,\lambda}\Bigr)
\]
belongs to $T_m$, minimizes $\phi_{r,\kappa}$ over $T_m$, and the relaxed reduced value is
\[
2m\,\lambda.
\]

\item \textbf{(Truncation correction.)}
If no relaxed minimizer $(x,y,z_{\mathrm{sp}})\in T_m$ satisfies $d_m\le y+z_{\mathrm{sp}}$, then the implementable reduced value equals the weighted-simplex objective with coefficients $\alpha^\sharp$, weights $\beta^\sharp$, and robustness weight $\kappa$, evaluated at the point on the truncation segment
\[
H_{d_m}=\{(2m-d_m,s,d_m-s):0\le s\le d_m\}
\]
selected by $\operatorname{truncSelector}(2m,d_m,\alpha^\sharp,\beta^\sharp,\kappa)$:
\[
\operatorname{val}_{\mathrm{impl}}(m,a,b,r,\kappa)
=
\Phi_{\alpha^\sharp,\beta^\sharp,\kappa}
\!\left(
2m-d_m,\,
s_\star,\,
d_m-s_\star
\right),
\]
where
\[
s_\star=\operatorname{truncSelector}(2m,d_m,\alpha^\sharp,\beta^\sharp,\kappa).
\]

\item \textbf{(Even block size.)}
If $m$ is even, then
\[
\rho_\star(m,a,b,r,\kappa)=0.
\]
\end{itemize}
\end{theoremv}

\begin{lemmav}{lem:block-spectral-coordinates}[Block Spectral Coordinates]
For any real numbers $r,\kappa,u,v$, assume:
\begin{itemize}
\item \textbf{(Homophily.)} \cref{obj:ass:two-block-homophily} holds for the block size $m$ and weights $a,b$.
\item \textbf{(Spectral coordinates.)} Set
\[
q=2(m-1),\qquad
x=1-u,\qquad
y=1+(m-1)u-mv,\qquad
z_{\mathrm{sp}}=1+(m-1)u+mv .
\]
\item \textbf{(Reduced coefficients.)} Set
\[
c_x=q\left((a+b)+\frac{r}{a+b}\right),\qquad
c_y=2b+\frac{r}{2b},\qquad
c_z=2m .
\]
\end{itemize}
Then the block covariance matrix $X(u,v)$ belongs to the block elliptope $\mathcal E_m^{\mathrm{blk}}$ if and only if
\[
x\ge 0,\qquad y\ge 0,\qquad z_{\mathrm{sp}}\ge 0,\qquad
qx+y+z_{\mathrm{sp}}=2m .
\]
Moreover, for the design objective of \cref{obj:def:design-objective},
\[
F_{r,\kappa}\bigl(X(u,v)\bigr)
=
c_xx+c_yy+c_zz_{\mathrm{sp}}
+\kappa\sqrt{qx^2+y^2+z_{\mathrm{sp}}^2}.
\]
Finally,
\[
X_{\mathrm{cut}}=X(1,-1)
\quad\text{and}\quad
(1-1,\;1+(m-1)\cdot 1-m(-1),\;1+(m-1)\cdot 1+m(-1))
=(0,2m,0),
\]
while
\[
I_n=X(0,0)
\quad\text{and}\quad
(1-0,\;1+(m-1)\cdot 0-m\cdot 0,\;1+(m-1)\cdot 0+m\cdot 0)
=(1,1,1).
\]
\end{lemmav}

\begin{lemmav}{lem:cut-vertex-certificate}[Cut Vertex Certificate]
Let \(m\) be fixed and write \(q=2(m-1)\). Suppose that
\begin{itemize}
\item \textbf{(Positive spectral multiplicity.)} \(q>0\).
\item \textbf{(Nonnegative robustness.)} \(\kappa\ge 0\).
\item \textbf{(Cut-side coefficient separation.)} The coefficients satisfy
\[
\frac{c_x}{q}>c_y+\kappa
\qquad\text{and}\qquad
c_z>c_y+\kappa .
\]
\end{itemize}
Then \((0,2m,0)\in T_m\), where
\[
T_m=\{(x,y,z_{\mathrm{sp}}): x\ge 0,\ y\ge 0,\ z_{\mathrm{sp}}\ge 0,\ qx+y+z_{\mathrm{sp}}=2m\}.
\]
Moreover, for every \((x,y,z_{\mathrm{sp}})\in T_m\) with
\((x,y,z_{\mathrm{sp}})\neq (0,2m,0)\),
\[
c_x\cdot 0+c_y(2m)+c_z\cdot 0
+\kappa\sqrt{q\cdot 0^2+(2m)^2+0^2}
<
c_xx+c_yy+c_zz_{\mathrm{sp}}
+\kappa\sqrt{qx^2+y^2+z_{\mathrm{sp}}^2}.
\]
Thus \((0,2m,0)\) is the unique minimizer on \(T_m\) of the reduced objective.
\end{lemmav}

\begin{lemmav}{lem:spread-vertex-certificate}[Spread Vertex Certificate]
Let \(m\in\mathbb N\), let \(c_x,c_y,c_z,\kappa\in\mathbb R\), and set \(q=2(m-1)\). Suppose that
\begin{itemize}
\item \textbf{(Positive multiplicity.)} \(q>0\).
\item \textbf{(Nonnegative robustness.)} \(\kappa\ge 0\).
\item \textbf{(The \(y\)-coefficient is large.)} \(c_y>c_x/q+\kappa/\sqrt q\).
\item \textbf{(The \(z_{\mathrm{sp}}\)-coefficient is large.)} \(c_z>c_x/q+\kappa/\sqrt q\).
\end{itemize}
Define
\[
T_m=\{(x,y,z_{\mathrm{sp}}):x\ge0,\ y\ge0,\ z_{\mathrm{sp}}\ge0,\ qx+y+z_{\mathrm{sp}}=2m\}
\]
and
\[
\phi(x,y,z_{\mathrm{sp}})
=
c_xx+c_yy+c_zz_{\mathrm{sp}}
+\kappa\sqrt{qx^2+y^2+z_{\mathrm{sp}}^2}.
\]
Then \((2m/q,0,0)\in T_m\), and for every \((x,y,z_{\mathrm{sp}})\in T_m\) with
\((x,y,z_{\mathrm{sp}})\neq (2m/q,0,0)\),
\[
\phi(2m/q,0,0)<\phi(x,y,z_{\mathrm{sp}}).
\]
\end{lemmav}

\begin{lemmav}{lem:frobenius-center-certificate}[Frobenius Center Certificate]
Let \(m\in\mathbb N\), set \(q=2(m-1)\), and suppose \(q>0\).  For any real coefficients \(c_x,c_y,c_z\) and any real \(\kappa\), define
\[
T_m=\{(x,y,z_{\mathrm{sp}}): x,y,z_{\mathrm{sp}}\ge 0,\ qx+y+z_{\mathrm{sp}}=2m\}
\]
and
\[
\phi_{r,\kappa}(x,y,z_{\mathrm{sp}})
=
c_xx+c_yy+c_zz_{\mathrm{sp}}
+\kappa\sqrt{q x^2+y^2+z_{\mathrm{sp}}^2}.
\]
Then:
\begin{itemize}
\item \textbf{(Frobenius center.)} The point \((1,1,1)\) belongs to \(T_m\), and for every \((x,y,z_{\mathrm{sp}})\in T_m\) with \((x,y,z_{\mathrm{sp}})\ne(1,1,1)\),
\[
\sqrt{q\cdot 1^2+1^2+1^2}
<
\sqrt{q x^2+y^2+z_{\mathrm{sp}}^2}.
\]
\item \textbf{(Certificate necessity.)} If \((1,1,1)\) minimizes \(\phi_{r,\kappa}\) over \(T_m\), that is, if
\[
\phi_{r,\kappa}(1,1,1)\le
\phi_{r,\kappa}(x,y,z_{\mathrm{sp}})
\quad\text{for all }(x,y,z_{\mathrm{sp}})\in T_m,
\]
then
\[
\frac{c_x}{q}=c_y
\quad\text{and}\quad
c_y=c_z.
\]
\item \textbf{(Strict optimality.)} If \(\kappa>0\), \(\frac{c_x}{q}=c_y\), and \(c_y=c_z\), then for every \((x,y,z_{\mathrm{sp}})\in T_m\) with \((x,y,z_{\mathrm{sp}})\ne(1,1,1)\),
\[
\phi_{r,\kappa}(1,1,1)
<
\phi_{r,\kappa}(x,y,z_{\mathrm{sp}}).
\]
\end{itemize}
\end{lemmav}

\begin{lemmav}{lem:pm-reduced-slice-characterization}[Parity-Truncated Reduced Slice]
Let \(m\ge 2\) be an integer and let \(u,v\in\mathbb R\). Write
\[
x=1-u,\qquad
y=1+(m-1)u-mv,\qquad
z_{\mathrm{sp}}=1+(m-1)u+mv,
\]
and let \(d_m=0\) when \(m\) is even and \(d_m=2/m\) when \(m\) is odd. Then the block covariance matrix \(X(u,v)\) belongs to the implementable covariance class \(\mathcal C_m^{\pm}\) of \cref{obj:def:implementable-covariance-class} if and only if
\[
x\ge 0,\qquad y\ge 0,\qquad z_{\mathrm{sp}}\ge 0,\qquad
2(m-1)x+y+z_{\mathrm{sp}}=2m,
\]
and
\[
y+z_{\mathrm{sp}}\ge d_m .
\]
\end{lemmav}

\begin{lemmav}{lem:weighted-simplex-active-set}[Weighted Simplex Active Set]
Let \(m\ge 2\), let \(M>0\) satisfy \(M=2m\), and let
\(\mathcal I=\{x,y,z_{\mathrm{sp}}\}\). Let
\(\alpha=(\alpha_i)_{i\in\mathcal I}\), let \(\kappa\ge 0\), and let
\(\beta=(\beta_i)_{i\in\mathcal I}\) have strictly positive entries with
\[
  \beta_x=\frac{1}{2(m-1)},\qquad
  \beta_y=\beta_{z_{\mathrm{sp}}}=1 .
\]
For \(t\in\mathbb R^{\mathcal I}\), write
\[
  \Delta_M=\Bigl\{t:\ t_i\ge 0\ \text{for all }i\in\mathcal I,\ 
  \sum_{i\in\mathcal I}t_i=M\Bigr\},
\]
and
\[
  \Phi(t)=\sum_{i\in\mathcal I}\alpha_i t_i
  +\kappa\sqrt{\sum_{i\in\mathcal I}\beta_i t_i^2}.
\]
Then the following hold.
\begin{itemize}
\item \textbf{(Positive robustness.)} If \(\kappa>0\), there is a unique pair
\((S,\lambda)\), where \(S\subseteq\mathcal I\) is nonempty and
\(\lambda\in\mathbb R\), such that
\[
  \sum_{i\in S}\frac{(\lambda-\alpha_i)^2}{\beta_i}=\kappa^2,\qquad
  \alpha_i<\lambda\ \text{for every }i\in S,\qquad
  \lambda\le \alpha_j\ \text{for every }j\notin S .
\]
For this unique pair, define
\[
  t_i^\star=
  \begin{cases}
  \displaystyle
  M\,\frac{(\lambda-\alpha_i)/\beta_i}
  {\sum_{h\in S}(\lambda-\alpha_h)/\beta_h}, & i\in S,\\[1.1em]
  0, & i\notin S .
  \end{cases}
\]
Then \(t^\star\in\Delta_M\), and \(t^\star\) is the unique minimizer of
\(\Phi\) over \(\Delta_M\) in the strict sense that, for every
\(s\in\Delta_M\) with \(s\ne t^\star\),
\[
  \Phi(t^\star)<\Phi(s).
\]
Its value is
\[
  \Phi(t^\star)=M\lambda .
\]

\item \textbf{(Zero robustness.)} If \(\kappa=0\), then for every
\(t\in\mathbb R^{\mathcal I}\),
\[
  \Bigl(t\in\Delta_M\ \text{and}\ 
  \Phi(t)\le \Phi(s)\ \text{for every }s\in\Delta_M\Bigr)
  \quad\Longleftrightarrow\quad
  t\in\Delta_M\ \text{and whenever }t_i\ne 0,\ 
  \alpha_i\le \alpha_j\ \text{for every }j\in\mathcal I .
\]
Equivalently, the minimizer set is exactly the exposed
\(\alpha\)-minimizing face of \(\Delta_M\).
\end{itemize}
\end{lemmav}

\begin{lemmav}{lem:weighted-simplex-truncation}[Weighted Simplex Truncation]
Let \(m\ge 2\), let \(M,d\in\mathbb R\), and let \(\alpha=(\alpha_x,\alpha_y,\alpha_z)\), \(\beta=(\beta_x,\beta_y,\beta_z)\), and \(\kappa\in\mathbb R\). Suppose:
\begin{itemize}
\item \textbf{(Mass and truncation.)} \(M>0\), \(M=2m\), and \(0\le d\le M\).
\item \textbf{(Weights.)} Each \(\beta_i\) is strictly positive and
\[
\beta=\left(\frac{1}{2(m-1)},1,1\right),
\]
so in particular \(\beta_y=\beta_z=1\).
\item \textbf{(Robustness.)} \(\kappa\ge 0\).
\end{itemize}
Write
\[
K_d=\{t\in\Delta_M:t_y+t_z\ge d\},
\qquad
\Phi(t)=\sum_i\alpha_i t_i+\kappa\sqrt{\sum_i\beta_i t_i^2}.
\]
For \(\kappa>0\), for every nonempty active support \(S\subseteq\{x,y,z\}\) and multiplier \(\lambda\) satisfying
\[
\sum_{i\in S}\frac{(\lambda-\alpha_i)^2}{\beta_i}=\kappa^2,\qquad
\alpha_i<\lambda\ \text{for }i\in S,\qquad
\lambda\le \alpha_j\ \text{for }j\notin S,
\]
let \(t^{\mathrm{rel}}\) be the active-set point
\[
t^{\mathrm{rel}}_i=
\begin{cases}
\displaystyle
\frac{M((\lambda-\alpha_i)/\beta_i)}
{\sum_{h\in S}(\lambda-\alpha_h)/\beta_h}, & i\in S,\\[1em]
0, & i\notin S .
\end{cases}
\]
If \(t^{\mathrm{rel}}\in K_d\), then
\[
\Phi(t^{\mathrm{rel}})\le \Phi(s)
\qquad\text{for every }s\in K_d.
\]
If \(t^{\mathrm{rel}}_y+t^{\mathrm{rel}}_z<d\), then
\[
t^\star=(M-d,s^\star,d-s^\star)\in K_d
\]
and
\[
\Phi(t^\star)\le \Phi(s)
\qquad\text{for every }s\in K_d,
\]
where \(s^\star=\operatorname{truncSelector}(M,d,\alpha,\beta,\kappa)\), namely with
\[
\delta=\alpha_y-\alpha_z,\qquad A=\beta_x(M-d)^2,
\]
\[
s^\star=
\begin{cases}
0, & \displaystyle \delta\ge \frac{\kappa d}{\sqrt{A+d^2}},\\[0.8em]
d, & \displaystyle \delta\le -\frac{\kappa d}{\sqrt{A+d^2}},\\[0.8em]
\displaystyle
\frac{d-\delta\sqrt{(A+d^2/2)/(\kappa^2-\delta^2/2)}}{2},
& \text{otherwise.}
\end{cases}
\]

For \(\kappa=0\), let \(t^{\mathrm{rel}}\) be a point of the exposed \(\alpha\)-minimizing face
\[
\{t\in\Delta_M:\ t_i\ne 0\Rightarrow \alpha_i\le \alpha_j\ \text{for every }j\}.
\]
Assume the tie convention that, if this exposed face contains at least one point of \(K_d\), then the selected point \(t^{\mathrm{rel}}\) lies in \(K_d\). If \(t^{\mathrm{rel}}\in K_d\), then
\[
\Phi(t^{\mathrm{rel}})\le \Phi(s)
\qquad\text{for every }s\in K_d.
\]
If \(t^{\mathrm{rel}}_y+t^{\mathrm{rel}}_z<d\), then
\[
t^\star=(M-d,s^\star,d-s^\star)\in K_d
\]
and
\[
\Phi(t^\star)\le \Phi(s)
\qquad\text{for every }s\in K_d,
\]
where \(s^\star=\operatorname{truncSelector}(M,d,\alpha,\beta,0)\), equivalently, in the ordered selector convention, \(s^\star=0\) if \(\alpha_y-\alpha_z\ge 0\) and \(s^\star=d\) if \(\alpha_y-\alpha_z<0\).
\end{lemmav}

\begin{lemmav}{lem:rounding-gap-reduction}[Reduced Gap Identity]
For any real $r$ and any $\kappa\ge 0$, under \cref{obj:ass:two-block-homophily}, the implementability loss of \cref{obj:def:implementability-gap} satisfies
\[
\Delta_m^{\pm}(r,\kappa)
=
\inf_{\substack{(x,y,z_{\mathrm{sp}})\in T_m\\ y+z_{\mathrm{sp}}\ge d_m}}
\phi_{r,\kappa}(x,y,z_{\mathrm{sp}})
-
\inf_{(x,y,z_{\mathrm{sp}})\in T_m}
\phi_{r,\kappa}(x,y,z_{\mathrm{sp}}).
\]
\end{lemmav}
