\begin{proof}[Proof of \cref{obj:thm:convex-design}]
\begin{enumerate}
\item Put \(m_n=|I_n|\).  First suppose that \(I_n\) is nonempty and define the constant design
\[
p_k=\frac{B_n}{m_n}\qquad(k\in I_n).
\]
Since \(m_n>0\), the budget bounds give
\[
\epsilon\le \frac{B_n}{m_n}\le 1-\epsilon.
\]
Thus \(p\) satisfies the probability, positivity, and budget requirements, the last because
\(\sum_{k\in I_n}p_k=m_n(B_n/m_n)=B_n\).  If instead \(I_n\) is empty, the two budget inequalities force \(B_n=0\), and the unique empty design is feasible.  Hence \(\mathcal P_{n,B_n,\epsilon}\) is nonempty.
% lean: feasibleSet_nonempty

\item The feasible class is closed.  Indeed, when the positivity-margin condition is admissible, it is the intersection of the coordinatewise closed probability constraints, the coordinatewise closed bounds
\(\epsilon\le p_k\le1-\epsilon\), and the closed affine hyperplane
\(\sum_k p_k=B_n\).  When that condition is not admissible, the feasible class is empty and hence closed.  It is contained in the finite product \([0,1]^{I_n}\), which is compact; therefore \(\mathcal P_{n,B_n,\epsilon}\) is compact.
% lean: feasibleSet_compact

\item Let \(p,q\in\mathcal P_{n,B_n,\epsilon}\), and let \(a,b\ge0\) with \(a+b=1\).  Coordinatewise,
\[
\epsilon=a\epsilon+b\epsilon
 \le ap_k+bq_k
 \le a(1-\epsilon)+b(1-\epsilon)=1-\epsilon.
\]
The same convex combination lies in \([0,1]\), and its budget is
\[
\sum_{k\in I_n}(ap_k+bq_k)
 =aB_n+bB_n=B_n.
\]
Thus \(ap+bq\in\mathcal P_{n,B_n,\epsilon}\), proving convexity.
% lean: feasibleSet_convex

\item The envelope is continuous on the feasible class.  For fixed outcome indices \(i,j\), write \(S_{ij}=S_{ij}(G_n)\).  If \(S_{ij}\) is nonempty, the positivity bounds imply that both
\[
\prod_{k\in S_{ij}}p_k
\quad\text{and}\quad
\prod_{k\in S_{ij}}(1-p_k)
\]
are nonzero throughout the feasible class.  Their reciprocal kernels, and hence
\(r^1_{ij}(G_n,p)\) and \(r^0_{ij}(G_n,p)\), are therefore continuous there.  If \(S_{ij}\) is empty, both kernels are identically zero.  Adding the constant cross term \(2r^{10}_{ij}(G_n)\) and taking the finite double sum shows that \(V_{\mathrm{env}}(G_n,\cdot)\) is continuous on \(\mathcal P_{n,B_n,\epsilon}\).
% lean: varEnvelope_continuousOn_feasible

\item For each \(i,j\), the maps
\[
p\longmapsto
\left(\prod_{k\in S_{ij}}p_k\right)^{-1},
\qquad
p\longmapsto
\left(\prod_{k\in S_{ij}}(1-p_k)\right)^{-1}
\]
are convex on the feasible class.  When \(S_{ij}\) is nonempty, this gives convexity of \(r^1_{ij}(G_n,\cdot)\) and \(r^0_{ij}(G_n,\cdot)\) after subtracting \(1\); when \(S_{ij}\) is empty, these kernels are constant.  Hence each summand
\[
r^1_{ij}(G_n,p)+r^0_{ij}(G_n,p)+2r^{10}_{ij}(G_n)
\]
is convex.  Finite summation preserves convexity, and multiplication by the nonnegative factor \(4/n\) preserves it as well.  Thus \(V_{\mathrm{env}}(G_n,\cdot)\) is convex on \(\mathcal P_{n,B_n,\epsilon}\).
% lean: prod_one_sub_inv_convexOn_feasible

\item By nonemptiness and compactness of the feasible class and continuity of the envelope, \(V_{\mathrm{env}}(G_n,\cdot)\) attains its minimum on \(\mathcal P_{n,B_n,\epsilon}\).  Denote one minimizer by \(p_n^*(G_n)\).  Then
\[
V_{\mathrm{env}}(G_n,p_n^*(G_n))
\le V_{\mathrm{env}}(G_n,q)
\]
for every \(q\in\mathcal P_{n,B_n,\epsilon}\).
% lean: IsCompact.exists_isMinOn

\item It remains to establish the multiplier statement.  Introduce a globally continuously differentiable reciprocal cutoff which agrees with \(x^{-1}\) whenever \(x\ge\epsilon/2\), and replace each reciprocal factor in the treated and control kernels by this cutoff.  This yields a globally continuously differentiable extension of \(V_{\mathrm{env}}(G_n,\cdot)\) which agrees with the original envelope whenever
\[
p_k\ge\epsilon/2,\qquad 1-p_k\ge\epsilon/2
\quad(k\in I_n).
\]
Transporting the resulting objective, divided by \(4\), to Euclidean coordinates indexed by \(\operatorname{Fin}(m_n)\) gives a differentiable objective.

For a feasible \(p\), a coordinate \(k\), and a shared neighborhood \(S_{ij}\), differentiation along the \(k\)-th coordinate line gives
\[
\left.\frac{d}{dt}\right|_{t=0}
 \prod_{\ell\in S_{ij}}
 \bigl(p_\ell+\mathbf 1\{\ell=k\}t\bigr)^{-1}
=
\begin{cases}
-\left(\prod_{\ell\in S_{ij}}p_\ell^{-1}\right)p_k^{-1},
   &k\in S_{ij},\\
0,&k\notin S_{ij},
\end{cases}
\]
and
\[
\left.\frac{d}{dt}\right|_{t=0}
 \prod_{\ell\in S_{ij}}
 \bigl(1-p_\ell-\mathbf 1\{\ell=k\}t\bigr)^{-1}
=
\begin{cases}
\left(\prod_{\ell\in S_{ij}}(1-p_\ell)^{-1}\right)(1-p_k)^{-1},
   &k\in S_{ij},\\
0,&k\notin S_{ij}.
\end{cases}
\]
The same formulas hold for the corresponding overlap kernels, including the case of an empty shared neighborhood.  Summing these derivatives shows that the derivative of \(V_{\mathrm{env}}(G_n,\cdot)/4\) in coordinate \(k\) is \(g_k(G_n,p)\).  Since sufficiently small coordinate perturbations remain in the region where the smooth extension equals the original envelope, the \(k\)-th Euclidean gradient coordinate of the transported objective at \(p\) is exactly \(g_k(G_n,p)\).
% lean: envelopeGrad_eq_gradient_varEnvelopeExt

\item Fix any minimizer \(p_n^*(G_n)\).  In Euclidean coordinates, impose the equality constraint
\[
\sum_{a=1}^{m_n}x_a-B_n=0
\]
and the \(2m_n\) inequality constraints
\[
x_a-\epsilon\ge0,
\qquad
1-\epsilon-x_a\ge0
\qquad(a=1,\ldots,m_n).
\]
The correspondence between these feasible Euclidean points and
\(\mathcal P_{n,B_n,\epsilon}\) is exact.  Moreover, the transported smooth objective agrees on this feasible set with \(V_{\mathrm{env}}(G_n,\cdot)/4\), so the Euclidean image of \(p_n^*(G_n)\) is a local minimum.  The objective is differentiable, all constraints are affine, and the linear constraint qualification holds.

The finite-dimensional first-order necessary condition therefore supplies a multiplier \(\lambda_n\) for the budget constraint and nonnegative multipliers for the lower and upper box constraints.  Let \(\nu^-_{k,n}\) be the lower-box multiplier for coordinate \(k\), and let \(\nu^+_{k,n}\) be the upper-box multiplier.  Coordinatewise stationarity, together with the preceding gradient identity, is
\[
g_k(G_n,p_n^*(G_n))
-\lambda_n-\nu^-_{k,n}+\nu^+_{k,n}=0,
\]
or equivalently
\[
g_k(G_n,p_n^*(G_n))
=\lambda_n-\nu^+_{k,n}+\nu^-_{k,n}.
\]
The multipliers satisfy \(\nu^+_{k,n}\ge0\) and \(\nu^-_{k,n}\ge0\).  Complementary slackness for the upper and lower constraints gives, respectively,
\[
\nu^+_{k,n}\bigl((p_n^*(G_n))_k-(1-\epsilon)\bigr)=0,
\qquad
\nu^-_{k,n}\bigl(\epsilon-(p_n^*(G_n))_k\bigr)=0.
\]
This holds for every \(k\in I_n\), as required.
% lean: convex_design
\end{enumerate}
\end{proof}
