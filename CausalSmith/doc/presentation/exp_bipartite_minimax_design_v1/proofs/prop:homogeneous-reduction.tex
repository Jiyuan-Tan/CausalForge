\begin{proof}[Proof of \cref{obj:prop:homogeneous-reduction}]
\begin{enumerate}
\item For every \(k\in I_n\), the homogeneity condition and \(p\in(0,1)\) give
\[
0<p_k=p<1.
\]
% lean: CausalSmith.Experimentation.BipartiteMinimaxDesign.homogeneous_reduction

\item Fix \(i,j\in O_n\).  If \(\lvert S_{ij}(G_n)\rvert>0\), substituting \(p_k=p\) into the product defining the treated overlap load yields
\[
r_{ij}^1(G_n,p)
=\prod_{k\in S_{ij}(G_n)}p^{-1}-1
=p^{-\lvert S_{ij}(G_n)\rvert}-1.
\]
If \(\lvert S_{ij}(G_n)\rvert=0\), that definition gives \(r_{ij}^1(G_n,p)=0\).
% lean: CausalSmith.Experimentation.BipartiteMinimaxDesign.homogeneous_reduction

\item The same split gives the control overlap load.  When
\(\lvert S_{ij}(G_n)\rvert>0\),
\[
r_{ij}^0(G_n,p)
=\prod_{k\in S_{ij}(G_n)}(1-p)^{-1}-1
=(1-p)^{-\lvert S_{ij}(G_n)\rvert}-1;
\]
when \(\lvert S_{ij}(G_n)\rvert=0\), it equals \(0\).
% lean: CausalSmith.Experimentation.BipartiteMinimaxDesign.homogeneous_reduction

\item By the variance-scale covariance expansion for independent heterogeneous Bernoulli assignment, applicable by step 1,
\[
\sigma_{G_n,p}^2(Y)
=
\frac{1}{n}\sum_{i\in O_n}\sum_{j\in O_n}
\left[
r_{ij}^1(G_n,p)(Y_i^1-\mu_1)(Y_j^1-\mu_1)
+r_{ij}^0(G_n,p)(Y_i^0-\mu_0)(Y_j^0-\mu_0)
+2r_{ij}^{10}(G_n)(Y_i^1-\mu_1)(Y_j^0-\mu_0)
\right].
\]
% lean: CausalSmith.Experimentation.BipartiteMinimaxDesign.varScale_homogeneous_formula
\end{enumerate}
\end{proof}
