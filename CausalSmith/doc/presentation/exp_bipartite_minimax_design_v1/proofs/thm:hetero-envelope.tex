\begin{proof}[Proof of \cref{obj:thm:hetero-envelope}]
\begin{enumerate}
\item Write
\[
a_i=Y_i^1-\mu_1,\qquad b_i=Y_i^0-\mu_0,
\]
and
\[
A_i=\frac{T_i(Z)}{\pi_i^1(p)}-1,\qquad
B_i=\frac{C_i(Z)}{\pi_i^0(p)}-1.
\]
Under \cref{obj:ass:independent-heterogeneous-bernoulli},
\[
\mathbb E_p[A_iA_j]=r_{ij}^1(G_n,p),\quad
\mathbb E_p[B_iB_j]=r_{ij}^0(G_n,p),
\]
while
\[
\mathbb E_p[A_iB_j]=\mathbb E_p[B_iA_j]=-r_{ij}^{10}(G_n).
\]
Since \(\eta_i(p,Z)=A_i a_i-B_i b_i\), expanding the product gives
\[
\mathbb E_p\!\left[\eta_i(p,Z)\eta_j(p,Z)\right]
=
r_{ij}^1(G_n,p)a_i a_j
+r_{ij}^0(G_n,p)b_i b_j
+r_{ij}^{10}(G_n)\bigl(a_i b_j+b_i a_j\bigr),
\]
which is the first claim.
% lean: linScore_pair_moment

\item The same calculation gives \(\mathbb E_p[\eta_i(p,Z)]=0\).  Expanding the variance scale into outcome-pair moments and using the symmetry of \(r_{ij}^{10}(G_n)\) therefore yields
\[
\sigma^2_{G_n,p}(Y)
=
\frac1n\sum_{i\in O_n}\sum_{j\in O_n}
\left\{
r_{ij}^1(G_n,p)a_i a_j
+r_{ij}^0(G_n,p)b_i b_j
+2r_{ij}^{10}(G_n)a_i b_j
\right\}.
\]
% lean: varScale_homogeneous_formula

\item \cref{obj:ass:bounded-outcomes} and the triangle inequality imply
\[
|\mu_1|\le 1,\qquad |\mu_0|\le 1;
\]
the zero-outcome case is immediate.  Hence, for every \(i\in O_n\),
\[
|a_i|\le2,\qquad |b_i|\le2.
\]
Moreover, if \(S_{ij}(G_n)=\varnothing\), all three overlap loads are zero.  Otherwise, \(0<p_k<1\) implies
\[
\prod_{k\in S_{ij}(G_n)}p_k^{-1}\ge1,
\qquad
\prod_{k\in S_{ij}(G_n)}(1-p_k)^{-1}\ge1.
\]
Thus
\[
r_{ij}^1(G_n,p)\ge0,\qquad
r_{ij}^0(G_n,p)\ge0,\qquad
r_{ij}^{10}(G_n)\ge0.
\]
% lean: pair_kernel_le_envelope_kernel

\item Consequently, each of the three terms in the preceding variance representation is bounded above by four times its nonnegative load:
\[
\begin{aligned}
&r_{ij}^1(G_n,p)a_i a_j
+r_{ij}^0(G_n,p)b_i b_j
+2r_{ij}^{10}(G_n)a_i b_j\\
&\hspace{2cm}\le
4\left\{
r_{ij}^1(G_n,p)+r_{ij}^0(G_n,p)+2r_{ij}^{10}(G_n)
\right\}.
\end{aligned}
\]
% lean: mul_centered_pair_le_four_mul_of_nonneg

\item Summing this inequality over \(i,j\in O_n\) and multiplying by \(n^{-1}\ge0\) gives
\[
\begin{aligned}
\sigma^2_{G_n,p}(Y)
&\le
\frac1n\sum_{i\in O_n}\sum_{j\in O_n}
4\left\{
r_{ij}^1(G_n,p)+r_{ij}^0(G_n,p)+2r_{ij}^{10}(G_n)
\right\}\\
&=
V_{\mathrm{env}}(G_n,p),
\end{aligned}
\]
by \cref{obj:def:graph-envelope}.
% lean: hetero_envelope
\end{enumerate}
\end{proof}
