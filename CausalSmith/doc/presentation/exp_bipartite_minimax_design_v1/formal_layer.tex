\begin{assumptionv}{ass:bipartite-interference}[Bipartite interference restriction]
For every $i\in O_n$ and every $z,z'\in\{0,1\}^{m_n}$,
\[
z_{N_i(G_n)}=z'_{N_i(G_n)}
\quad\Longrightarrow\quad
Y_i\!\left(z_{N_i(G_n)}\right)=Y_i\!\left(z'_{N_i(G_n)}\right).
\]
\end{assumptionv}

\begin{assumptionv}{ass:independent-heterogeneous-bernoulli}[Independent heterogeneous Bernoulli assignment]
For every $k\in I_n$, $Z_k$ is Bernoulli with success probability $p_k$, and the collection $\{Z_k:k\in I_n\}$ is mutually independent.
\end{assumptionv}

\begin{assumptionv}{ass:positivity-floor}[Assignment positivity]
For every $k\in I_n$,
\[
\epsilon\leq p_k\leq 1-\epsilon.
\]
\end{assumptionv}

\begin{assumptionv}{ass:budget-balance}[Assignment budget balance]
\[
\sum_{k\in I_n}p_k=B_n.
\]
\end{assumptionv}

\begin{assumptionv}{ass:bounded-outcomes}[Bounded potential outcomes]
For every $i\in O_n$,
\[
\lvert Y_i^1\rvert\leq 1
\qquad\text{and}\qquad
\lvert Y_i^0\rvert\leq 1.
\]
\end{assumptionv}

\begin{assumptionv}{ass:bounded-outcome-degree}[Bounded outcome degree]
\[
0 < \bar d
\quad\text{and}\quad
\max_{i\in O_n} d_i \leq \bar d.
\]
\end{assumptionv}

\begin{assumptionv}{ass:bounded-overlap-dependency}[Bounded overlap dependency]
\[
0 < \bar D \quad\text{and}\quad \Delta_n \le \bar D.
\]
\end{assumptionv}

\begin{assumptionv}{ass:variance-nondegenerate}[Nondegenerate variance scale]
\[
\liminf_{n\to\infty}\sigma^2_{G_n,p_n^*(G_n)}(Y)>0.
\]
\end{assumptionv}

\begin{definitionv}{def:feasible-designs}[Feasible design class $\mathcal P_{n,B_n,\epsilon}$]
Under \cref{obj:ass:positivity-floor} and~\cref{obj:ass:budget-balance}, with \(0<\epsilon<1/2\), define
\[
\mathcal P_{n,B_n,\epsilon}
=
\left\{
p\in[0,1]^{m_n}:
\epsilon\le p_k\le 1-\epsilon\ \text{for every } k\in I_n,
\quad
\sum_{k\in I_n}p_k=B_n
\right\}.
\]
\end{definitionv}

\begin{definitionv}{def:hajek-denominators}[Hájek denominators $D_1(p,Z)$ and $D_0(p,Z)$]
Define
\[
D_1(p,Z)=\sum_{i\in O_n}\frac{T_i(Z)}{\pi_i^1(p)},
\qquad
D_0(p,Z)=\sum_{i\in O_n}\frac{C_i(Z)}{\pi_i^0(p)}.
\]
\end{definitionv}

\begin{definitionv}{def:hetero-hajek-estimator}[Hájek estimator $\widehat\tau_H(p)$]
Define
\[
\widehat\tau_H(p)
=
\mathbf 1\{D_1(p,Z)>0\}
\frac{\displaystyle\sum_{i\in O_n}\frac{T_i(Z)Y_i^{obs}}{\pi_i^1(p)}}{D_1(p,Z)}
-
\mathbf 1\{D_0(p,Z)>0\}
\frac{\displaystyle\sum_{i\in O_n}\frac{C_i(Z)Y_i^{obs}}{\pi_i^0(p)}}{D_0(p,Z)}.
\]
\end{definitionv}

\begin{definitionv}{def:first-order-linearization}[Linearization summand $\eta_i(p,Z)$]
For each \(i\in O_n\), define
\[
\eta_i(p,Z)
=
\left(\frac{T_i(Z)}{\pi_i^1(p)}-1\right)(Y_i^1-\mu_1)
-
\left(\frac{C_i(Z)}{\pi_i^0(p)}-1\right)(Y_i^0-\mu_0).
\]
\end{definitionv}

\begin{definitionv}{def:variance-scale}[Variance scale $\sigma^2_{G_n,p}(Y)$]
For a working finite design $D$, define
\[
\sigma^2_{G_n,D,p}(Y)
=
n\,\operatorname{Var}_D\!\left(
\frac{1}{n}\sum_{i\in O_n}\eta_i(p,Z)
\right).
\]
When $D$ is the heterogeneous Bernoulli design indexed by $p$, abbreviate this quantity as $\sigma^2_{G_n,p}(Y)$.
\end{definitionv}

\begin{definitionv}{def:graph-envelope}[Graph envelope $V_{\mathrm{env}}(G_n,p)$]
Define
\[
V_{\mathrm{env}}(G_n,p)
=
\frac{4}{n}\sum_{i,j\in O_n}
\left\{
r_{ij}^1(G_n,p)
+
r_{ij}^0(G_n,p)
+
2r_{ij}^{10}(G_n)
\right\}.
\]
\end{definitionv}

\begin{definitionv}{def:optimal-design}[Envelope-optimal design $p_n^*(G_n)$]
Whenever the minimizer set is nonempty, define
\[
p_n^*(G_n)
\in
\operatorname{argmin}_{p\in\mathcal P_{n,B_n,\epsilon}}
V_{\mathrm{env}}(G_n,p).
\]
\end{definitionv}

\begin{definitionv}{def:kkt-gradient}[Envelope gradient score $g_k(G_n,p)$]
Define
\[
g_k(G_n,p)
=
\frac{1}{n}
\sum_{\substack{i,j\in O_n:\ k\in S_{ij}(G_n)}}
\left\{
-
\left(\prod_{\ell\in S_{ij}(G_n)}p_\ell^{-1}\right)p_k^{-1}
+
\left(\prod_{\ell\in S_{ij}(G_n)}(1-p_\ell)^{-1}\right)(1-p_k)^{-1}
\right\}.
\]
\end{definitionv}

\begin{definitionv}{def:conservative-variance-estimator}[Conservative variance estimator $\widehat V_{\mathrm{cons}}(G_n,p)$]
Define
\[
\widehat V_{\mathrm{cons}}(G_n,p)
=
V_{\mathrm{env}}(G_n,p).
\]
\end{definitionv}

\begin{definitionv}{def:surrogate-design}[Surrogate design $p_n^{\deg}(G_n)$]
Whenever the minimizer set is nonempty, define
\[
p_n^{\deg}(G_n)\in
\operatorname{argmin}_{p\in\mathcal P_{n,B_n,\epsilon}}
\sum_{k\in I_n} h_k(G_n)
\left\{p_k^{-1}+(1-p_k)^{-1}\right\}.
\]
\end{definitionv}

\begin{definitionv}{def:approximation-ratio}[Approximation ratio $\alpha_{\mathrm{cert}}(G_n)$]
Define
\[
\alpha_{\mathrm{cert}}(G_n)=
\begin{cases}
\dfrac{V_{\mathrm{env}}\!\left(G_n,p_n^{\deg}(G_n)\right)}
{V_{\mathrm{env}}\!\left(G_n,p_n^*(G_n)\right)},
& \text{if }0<V_{\mathrm{env}}\!\left(G_n,p_n^*(G_n)\right),\\
1,
& \text{otherwise}.
\end{cases}
\]
Here, each design selector equals its respective feasible minimizer when one exists, and equals the homogeneous budget vector otherwise.
\end{definitionv}

\begin{theoremv}{prop:homogeneous-reduction}[Homogeneous Bernoulli Reduction]
Under \cref{obj:ass:independent-heterogeneous-bernoulli}, suppose that the assignment probabilities are constant, \(p_k=p\) for every \(k\in I_n\), for some \(p\in(0,1)\). Then, for every \(i,j\in O_n\),
\[
r_{ij}^1(G_n,p)=
\begin{cases}
p^{-|S_{ij}(G_n)|}-1, & |S_{ij}(G_n)|>0,\\
0, & |S_{ij}(G_n)|=0,
\end{cases}
\qquad
r_{ij}^0(G_n,p)=
\begin{cases}
(1-p)^{-|S_{ij}(G_n)|}-1, & |S_{ij}(G_n)|>0,\\
0, & |S_{ij}(G_n)|=0.
\end{cases}
\]
Moreover,
\[
\sigma_{G_n,p}^2(Y)
=
\frac{1}{n}\sum_{i\in O_n}\sum_{j\in O_n}
\left[
r_{ij}^1(G_n,p)(Y_i^1-\mu_1)(Y_j^1-\mu_1)
+r_{ij}^0(G_n,p)(Y_i^0-\mu_0)(Y_j^0-\mu_0)
+2r_{ij}^{10}(G_n)(Y_i^1-\mu_1)(Y_j^0-\mu_0)
\right].
\]
\end{theoremv}

\begin{theoremv}{thm:hetero-envelope}[Heterogeneous envelope kernel]
Suppose that:
\begin{itemize}
\item \textbf{(Assignment probabilities.)} $0<p_k<1$ for every $k\in I_n$.
\item \textbf{(Design.)} \cref{obj:ass:independent-heterogeneous-bernoulli} holds with assignment-probability vector $p$.
\item \textbf{(Outcomes.)} \cref{obj:ass:bounded-outcomes} holds.
\end{itemize}
Then, for every $i,j\in O_n$,
\[
\mathbb E_p\!\left[\eta_i(p,Z)\eta_j(p,Z)\right]
=
r_{ij}^1(G_n,p)(Y_i^1-\mu_1)(Y_j^1-\mu_1)
+r_{ij}^0(G_n,p)(Y_i^0-\mu_0)(Y_j^0-\mu_0)
+r_{ij}^{10}(G_n)\!\left\{(Y_i^1-\mu_1)(Y_j^0-\mu_0)+(Y_i^0-\mu_0)(Y_j^1-\mu_1)\right\}.
\]
Moreover,
\[
\sigma^2_{G_n,p}(Y)\le V_{\mathrm{env}}(G_n,p).
\]
\end{theoremv}

\begin{theoremv}{thm:convex-design}[Convex feasible design]
Let $G_n$ be a bipartite experiment with intervention-unit set $I_n$, and let $\epsilon,B_n\in\mathbb R$. Suppose that
\begin{itemize}
\item \textbf{(Positivity margin.)} $0<\epsilon<1/2$;
\item \textbf{(Feasible budget.)} $m_n\epsilon\le B_n\le m_n(1-\epsilon)$.
\end{itemize}
Then $\mathcal P_{n,B_n,\epsilon}$ is nonempty, compact, and convex, and $p\mapsto V_{\mathrm{env}}(G_n,p)$ is convex on $\mathcal P_{n,B_n,\epsilon}$. Moreover, there exists a minimizer $p_n^*(G_n)\in\mathcal P_{n,B_n,\epsilon}$ such that
\[
V_{\mathrm{env}}(G_n,p_n^*(G_n))
\le V_{\mathrm{env}}(G_n,q)
\qquad\text{for every }q\in\mathcal P_{n,B_n,\epsilon}.
\]
For every such minimizer $p_n^*(G_n)$, there exist $\lambda_n\in\mathbb R$ and functions $\nu^+_{\cdot,n},\nu^-_{\cdot,n}:I_n\to\mathbb R$ such that, for every $k\in I_n$,
\[
\nu^+_{k,n}\ge0,\qquad \nu^-_{k,n}\ge0,\qquad
g_k(G_n,p_n^*(G_n))=\lambda_n-\nu^+_{k,n}+\nu^-_{k,n},
\]
and
\[
\nu^+_{k,n}\bigl((p_n^*(G_n))_k-(1-\epsilon)\bigr)=0,
\qquad
\nu^-_{k,n}\bigl(\epsilon-(p_n^*(G_n))_k\bigr)=0.
\]
\end{theoremv}

\begin{lemmav}{lem:denominator-positivity}[Denominator Degeneracy Bound]
Let $\epsilon\in(0,1/2)$, and let $(G_n)_{n\geq 0}$ be a sequence of bipartite experiments with intervention sets $I_n$ and outcome sets $O_n$. Suppose that:

\begin{itemize}

\item \textbf{(Outcome-population indexing.)} $|O_n|=n$ for every $n\geq 0$.

\item \textbf{(Admissible budgets.)} For every $n\geq 0$,
\[
|I_n|\epsilon\leq B_n\leq |I_n|(1-\epsilon).
\]

\item \textbf{(Bounded degrees.)} \cref{obj:ass:bounded-outcome-degree} holds with constant $\bar d$.

\item \textbf{(Bounded overlap dependency.)} \cref{obj:ass:bounded-overlap-dependency} holds with constant $\bar D$.
\end{itemize}

Then there exists a constant $C\geq 0$ such that, for all sufficiently large $n$, every $q=(q_k)_{k\in I_n}$ with $0\leq q_k\leq1$ for all $k\in I_n$ and $q\in\mathcal P_{n,B_n,\epsilon}$ as in \cref{obj:def:feasible-designs} satisfies
\[
\Pr_q\!\left(D_1(q,Z)=0\ \text{or}\ D_0(q,Z)=0\right)
\leq \frac{C}{n},
\]
where $Z$ has independent Bernoulli coordinates with assignment probabilities $q$, and $D_1(q,Z)$ and $D_0(q,Z)$ are as in \cref{obj:def:hajek-denominators}.
\end{lemmav}

\begin{theoremv}{thm:hetero-clt}[Heterogeneous Hájek CLT]
Consider a sequence of bipartite experiments indexed by $n$. Suppose:
\begin{itemize}
\item \textbf{(Outcome indexing.)} For all sufficiently large $n$, $|O_n|=n$.
\item \textbf{(Assignment and interference.)} \cref{obj:ass:independent-heterogeneous-bernoulli} and \cref{obj:ass:bipartite-interference} hold at every $n$ for the assignment design $D_n$ and probabilities $p_n$.
\item \textbf{(Outcome and graph regularity.)} \cref{obj:ass:bounded-outcomes}, \cref{obj:ass:bounded-outcome-degree}, and \cref{obj:ass:bounded-overlap-dependency} hold at every $n$, with constants $\bar d$ and $\bar D$, respectively.
\item \textbf{(Feasibility and optimality.)} For every $n$, $p_n\in\mathcal P_{n,B_n,\epsilon_n}$ as in \cref{obj:def:feasible-designs}, and
\[
p_n=p_n^*(G_n)
\]
as in \cref{obj:def:optimal-design}.
\item \textbf{(Uniform positivity.)} There exists $\epsilon_0>0$ such that $\epsilon_0\le\epsilon_n$ for all sufficiently large $n$.
\item \textbf{(Variance.)} \cref{obj:ass:variance-nondegenerate} holds for $\sigma^2_{G_n,p_n}(Y)$ as in \cref{obj:def:variance-scale}.
\end{itemize}
Then, for every $\delta>0$,
\[
\Pr_{D_n}\!\left(
\left|
\sqrt n\bigl\{\widehat\tau_H(p_n^*(G_n))-\tau_n\bigr\}
-n^{-1/2}\sum_{i\in O_n}\eta_i(p_n^*(G_n),Z)
\right|\ge\delta
\right)\longrightarrow 0,
\]
and, for every $s\in\mathbb R$,
\[
\Pr_{D_n}\!\left(
\frac{\sqrt n\bigl\{\widehat\tau_H(p_n^*(G_n))-\tau_n\bigr\}}
{\sqrt{\sigma^2_{G_n,p_n^*(G_n)}(Y)}}\le s
\right)
\longrightarrow
\Pr\!\left\{N(0,1)\le s\right\}.
\]
\end{theoremv}

\begin{theoremv}{thm:postdesign-wald}[Post-design Wald coverage]
For a sequence of bipartite experiments $G_n$ with outcome sets $O_n$, finite assignment designs, probability vectors $p_n$, positivity margins $\epsilon_n$, and budgets $B_n$, suppose:
\begin{itemize}
\item \textbf{(Outcome indexing.)} $|O_n|=n$ for all sufficiently large $n$.
\item \textbf{(Assignment law.)} The assignment design satisfies \cref{obj:ass:independent-heterogeneous-bernoulli} with $0\le p_{n,k}\le 1$ for every $n$ and $k\in I_n$.
\item \textbf{(Interference and boundedness.)} \cref{obj:ass:bipartite-interference}, \cref{obj:ass:bounded-outcomes}, \cref{obj:ass:bounded-outcome-degree}, and \cref{obj:ass:bounded-overlap-dependency} hold with constants $\bar d$ and $\bar D$.
\item \textbf{(Feasibility and optimality.)} For every $n$, $p_n$ is a feasible design in the sense of \cref{obj:def:feasible-designs}, $\epsilon_n\in(0,1/2)$, and
\[
p_n=p_n^*(G_n)
\]
in the sense of \cref{obj:def:optimal-design}.
\item \textbf{(Uniform positivity.)} There exists $\epsilon_0>0$ such that $\epsilon_0\le\epsilon_n$ for all sufficiently large $n$.
\item \textbf{(Nondegenerate variance.)} \cref{obj:ass:variance-nondegenerate} holds for $\sigma^2_{G_n,p_n}(Y)$.
\item \textbf{(Wald critical value.)} $\alpha_{\mathrm{cov}}\in(0,1)$, $z_{1-\alpha_{\mathrm{cov}}/2}\ge0$, and
\[
\Phi\!\left(z_{1-\alpha_{\mathrm{cov}}/2}\right)
=1-\frac{\alpha_{\mathrm{cov}}}{2}.
\]
\end{itemize}
Then, for every $n$,
\[
\sigma^2_{G_n,p_n}(Y)\le \widehat V_{\mathrm{cons}}(G_n,p_n),
\]
where $\sigma^2_{G_n,p_n}(Y)$ and $\widehat V_{\mathrm{cons}}(G_n,p_n)$ are as in \cref{obj:def:variance-scale} and \cref{obj:def:conservative-variance-estimator}, respectively; moreover,
\[
1-\alpha_{\mathrm{cov}}
\le
\liminf_{n\to\infty}
\Pr_{p_n}\!\left(
\left|\tau_n-\widehat\tau_H(p_n)\right|
\le
z_{1-\alpha_{\mathrm{cov}}/2}
\sqrt{\frac{\widehat V_{\mathrm{cons}}(G_n,p_n)}{n}}
\right).
\]
\end{theoremv}

\begin{theoremv}{thm:dispersion-certificate-unbounded}[Unbounded dispersion certificate]
For every $\epsilon\in(0,1/2)$, $c_{\mathrm{disp}}>0$, and $C_{\mathrm{disp}}\geq 1$, there exist sequences of finite intervention-unit sets $I_n$, finite outcome-unit sets $O_n$, finite bipartite experiments $G_n$, and budgets $B_n$ such that, for every $n$,

\begin{itemize}

\item \textbf{(Admissible budget.)} With $m_n=|I_n|$,
\[
m_n\epsilon\leq B_n\leq m_n(1-\epsilon).
\]

\item \textbf{(Positive energy.)}
\[
0<\sum_{k\in I_n}s_k^2.
\]

\end{itemize}

Moreover, for all sufficiently large $n$,

\begin{itemize}

\item \textbf{(Degree dispersion.)} For every $k\in I_n$,
\[
s_k^2\leq c_{\mathrm{disp}}\sum_{l\in I_n}s_l^2.
\]

\item \textbf{($h$-weight ratio.)} For every $k,l\in I_n$,
\[
h_l(G_n)>0\ \Longrightarrow\ h_k(G_n)\leq C_{\mathrm{disp}}\,h_l(G_n).
\]

\end{itemize}

Finally, the approximation ratio of \cref{obj:def:approximation-ratio} satisfies
\[
\alpha_{\mathrm{cert}}(G_n)\longrightarrow\infty .
\]
\end{theoremv}

\begin{theoremv}{thm:heterogeneity-separation}[Heterogeneity separation]
Let \(G_n\) be a bipartite experiment with \(m_n=|I_n|>0\), and let \(\epsilon\in(0,1/2)\). Suppose that:
\begin{itemize}
\item \textbf{(Admissible budget.)} \(m_n\epsilon\le B_n\le m_n(1-\epsilon)\).
\item \textbf{(Homogeneous rate.)} \(\rho=B_n/m_n\), with \(\epsilon<\rho<1-\epsilon\) and \(\rho\neq 1/2\).
\item \textbf{(Homogeneous design.)} \(p^{\mathrm{hom}}_k=\rho\) for every \(k\in I_n\).
\end{itemize}
Then all of the following hold:
\begin{itemize}
\item If there exist \(a,b\in I_n\) such that
\[
g_a(G_n,p^{\mathrm{hom}})\neq g_b(G_n,p^{\mathrm{hom}}),
\]
then \(p^{\mathrm{hom}}\) is not a minimizer of \(V_{\mathrm{env}}(G_n,\cdot)\) over \(\mathcal P_{n,B_n,\epsilon}\). Moreover, there exist \(a,b\in I_n\) attaining, respectively, the maximum and minimum homogeneous-point gradient scores, such that
\[
\Delta_g:=g_a(G_n,p^{\mathrm{hom}})-g_b(G_n,p^{\mathrm{hom}})>0.
\]
For every \(p_n^*(G_n)\in\operatorname{argmin}_{p\in\mathcal P_{n,B_n,\epsilon}}V_{\mathrm{env}}(G_n,p)\),
\[
p_n^*(G_n)\neq p^{\mathrm{hom}},\qquad
V_{\mathrm{env}}(G_n,p_n^*(G_n))<V_{\mathrm{env}}(G_n,p^{\mathrm{hom}}),
\]
and
\[
V_{\mathrm{env}}(G_n,p^{\mathrm{hom}})-V_{\mathrm{env}}(G_n,p_n^*(G_n))
\ge
2\Delta_g
\begin{cases}
\min\{\rho-\epsilon,\,1-\epsilon-\rho\}, & L_{ab}=0,\\
\min\!\left\{\min\{\rho-\epsilon,\,1-\epsilon-\rho\},\,\Delta_g/L_{ab}\right\}, & L_{ab}\neq0,
\end{cases}
\]
where \(L_{ab}\) is the directional modulus along \(e_b-e_a\).

\item If \(|N_i(G_n)|=1\) for every \(i\in O_n\), then, for every \(k\in I_n\),
\[
g_k(G_n,p^{\mathrm{hom}})
=
n^{-1}s_k^2\left((1-\rho)^{-2}-\rho^{-2}\right).
\]

\item If \(|N_i(G_n)|=1\) for every \(i\in O_n\) and there exist \(a,b\in I_n\) such that \(s_a^2\neq s_b^2\), then the preceding non-homogeneity, strict-improvement, and gap conclusions hold.
\end{itemize}
\end{theoremv}

\begin{theoremv}{thm:surrogate-certificate}[Surrogate approximation certificate]
For a bipartite experiment on \(G_n\), suppose:
\begin{itemize}
\item \textbf{(Admissible floor.)} \(0<\epsilon<1/2\).
\item \textbf{(Bounded outcome degree.)} \cref{obj:ass:bounded-outcome-degree} holds with bound \(\bar d\).
\item \textbf{(Admissible budget.)} \(m_n\epsilon\le B_n\le m_n(1-\epsilon)\).
\end{itemize}
Then, for every \(p\in\mathcal P_{n,B_n,\epsilon}\) of \cref{obj:def:feasible-designs},
\[
\sum_{k\in I_n}h_k(G_n)\!\left\{p_k^{-1}+(1-p_k)^{-1}\right\}
\le \frac{V_{\mathrm{env}}(G_n,p)}{4}
\le
\max\!\left\{1,\epsilon^{-(\bar d-1)}\right\}
\sum_{k\in I_n}h_k(G_n)\!\left\{p_k^{-1}+(1-p_k)^{-1}\right\}.
\]
Moreover, the approximation ratio of \cref{obj:def:approximation-ratio} satisfies
\[
\alpha_{\mathrm{cert}}(G_n)\le
\max\!\left\{1,\epsilon^{-(\bar d-1)}\right\}.
\]
\end{theoremv}

\begin{lemmav}{lem:bounded-degree-dependency-clt}[Bounded-Degree Dependency CLT]
Let \(\{X_{n,i}: i\in\iota_n\}\) be real-valued random variables, where \(|\iota_n|=n\), under the usual measurability and integrability conditions. Suppose:
\begin{itemize}
\item \textbf{(Centered summands.)} For every \(n\) and \(i\in\iota_n\), \(\mathbb E[X_{n,i}]=0\).
\item \textbf{(Dependency degree.)} The variables admit a dependency graph whose neighborhood of every \(i\in\iota_n\) has cardinality at most \(D\), for a fixed \(D\in\mathbb N\).
\item \textbf{(Uniform bound.)} There is an \(M\ge 0\) such that \(|X_{n,i}|\le M\) for every \(n\), \(i\in\iota_n\), and outcome.
\item \textbf{(Variance growth.)} Writing \(S_n=\sum_{i\in\iota_n}X_{n,i}\), suppose \(\mathbb E[S_n^2]=v_n\) for every \(n\), and, for some \(c>0\), \(v_n\ge cn\) for all sufficiently large \(n\).
\end{itemize}
Then, for every \(s\in\mathbb R\),
\[
\lim_{n\to\infty}\Pr\!\left(\frac{S_n}{\sqrt{v_n}}\le s\right)
=
\Pr\!\left(Z\le s\right),
\qquad Z\sim\mathcal N(0,1).
\]
\end{lemmav}
