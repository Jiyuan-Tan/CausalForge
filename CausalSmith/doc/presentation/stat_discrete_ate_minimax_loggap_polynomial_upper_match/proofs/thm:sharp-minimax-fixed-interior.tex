\begin{proof}[Proof of \cref{obj:thm:sharp-minimax-fixed-interior}]
\begin{enumerate}
\item First record the upper estimate for the hybrid estimator.  \Cref{obj:lem:light-cell-polynomial} gives constants \(C_L,c_L>0\) for the light-cell error, and \cref{obj:lem:universal-heavy-cell-rate} gives constants \(C_H,c_H>0\) and a cutoff \(N_U\) for the heavy-cell error.  Set
\[
C_U=2(C_L+C_H),\qquad \rho_U=\min\{c_L,c_H\}.
\]
For any admissible product law satisfying \(n\ge N_U\) and
\[
d\le \rho_U n\log n,
\]
the light and heavy bounds apply simultaneously.  Pointwise, the truncation in \(\widehat\tau_n^{\mathrm{hyb}}\) can only reduce squared error because \(\tau(P)\in[-1,1]\).  Also
\[
\sum_{k\in\widehat{\mathcal H}_n}\phi(q_k)
+
\sum_{k\in\widehat{\mathcal L}_n}\phi(q_k)
=
\tau(P),
\]
and hence
\[
(\widehat\tau_n^{\mathrm{hyb}}-\tau(P))^2
\le
2\left(\widehat T_{\mathcal H}-\sum_{k\in\widehat{\mathcal H}_n}\phi(q_k)\right)^2
+
2\left(\widehat T_{\mathcal L}-\sum_{k\in\widehat{\mathcal L}_n}\phi(q_k)\right)^2 .
\]
After integration and applying the two component bounds,
\[
E_P\!\left[(\widehat\tau_n^{\mathrm{hyb}}-\tau(P))^2\right]
\le
C_U\left\{\frac1n+\frac{d^2}{n^2(\log n)^2}\right\}.
\]
Taking the supremum over the experiment class gives the same bound for the worst-case risk of \(\widehat\tau_n^{\mathrm{hyb}}\), and the minimax risk is no larger because the hybrid estimator is one admissible estimator.

\item The lower-bound input is \cref{obj:lem:ate-lower-bound-transfer}, obtained by transferring the fixed-sample one-arm lower bound of \citep[Theorem 2; Appendix C.5; Lemma 4; Appendix D.2]{ZengBalakrishnanHanKennedy2024}.  It gives constants \(a>0\), \(\rho_L>0\), and \(N_L<\infty\) such that, whenever \(d>0\), \(n\ge N_L\), and
\[
d\le \rho_L n\log n,
\]
one has
\[
\mathsf R_{n,d,\epsilon}
\ge
 a\left\{\frac1n+\frac{d^2}{n^2(\log n)^2}\right\}.
\]
Now define
\[
a_\epsilon=a,\qquad
\rho_\epsilon=\min\{\rho_U,\rho_L\},\qquad
C_\epsilon=C_U,\qquad
N_\epsilon=\max\{N_U,N_L\}.
\]
These constants are positive, with \(N_\epsilon<\infty\).

\item Under the hypotheses of the sharp sandwich, \(n\ge N_\epsilon\) implies both \(n\ge N_U\) and \(n\ge N_L\), while
\[
d\le \rho_\epsilon n\log n
\]
implies both required growth bounds, with \(\rho_U\) and with \(\rho_L\).  The lower estimate from Step 2 and the upper estimate from Step 1 therefore give
\[
a_\epsilon\left\{\frac1n+\frac{d^2}{n^2(\log n)^2}\right\}
\le
\mathsf R_{n,d,\epsilon}
\le
\sup_{Q^{\otimes n}\in\mathcal E_{n,d,\epsilon}}
E_Q\!\left[(\widehat\tau_n^{\mathrm{hyb}}-\tau(Q))^2\right]
\le
C_\epsilon\left\{\frac1n+\frac{d^2}{n^2(\log n)^2}\right\}.
\]
This proves the sharp sandwich.

\item Fix \(K>0\) and put
\[
C_K=C_U(1+K^2).
\]
For the parametric window, keep the interior growth gate as part of the hypotheses: assume \(d>0\), \(n\ge N_\epsilon\),
\[
d\le K\sqrt n\,\log n,
\qquad
 d\le \rho_\epsilon n\log n.
\]
The first of these inequalities, together with \(d>0\) and \(K>0\), forces \(n>0\) and \(\log n>0\).  Squaring the \(K\sqrt n\log n\) bound and dividing by the positive denominator gives
\[
\frac{d^2}{n^2(\log n)^2}\le \frac{K^2}{n}.
\]
The lower bound uses the retained \(\rho_\epsilon n\log n\) gate to invoke the lower estimate:
\[
\frac{a_\epsilon}{n}
\le
a_\epsilon\left\{\frac1n+\frac{d^2}{n^2(\log n)^2}\right\}
\le
\mathsf R_{n,d,\epsilon}.
\]
The upper bound uses the same gate to invoke the hybrid upper estimate, and then the displayed ratio bound:
\[
\mathsf R_{n,d,\epsilon}
\le
C_U\left\{\frac1n+\frac{d^2}{n^2(\log n)^2}\right\}
\le
\frac{C_U(1+K^2)}{n}
=
\frac{C_K}{n}.
\]
This proves the parametric window under the stated interior growth condition.

\item Finally consider sequences \(n_j,d_j\) with \(n_j\to\infty\), eventually \(d_j>0\), and eventually
\[
d_j\le \rho_\epsilon n_j\log n_j.
\]
Set
\[
q_j=\frac{d_j}{n_j\log n_j}.
\]
Since \(n_j\to\infty\), we have \(1/n_j\to0\), and eventually \(n_j\ge2\), so \(q_j\ge0\).  Algebraically, for all sufficiently large \(j\),
\[
\frac1{n_j}+\frac{d_j^2}{n_j^2(\log n_j)^2}
=
\frac1{n_j}+q_j^2 .
\]
For those same indices, using a canonical exact-randomization law in the class to instantiate the upper estimate, Steps 1 and 2 give the two-sided bound
\[
a_\epsilon\left(\frac1{n_j}+q_j^2\right)
\le
\mathsf R_{n_j,d_j,\epsilon}
\le
C_U\left(\frac1{n_j}+q_j^2\right).
\]

If \(\mathsf R_{n_j,d_j,\epsilon}\to0\), the lower bound and \(a_\epsilon>0\) imply
\[
\frac1{n_j}+q_j^2\to0.
\]
Then \(0\le q_j^2\le n_j^{-1}+q_j^2\) eventually, so \(q_j^2\to0\).  Because \(q_j\ge0\) eventually, continuity of the square-root map gives \(q_j\to0\).

Conversely, if \(q_j\to0\), then \(q_j^2\to0\), and together with \(1/n_j\to0\) this yields
\[
\frac1{n_j}+q_j^2\to0.
\]
The minimax risk is nonnegative, and the upper bound above squeezes \(\mathsf R_{n_j,d_j,\epsilon}\) to \(0\).  This proves the consistency threshold and completes the proof.
\end{enumerate}
\end{proof}
