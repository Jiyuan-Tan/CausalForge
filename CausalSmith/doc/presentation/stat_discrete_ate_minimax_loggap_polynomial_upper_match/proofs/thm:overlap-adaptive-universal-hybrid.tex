\begin{proof}[Proof of \cref{obj:thm:overlap-adaptive-universal-hybrid}]
\begin{enumerate}
\item The computability assertion is obtained by taking the universal program constant
\(K=450\).  For each pair \((n,d)\), the corresponding real-arithmetic program
reads the hybrid count vector, evaluates exactly
\(\widehat\tau_n^{\mathrm{hyb}}\) on every sample \(O_1,\ldots,O_n\), and has
operation count at most \(450\,d\,M(n)^4\).  This is the required first clause,
with \(K=450\).

\item Fix \(0<\epsilon<1/2\).  \Cref{obj:lem:light-cell-polynomial}
gives constants \(C_L,c_L>0\) for the light-cell contribution, and
\cref{obj:lem:universal-heavy-cell-rate} gives constants
\(C_H,c_H>0\) and \(N_H\in\mathbb N\) for the heavy-cell contribution.  Set
\[
C_\epsilon=2(C_L+C_H),\qquad
\rho_\epsilon=\min\{c_L,c_H\},\qquad
N_\epsilon=N_H .
\]
Then \(C_\epsilon>0\) and \(\rho_\epsilon>0\).  If
\(n\ge N_\epsilon\) and \(d\le \rho_\epsilon n\log n\), then the same growth
condition holds with both \(c_L\) and \(c_H\).  For every
\(P^{\otimes n}\in\mathcal E_{n,d,\epsilon}\),
\[
\mathbb E_{P^{\otimes n}}
 \left[(\widehat\tau_n^{\mathrm{hyb}}-\tau(P))^2\right]
\le
2\,\mathbb E_{P^{\otimes n}}
 \left[\left(\widehat T_{\mathcal H}
       -\sum_{k\in\widehat{\mathcal H}_n}\phi(q_k)\right)^2\right]
+
2\,\mathbb E_{P^{\otimes n}}
 \left[\left(\widehat T_{\mathcal L}
       -\sum_{k\in\widehat{\mathcal L}_n}\phi(q_k)\right)^2\right].
\]
Applying the heavy- and light-cell bounds yields
\[
\mathbb E_{P^{\otimes n}}
 \left[(\widehat\tau_n^{\mathrm{hyb}}-\tau(P))^2\right]
\le
C_\epsilon\left\{\frac1n+
\frac{d^2}{n^2(\log n)^2}\right\}.
\]
Taking the supremum over \(P^{\otimes n}\in\mathcal E_{n,d,\epsilon}\)
gives the stated hybrid envelope.

\item With this hybrid bound in hand, apply
\cref{obj:lem:combined-upper-envelope}.  The selected estimator is an
oracle-calibrated measurable deterministic competitor in the minimax infimum, hence
\[
\mathsf R_{n,d,\epsilon}
\le
\sup_{P^{\otimes n}\in\mathcal E_{n,d,\epsilon}}
\mathbb E_{P^{\otimes n}}
 \left[(\widehat\tau^{\mathrm{sel}}_{C_\epsilon,\epsilon}
        -\tau(P))^2\right].
\]
Writing
\[
u=\frac{d^2}{n^2(\log n)^2},
\qquad
v=(1/2-\epsilon)^2,
\qquad
K_\epsilon=\max\{C_\epsilon,4\},
\]
the selector first gives the risk bound
\(\min\{C_\epsilon(1/n+u),\,1/n+4v\}\).  Splitting into the two cases
\(u\le v\) and \(v<u\), and using \(C_\epsilon\le K_\epsilon\) and
\(4\le K_\epsilon\), gives
\[
\sup_{P^{\otimes n}\in\mathcal E_{n,d,\epsilon}}
\mathbb E_{P^{\otimes n}}
 \left[(\widehat\tau^{\mathrm{sel}}_{C_\epsilon,\epsilon}
        -\tau(P))^2\right]
\le
\max\{C_\epsilon,4\}
\left\{
\frac1n+
\min\left[
\frac{d^2}{n^2(\log n)^2},(1/2-\epsilon)^2
\right]
\right\}.
\]

\item Now fix \(0<\epsilon\le 1/2\) and \(n,d>0\).  If
\(\mathcal E_{n,d,\epsilon}\) is empty, the supremum bound is trivial.  Otherwise,
for each \(P^{\otimes n}\in\mathcal E_{n,d,\epsilon}\),
\cref{obj:lem:near-randomization-linear-upper} gives
\[
\mathbb E_{P^{\otimes n}}
 \left[(\widehat\tau_{\mathrm{ctr}}-\tau(P))^2\right]
\le
\frac1n+4(1/2-\epsilon)^2 .
\]
Taking the supremum over the class proves the centered-estimator clause.

\item Finally, for every \(n,d>0\), \cref{obj:lem:randomized-endpoint-minimax}
gives directly
\[
\frac{1}{100n}\le \mathsf R_{n,d,1/2}\le \frac1n .
\]
Together with the preceding clauses, this proves the theorem.
\end{enumerate}
\end{proof}
