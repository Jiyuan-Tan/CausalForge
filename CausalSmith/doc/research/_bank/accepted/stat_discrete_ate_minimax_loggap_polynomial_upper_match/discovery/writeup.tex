\section*{stat\_\allowbreak{}discrete\_\allowbreak{}ate\_\allowbreak{}minimax\_\allowbreak{}loggap}

\noindent\textit{Specialization:} polynomial\_\allowbreak{}upper\_\allowbreak{}match\par

\paragraph{TL;DR.} One computable balanced ratio-polynomial hybrid, with universal numerical tuning \(A=6\), \(\lambda_0=256\), and \(b_0=4096\), attains the fixed-interior minimax MSE \(n^{-1}+d^2/(n^2\log^2 n)\) for every fixed \(0<\epsilon<1/2\) without taking \(\epsilon\) as input. Parametric risk holds through \(d=O(\sqrt n\log n)\), and consistency holds exactly for \(d=o(n\log n)\) in the proved range. For every \(n,d\), the centered estimator has MSE at most \(n^{-1}+4(1/2-\epsilon)^2\); at \(\epsilon=1/2\), minimax risk lies between \(1/(100n)\) and \(1/n\), uniformly in \(d\).

\section{Project justification}

\paragraph{Gap.} Zeng, Balakrishnan, Han, and Kennedy prove the strict-interior moment-matching lower scale, while the standard estimators they analyze leave a log-squared upper-bound gap.

\paragraph{Niche.} The missing upper kernel is a coupled four-cell Chebyshev reciprocal polynomial with unbiased fixed-multinomial factorial lifts on pilot-certified light cells, combined with an aggregate ratio estimator on heavy cells. The light-cell proof itself derives the exact \(A=6\) coefficient certificate and normalizes the within- and across-category covariances.

\paragraph{Fill.} The universal balanced two-split hybrid closes the fixed-interior gap. The converse is supplied by the source-backed fixed-sample control-zero lower-bound leaf and transfers because \(\tau=\psi_1\) on that subclass. Separate proved nodes give the all-\(n,d\) centered near-randomization bound \(n^{-1}+4(1/2-\epsilon)^2\), the randomized endpoint bracket \(1/(100n)\le\mathsf R_{n,d,1/2}\le1/n\), and, for \(n\ge N_\epsilon\), the combined upper envelope.

\section{Related work}

Zeng, Balakrishnan, Han, and Kennedy, \cite{ZengBalakrishnanHanKennedy2024}, Theorem 2, give the lower scale \(n^{-1}+d^2/(n^2\log^2 n)\) for fixed strict-interior overlap. Their Appendix C.5 gives the control-zero moment-matched construction and the \(d\lesssim n\log n\) range, while Lemma 4 and Appendix D.2 transfer the relaxed Poisson experiment to fixed sample size; on this subclass the treated mean equals the ATE. Jiao, Venkat, Han, and Weissman, \cite{JiaoVenkatHanWeissman2015}, Han, Jiao, and Weissman, \cite{HanJiaoWeissman2020}, and Wu and Yang, \cite{WuYang2016}, develop the large-alphabet architecture of best-polynomial approximation with unbiased factorial-moment lifting for nonsmooth discrete functionals. Higher-order influence-function estimators, such as \cite{LiuMukherjeeNeweyRobins2017}, address a different semiparametric architecture. Those precedents do not supply the coupled four-cell overlap-cone approximation needed here. The contribution proved here is the causal ATE construction: the exact \(A=6\) cone certificate, fixed-multinomial within- and cross-category covariance normalization, pilot false-light control, heavy-cell aggregation, and a single universal numerical calibration that matches the published converse.

\section{Assumptions}

\begin{assumption}[ass:iid-sampling]\label{ass:iid-sampling}
\(O_1,\ldots,O_n\stackrel{\mathrm{iid}}{\sim}P\).
\par\smallskip\noindent\emph{Standard} (independent and identically distributed sampling, \cite{ZengBalakrishnanHanKennedy2024}).
\end{assumption}

\begin{assumption}[ass:overlap]\label{ass:overlap}
\(\epsilon\le\pi_k\le1-\epsilon\) for every \(k\) with \(p_k>0\).
\par\smallskip\noindent\emph{Standard} (strong overlap, \cite{ZengBalakrishnanHanKennedy2024}).
\end{assumption}

\begin{assumption}[ass:consistency]\label{ass:consistency}
\(Y=Y(A)\) almost surely.
\par\smallskip\noindent\emph{Standard} (consistency, \cite{ZengBalakrishnanHanKennedy2024}).
\end{assumption}

\begin{assumption}[ass:conditional-exchangeability]\label{ass:conditional-exchangeability}
\((Y(0),Y(1))\perp A\mid X\).
\par\smallskip\noindent\emph{Standard} (conditional exchangeability, \cite{ZengBalakrishnanHanKennedy2024}).
\end{assumption}

\section{Definitions}

\begin{definition}[def:experiment-class]\label{def:experiment-class}
\par\smallskip\noindent\emph{Defined object.} \texttt{unrestricted discrete-confounder experiment class}
\par\smallskip\noindent\emph{Construction.} \(\mathcal E_{n,d,\epsilon}:=\{P^{\otimes n}: P\in\mathsf{Prob}(\{1,\ldots,d\}\times\{0,1\}^2),\ n,d\in\mathbb N,\ 0<\epsilon\le1/2,\text{ and }P\text{ satisfies }\mathrm{ass:overlap}\}\).
\par\smallskip\noindent\emph{Member properties.} \texttt{ass:overlap}.
\end{definition}

\begin{definition}[def:overlap-cone]\label{def:overlap-cone}
\par\smallskip\noindent\emph{Defined object.} \texttt{overlap-restricted four-cell cone}
\par\smallskip\noindent\emph{Construction.} \(\mathcal C_\epsilon:=\{u=(u_{00},u_{01},u_{10},u_{11})\in\mathbb R_+^4: \epsilon(u_{00}+u_{01}+u_{10}+u_{11})\le u_{10}+u_{11}\le(1-\epsilon)(u_{00}+u_{01}+u_{10}+u_{11})\}\).
\par\smallskip\noindent\emph{Inputs.} \(\epsilon\); \texttt{ass:overlap}.
\end{definition}

\begin{definition}[def:cell-functional]\label{def:cell-functional}
\par\smallskip\noindent\emph{Defined object.} \texttt{homogeneous four-cell ATE functional}
\par\smallskip\noindent\emph{Construction.} For \(u=(u_{00},u_{01},u_{10},u_{11})\in\mathcal C_\epsilon\), define \(\phi(u):=(u_{00}+u_{01}+u_{10}+u_{11})\left[\frac{u_{11}}{u_{10}+u_{11}}-\frac{u_{01}}{u_{00}+u_{01}}\right]\) when \(u\neq0\), and define \(\phi(0,0,0,0):=0\).
\par\smallskip\noindent\emph{Inputs.} \(\mathcal C_\epsilon\); \(q_k\); \texttt{ass:overlap}.
\end{definition}

\begin{definition}[def:ate-functional]\label{def:ate-functional}
\par\smallskip\noindent\emph{Defined object.} \texttt{average treatment effect functional}
\par\smallskip\noindent\emph{Construction.} \(\tau(P):=\sum_{k=1}^{d}p_k(\mu_{1k}-\mu_{0k})=\sum_{k=1}^{d}\phi(q_k)\).
\par\smallskip\noindent\emph{Inputs.} \(p_k\); \(\mu_{ak}\); \(\phi\); \(q_k\); \texttt{ass:consistency}; \texttt{ass:conditional-exchangeability}; \texttt{ass:overlap}.
\end{definition}

\begin{definition}[def:hybrid-estimator-handle]\label{def:hybrid-estimator-handle}
\par\smallskip\noindent\emph{Defined object.} \texttt{universally tuned balanced two-split ratio-polynomial hybrid estimator}
\par\smallskip\noindent\emph{Construction.} Set \(I_0=\{1,\ldots,\lfloor n/2\rfloor\}\), \(I_1=\{\lfloor n/2\rfloor+1,\ldots,n\}\), \(m_j=|I_j|\), and \(L=\log(en)\). Set \(A=6\), and put \(\lambda_0=256\), \(b_0=4096\), \(D_A=8\log(9A/8)=8\log(27/4)\), and
\[
\alpha_0=\min\left\{1,\frac1{32\log A},\frac1{256D_A}\right\}.
\]
For \(n\ge2\), put \(M(n)=\max\{2,\lfloor\alpha_0L\rfloor\}\) and \(B(n)=b_0L/m_1\). Let \(N_0\) be the least numerical cutoff such that, for every \(n\ge N_0\), \(\alpha_0L\ge2\), \(4M(n)^2\le m_1\), and \(4M(n)/m_1\le3B(n)/4\). For \(n\ge N_0\), abbreviate \(M=M(n)\), \(B=B(n)\), and set
\[
\widehat{\mathcal H}_n=\left\{k:\sum_{a,y}N^{(0)}_{aky}>\lambda_0L\right\},\qquad \widehat{\mathcal L}_n=\widehat{\mathcal H}_n^c.
\]
Put \(N_k^{(1)}=\sum_{a,y}N^{(1)}_{aky}\), \(N_{ak}^{(1)}=\sum_yN^{(1)}_{aky}\), and
\[
\widehat h_k=\frac{N_k^{(1)}}{m_1}\left\{\frac{N^{(1)}_{1k1}}{N^{(1)}_{1k}}\mathbf1\{N^{(1)}_{1k}>0\}-\frac{N^{(1)}_{0k1}}{N^{(1)}_{0k}}\mathbf1\{N^{(1)}_{0k}>0\}\right\}.
\]
Define \(H_M(x)=\{1-T_M(1-2x)\}/(2M^2)\), \(E_M(x)=H_M(x)/x\), and \(G_M(x)=\{1-E_M(x)\}/x\), using polynomial continuation at zero, and
\[
P_{M,B}(u)=\frac{s(u)t_1(u)}B G_M\!\left(\frac{s_1(u)}B\right)-\frac{s(u)t_0(u)}B G_M\!\left(\frac{s_0(u)}B\right).
\]
Here \(s_a(u)=u_{a0}+u_{a1}\), \(s(u)=s_0(u)+s_1(u)\), \(t_1(u)=u_{11}\), and \(t_0(u)=u_{01}\). Expand \(P_{M,B}(u)=\sum_{|r|\le M}c_ru^r\), set
\[
\widehat P_k=\sum_{|r|\le M}c_r\frac{\prod_{a,y}(N^{(1)}_{aky})_{r_{ay}}}{(m_1)_{|r|}},\quad \widehat T_{\mathcal H}=\sum_{k\in\widehat{\mathcal H}_n}\widehat h_k,\quad \widehat T_{\mathcal L}=\sum_{k\in\widehat{\mathcal L}_n}\widehat P_k,
\]
and define \(\widehat\tau_n^{\mathrm{hyb}}=\max\{-1,\min(1,\widehat T_{\mathcal H}+\widehat T_{\mathcal L})\}\). For \(n<N_0\), set \(\widehat{\mathcal L}_n=\varnothing\) and use the ratio branch on every category. The estimator uses no overlap constant and is computable in \(O(dM^4)\) arithmetic operations.
\par\smallskip\noindent\emph{Inputs.} \(O_i\); \(I_0,I_1\); \(N^{(j)}_{aky}\).
\end{definition}

\begin{definition}[def:minimax-risk]\label{def:minimax-risk}
\par\smallskip\noindent\emph{Defined object.} \texttt{minimax ATE risk}
\par\smallskip\noindent\emph{Construction.} \(\mathsf R_{n,d,\epsilon}:=\inf_{\widehat\tau}\sup_{P^{\otimes n}\in\mathcal E_{n,d,\epsilon}}E_P[(\widehat\tau-\tau(P))^2]\), where the infimum is over measurable estimators based on \(O_1,\ldots,O_n\).
\par\smallskip\noindent\emph{Inputs.} \(\mathcal E_{n,d,\epsilon}\); \(\tau(P)\).
\end{definition}

\begin{definition}[def:two-category-witness]\label{def:two-category-witness}
\par\smallskip\noindent\emph{Defined object.} \texttt{two-category confounding witness}
\par\smallskip\noindent\emph{Construction.} For \(d=2\), set \(p_1=p_2=1/2\), \(\pi_1=\epsilon\), \(\pi_2=1-\epsilon\), \(\mu_{01}=\mu_{02}=0\), \(\mu_{11}=1\), and \(\mu_{12}=0\). Define a full-data extension by drawing \((Y(0),Y(1))=(0,\mathbf 1\{X=1\})\) and then drawing \(A\mid X=k\sim\mathsf{Bernoulli}(\pi_k)\).
\par\smallskip\noindent\emph{Inputs.} \(\epsilon\); \(p_k\); \(\pi_k\); \(\mu_{ak}\); \(X\); \(A\); \(Y(a)\).
\end{definition}

\section{Main results}

\begin{proposition}[prop:two-category-confounding]\label{prop:two-category-confounding}
The full-data law in \(def:two-category-witness\) satisfies \(ass:consistency\), \(ass:conditional-exchangeability\), and \(ass:overlap\), has \(\tau(P)=1/2\), and has naive observed contrast \(E_P[Y\mid A=1]-E_P[Y\mid A=0]=\epsilon\).
\end{proposition}
\begin{proof}
Under the construction, \(Y(0)=0\) and \(Y(1)=\mathbf 1\{X=1\}\) are deterministic functions of \(X\), and the observed outcome is \(Y=Y(A)\). Thus \(ass:consistency\) holds. Conditional on \(X\), the potential-outcome vector is constant while \(A\sim\mathsf{Bernoulli}(\pi_X)\), so \((Y(0),Y(1))\perp A\mid X\), which is \(ass:conditional-exchangeability\). Since \(\pi_1=\epsilon\) and \(\pi_2=1-\epsilon\), \(ass:overlap\) also holds. Furthermore,
\[
\tau(P)=E_P[Y(1)-Y(0)]=P(X=1)=\frac12.
\tag{1}
\]
Also \(P(A=1)=\{\epsilon+(1-\epsilon)\}/2=1/2\), whereas \(P(A=1,Y=1)=P(X=1,A=1)=\epsilon/2\). Consequently,
\[
E_P[Y\mid A=1]=\epsilon.
\tag{2}
\]
Every unit with \(A=0\) has \(Y=Y(0)=0\), so \(E_P[Y\mid A=0]=0\). The naive observed contrast is therefore \(\epsilon\). For \(0<\epsilon<1/2\) it differs from the ATE by \(1/2-\epsilon\); at \(\epsilon=1/2\) the construction is randomized and the two coincide.
\end{proof}
\par\smallskip\noindent \textit{Justification.} The full-data coupling verifies consistency, conditional exchangeability, overlap, the ATE, and the observed contrast exactly. \textit{Gap.} Strict confounding occurs for \(0<\epsilon<1/2\); at \(\epsilon=1/2\) the witness becomes randomized. \textit{Consumer.} Causal interpretation.

\begin{lemma}[lem:light-cell-polynomial]\label{lem:light-cell-polynomial}
There are constants \(C_\epsilon,c_\epsilon>0\) and a computable calibration of \(def:hybrid-estimator-handle\) such that, for every \(d\le c_\epsilon n\log n\) and every \(P^{\otimes n}\in\mathcal E_{n,d,\epsilon}\), its light-cell contribution \(\widehat T_{\mathcal L}\) obeys \[ E_P\!\left[\left\{\widehat T_{\mathcal L}-\sum_{k\in\widehat{\mathcal L}_n}\phi(q_k)\right\}^2\right]\le C_\epsilon\left\{\frac{1}{n}+\frac{d^2}{n^2\log^2 n}\right\}. \] The calibration uses a polynomial approximation of \(\phi\) on the overlap-restricted four-cell cone and factorial-moment monomials from \(I_1\).
\end{lemma}
\begin{proof}
Put \(L=\log(en)\), \(m_0=\lfloor n/2\rfloor\), and \(m_1=n-m_0\). Use exactly the calibration in Definition~\ref{def:hybrid-estimator-handle}: \(A=6\), \(\lambda_0=256\), \(b_0=4096\), \(B=b_0L/m_1\), \(D_A=8\log(27/4)\), and, for \(n\ge N_0\), \(M=\lfloor\alpha_0L\rfloor\). For \(n<N_0\), \(\widehat{\mathcal L}_n=\varnothing\), so the asserted error is zero.

We first derive the approximation and coefficient bounds. Direct expansion of the shifted Chebyshev polynomial gives
\[
T_M(1-2x)=1+\sum_{j=1}^M(-1)^j\frac{M}{M+j}\binom{M+j}{2j}4^jx^j.
\tag{1}
\]
The identities \(T_M(1)=1\) and \(T_M'(1)=M^2\) show that \(E_M\) and \(G_M\) have polynomial continuations at zero, and division of (1) twice gives
\[
G_M(x)=\sum_{j=0}^{M-2}(-1)^j\frac{2^{2j+3}}{M(M+j+2)}\binom{M+j+2}{2j+4}x^j.
\tag{2}
\]
For \(0\le x\le1\), \(|T_M(1-2x)|\le1\), so the definitions imply
\[
x|1-xG_M(x)|=H_M(x)\le M^{-2}.
\tag{3}
\]
If \(u\in\mathcal C_\epsilon\) and \(\|u\|_1\le B\), then \(s_a(u)\ge\epsilon s(u)\) and \(0\le t_a(u)\le s_a(u)\). Applying (3) at \(x_a=s_a(u)/B\) yields
\[
|P_{M,B}(u)-\phi(u)|\le\sum_{a=0}^1\frac{s(u)t_a(u)B}{s_a(u)^2}H_M(x_a)\le\frac{2B}{\epsilon M^2}.
\tag{4}
\]
The same claim is immediate at \(u=0\).

Write \(P_{M,B}(u)=\sum_rc_ru^r\) and \(\mathfrak A(v)=\sum_r|c_r|v^r\) for \(v\in\mathbb R_+^4\). Alternation in (2) gives
\[
G_M^+(z):=G_M(-z)=\frac{T_M(1+2z)-1-2M^2z}{2M^2z^2}.
\tag{5}
\]
For \(R=\|v\|_1\) and \(z=R/B\), expanding the factors \(s,t_a,s_a^j\) and replacing every coefficient by its absolute value gives the coefficientwise domination
\[
\mathfrak A(v)\le\frac{R^2}{B}G_M^+(z).
\tag{6}
\]
For \(z\le1\), positivity of the coefficients of \(G_M^+\) and the closed form \(T_M(t)=\{(t+\sqrt{t^2-1})^M+(t-\sqrt{t^2-1})^M\}/2\) for \(t\ge1\) give \(G_M^+(z)\le G_M^+(1)\le T_M(3)\le6^M\). For \(z\ge1\), \(G_M^+\) has degree \(M-2\) and nonnegative coefficients, hence \(G_M^+(z)\le z^{M-2}G_M^+(1)\le6^Mz^{M-2}\). Therefore
\[
\mathfrak A(v)\le B6^M\max\{z^2,z^M\},\qquad \mathfrak A(v)\le\frac{R^2}{B}6^M\quad(z\le1).
\tag{7}
\]
Equations (1)--(7) derive the numerical coefficient certificate \(A=6\) used in the estimator.

For a multi-index \(r\), define
\[
U_{k,r}:=\frac{\prod_{a,y}(N^{(1)}_{aky})_{r_{ay}}}{(m_1)_{|r|}}.
\tag{8}
\]
Counting ordered selections of \(|r|\) distinct observations from the fixed multinomial block gives
\[
E_P[U_{k,r}]=q_k^r.
\tag{9}
\]
The elementary identity
\[
(z)_a(z)_b=\sum_{h=0}^{a\wedge b}\binom ah\binom bhh!(z)_{a+b-h}
\tag{10}
\]
applied coordinatewise controls products. If \(|r|,|s|\le M\), \(H=|h|\), and \(4M^2\le m_1\), then
\[
\frac{(m_1)_{|r|+|s|-H}}{(m_1)_{|r|}(m_1)_{|s|}}\le\frac{e}{m_1^H}.
\tag{11}
\]
Indeed, after cancelling the factors corresponding to two disjoint ordered selections, their ratio is at most \((m_1-|s|)^{-H}\); moreover \(\{m_1/(m_1-|s|)\}^H\le e\) because \(H,|s|\le M\) and \(4M^2\le m_1\). Since \(\binom{r_j}{h_j}h_j!m_1^{-h_j}\le(M/m_1)^{h_j}\), summing the coordinatewise expansion (10) over \(h\) gives
\[
E_P[U_{k,r}U_{k,s}]\le e\,q_k^r\left(q_k+\frac{M}{m_1}\mathbf1_4\right)^s.
\tag{12}
\]
This inequality also covers zero coordinates, since both sides are polynomials with nonnegative coefficients. For \(k\ne\ell\), selected observations carrying distinct category labels cannot overlap, and the same ordered-selection count gives the exact identity
\[
E_P[U_{k,r}U_{\ell,s}]=\frac{(m_1)_{|r|+|s|}}{(m_1)_{|r|}(m_1)_{|s|}}q_k^rq_\ell^s.
\tag{13}
\]
Writing the factorial ratio as a product and using \(1-\prod_j(1-x_j)\le\sum_jx_j\) yields
\[
\left|\frac{(m_1)_{|r|+|s|}}{(m_1)_{|r|}(m_1)_{|s|}}-1\right|\le\frac{2M^2}{m_1}.
\tag{14}
\]

Let \(S\subseteq\{k:p_k\le B/4\}\) be deterministic. The cutoff conditions in Definition~\ref{def:hybrid-estimator-handle} imply
\[
\left\|q_k+\frac{M}{m_1}\mathbf1_4\right\|_1\le\frac B4+\frac{4M}{m_1}\le B.
\tag{15}
\]
Summing (12) against \(|c_rc_s|\) and applying (7) gives \(E_P[\widehat P_k^2]\le CB^26^{2M}\). Equations (13)--(14), followed again by (7), give, for \(k\ne\ell\),
\[
|\operatorname{Cov}_P(\widehat P_k,\widehat P_\ell)|\le C\frac{M^2}{m_1}B^26^{2M}.
\tag{16}
\]
Consequently,
\[
\operatorname{Var}_P\left(\sum_{k\in S}\widehat P_k\right)\le C6^{2M}\left\{dB^2+\frac{d^2M^2B^2}{m_1}\right\}.
\tag{17}
\]
Let \(\kappa=2\alpha_0\log6\le1/16\). Since \(6^{2M}\le Cn^\kappa\), \(B\asymp L/n\), and \(M\le\alpha_0L\),
\[
n^\kappa d\frac{L^2}{n^2}\le C\left\{\frac1n+\frac{d^2}{n^2L^2}\right\},\qquad n^\kappa d^2\frac{L^4}{n^3}\le C\frac{d^2}{n^2L^2}.
\tag{18}
\]
For the first inequality, put \(x=d/(\sqrt nL)\), apply \(2x\le1+x^2\), and use boundedness of \(n^{\kappa-1/2}L^3\). The second follows from boundedness of \(n^{\kappa-1}L^6\).

By (9), \(\widehat P_k\) is unbiased for \(P_{M,B}(q_k)\). Overlap gives \(q_k\in\mathcal C_\epsilon\), while \(M\ge\alpha_0L/2\) for \(n\ge N_0\). Hence (4) implies
\[
\left|E_P\sum_{k\in S}\widehat P_k-\sum_{k\in S}\phi(q_k)\right|^2\le\frac{4d^2B^2}{\epsilon^2M^4}\le C_\epsilon\frac{d^2}{n^2L^2}.
\tag{19}
\]
Equations (17)--(19) prove the required bound for every deterministic genuinely light set.

It remains to control categories falsely classified as light. Put \(B_+=2\lambda_0L/m_0\). Balanced splitting and \(b_0=16\lambda_0\) imply \(B_+\le B/4\). If \(p_k>B_+\), then \(m_0p_k>2\lambda_0L\). For \(N_k^{(0)}\sim\mathsf{Binomial}(m_0,p_k)\), exponential Markov inequality applied to \(e^{-tN_k^{(0)}}\), optimized at the usual binomial lower-tail value, gives
\[
P\{N_k^{(0)}\le\lambda_0L\}\le e^{-m_0p_k/8}.
\tag{20}
\]
Set \(z=p_k/B\), so \(z\ge1/8\). From (12), (7), and \(4M/m_1\le3B/4\),
\[
\|\widehat P_k\|_2+|\phi(q_k)|\le CB\{6(1+z)\}^{M}.
\tag{21}
\]
Indeed, the two coefficient envelopes are bounded by \(B6^M(1+z)^M\), and \(|\phi(q_k)|\le p_k=Bz\). The function \(z\mapsto z^{-1}\log\{6(1+z)\}\) is decreasing, so, for \(z\ge1/8\),
\[
\log\{6(1+z)\}\le D_Az,\qquad D_A=8\log(27/4).
\tag{22}
\]
The pilot and estimation blocks are independent. Minkowski's inequality and (20)--(22) give
\[
\left\|\sum_{k:p_k>B_+}\mathbf1\{N_k^{(0)}\le\lambda_0L\}\{\widehat P_k-\phi(q_k)\}\right\|_2\le\sum_{k:p_k>B_+}CB\exp\left\{MD_Az-\frac{m_0Bz}{16}\right\}.
\tag{23}
\]
The degree inequality is \(MD_A\le\alpha_0LD_A\le L/256\), and balanced splitting gives \(m_0B/16\ge128L\) for \(n\ge N_0\). Since \(z\ge1/8\), the right side of (23) is at most
\[
CBd e^{-15L}.
\tag{24}
\]
Taking \(c_\epsilon=1\), the square of (24) is at most \(C_\epsilon n^{-2}\) uniformly for \(d\le c_\epsilon nL\).

Finally condition on the pilot block. The set
\[
S_0=\widehat{\mathcal L}_n\cap\{k:p_k\le B_+\}
\tag{25}
\]
is then deterministic for the independent estimation block and is contained in \(\{k:p_k\le B/4\}\), so (17)--(19) apply conditionally. All remaining selected categories are controlled by (23)--(24). Applying \((x+y)^2\le2x^2+2y^2\), averaging over the pilot, and using \(L\asymp\log n\) for \(n\ge2\) proves the displayed MSE bound. The constants in the approximation step may depend on the fixed \(\epsilon\), while the calibration itself is numerical, independent of \(\epsilon\), and computable as stated.
\end{proof}
\par\smallskip\noindent \textit{Justification.} Equations \(1\)--\(7\) derive the exact \(A=6\) Chebyshev certificate, and the remaining argument proves factorial-moment normalization, fixed-multinomial covariance control, approximation bias, and pilot false-light control. \textit{Gap.} No fixed-interior light-cell obligation remains. \textit{Consumer.} The fixed-interior upper bound.

\begin{lemma}[lem:pilot-sandwich]\label{lem:pilot-sandwich}
Let \(m_0\asymp n\), \(L=\log(en)\), \(\widehat{\mathcal H}=\{k:N_k^{(0)}>tL\}\), \(\widehat{\mathcal L}=\widehat{\mathcal H}^c\), \(B_-=tL/(2m_0)\), and \(B_+=2tL/m_0\). For every \(K>0\), sufficiently large numerical \(t\) gives, uniformly over \(d\le cnL\),
\[
P\left(\widehat{\mathcal H}\not\subseteq\{k:p_k\ge B_-\}\text{ or }\widehat{\mathcal L}\not\subseteq\{k:p_k\le B_+\}\right)\le C_Kn^{-K}.
\]
The choice \(t=256\) is valid for \(K=4\).
\end{lemma}
\begin{proof}
For \(N\sim\mathsf{Binomial}(m_0,p)\), exponential Markov inequality at exponent \(\log2\) gives, if \(m_0p<tL/2\),
\[
P(N>tL)\le\exp\{-tL(\log2-1/2)\}.
\tag{1}
\]
If \(m_0p>2tL\), the lower-tail Chernoff bound gives \(P(N\le tL)\le e^{-m_0p/8}\le e^{-tL/4}\). Apply these bounds cellwise and take a union bound over at most \(cnL\) cells. For general \(K\), choosing \(t\) so that \(t(\log2-1/2)>K+3\) and \(t/4>K+3\) makes the union bound at most \(C_Kn^{-K}\). The value \(t=256\) satisfies both inequalities for \(K=4\).
\end{proof}

\begin{lemma}[lem:universal-heavy-cell-rate]\label{lem:universal-heavy-cell-rate}
For the threshold \(\lambda_0=256\), every fixed \(0<\epsilon<1/2\) admits \(C_\epsilon,\rho_\epsilon>0\) and \(N_\epsilon<\infty\) such that, for \(n\ge N_\epsilon\), \(d\le\rho_\epsilon n\log n\), and \(P^{\otimes n}\in\mathcal E_{n,d,\epsilon}\),
\[
E_P\left[\left\{\widehat T_{\mathcal H}-\sum_{k\in\widehat{\mathcal H}_n}\phi(q_k)\right\}^2\right]\le C_\epsilon\left\{\frac1n+\frac{d^2}{n^2\log^2n}\right\}.
\]
\end{lemma}
\begin{proof}
Put \(L=\log(en)\), \(B_-=\lambda_0L/(2m_0)\), and \(\mathcal G=\{\widehat{\mathcal H}_n\subseteq\{k:p_k\ge B_-\}\}\). By Lemma~\ref{lem:pilot-sandwich}, \(P(\mathcal G^c)\le Cn^{-4}\). Condition on the pilot and \(\mathcal G\), so \(H=\widehat{\mathcal H}_n\) is fixed for \(I_1\) and \(p_k\ge B_-\) on \(H\).

For \(a\in\{0,1\}\), put \(\rho_{1k}=\pi_k\), \(\rho_{0k}=1-\pi_k\), and
\[
S_a=\sum_{k\in H}\frac{N_k}{m_1}\frac{N_{ak1}}{N_{ak}}\mathbf1\{N_{ak}>0\},\qquad \theta_a=\sum_{k\in H}p_k\mu_{ak}.
\tag{1}
\]
Here and below the counts have superscript \((1)\), which is suppressed. Given \(N_k=t\), \(N_{ak}\sim\mathsf{Binomial}(t,\rho_{ak})\). For \(D\sim\mathsf{Binomial}(t,\rho)\),
\[
E\left[\frac{\mathbf1\{D>0\}}D\right]\le2E[(D+1)^{-1}]=\frac{2\{1-(1-\rho)^{t+1}\}}{(t+1)\rho}\le\frac2{(t+1)\rho}.
\tag{2}
\]
Conditional on the category and treatment counts, Bernoulli outcome noise is independent across cells. Equation (2), overlap, and \(\sum_kE[N_k]\le m_1\) therefore bound the variance of the aggregate outcome-noise part of \(S_a\) by \(C/(\epsilon m_1)\).

The conditional mean given category and treatment counts is \(L_a-Q_a\), where
\[
L_a=m_1^{-1}\sum_{i\in I_1}\mu_{aX_i}\mathbf1\{X_i\in H\},\qquad Q_a=\sum_{k\in H}\frac{N_k}{m_1}\mu_{ak}\mathbf1\{N_{ak}=0\}.
\tag{3}
\]
We have \(E[L_a]=\theta_a\) and \(\operatorname{Var}(L_a)\le1/m_1\). Let \(r_k=p_k\rho_{ak}\), \(s_k=p_k(1-\rho_{ak})\), and let \(C_k\) count arm-\(1-a\) observations in category \(k\). On \(\{N_{ak}=0\}\), \(N_k=C_k\). Exact multinomial counting gives
\[
\begin{aligned}
E[C_k^2\mathbf1\{N_{ak}=0\}]&=m_1s_k(1-r_k)^{m_1-1}+(m_1)_2s_k^2(1-r_k)^{m_1-2},\\
E[C_kC_\ell\mathbf1\{N_{ak}=N_{a\ell}=0\}]&=(m_1)_2s_ks_\ell(1-r_k-r_\ell)^{m_1-2}.
\end{aligned}
\tag{4}
\]
Using \(r_k\ge\epsilon p_k\), \(s_k\le p_k\), and \((1-x)^j\le e^{-jx}\), the diagonal linear terms are at most \(1/m_1\) after division by \(m_1^2\), while all remaining diagonal and cross terms are bounded by a constant times the square below. Thus
\[
E[Q_a^2]\le\frac1{m_1}+C\left\{\sum_{k\in H}p_ke^{-c\epsilon m_1p_k}\right\}^2.
\tag{5}
\]
On \(H\), \(m_1p_k\ge c_0L\). Since \(\sup_{L\ge1,u\ge c_0L}Lu e^{-c\epsilon u}<\infty\), each summand in (5) is at most \(C_\epsilon/(m_1L)\). Hence (5) is at most \(m_1^{-1}+C_\epsilon d^2/(m_1^2L^2)\). The conditional variance decomposition, \((x+y+z)^2\le3(x^2+y^2+z^2)\), and the preceding bounds give the same order for \(S_a-\theta_a\); \((x-y)^2\le2x^2+2y^2\) combines the arms. Off \(\mathcal G\), \(|\widehat T_{\mathcal H}|\le1\) because its cell weights sum to at most one, and \(\left|\sum_{k\in\widehat{\mathcal H}_n}\phi(q_k)\right|\le1\), so squared error is at most four. Averaging, using \(P(\mathcal G^c)\le Cn^{-4}\), and \(m_1\asymp n\) proves the lemma.
\end{proof}
\par\smallskip\noindent \textit{Justification.} The inverse-binomial variance calculation and exact multinomial missing-arm moments control the aggregate ratio branch. \textit{Gap.} No fixed-interior heavy-cell obligation remains. \textit{Consumer.} The fixed-interior upper bound.

\begin{lemma}[lem:zeng-one-arm-minimax-lower]\label{lem:zeng-one-arm-minimax-lower}
Fix \(0<\epsilon<1/2\). Let \(\mathcal D_0^{(n,d)}(\epsilon)\) be the fixed-sample binary-outcome subclass of \(\mathcal E_{n,d,\epsilon}\) consisting of laws \(P^{\otimes n}\) with arbitrary masses \(p_k\ge0\), \(\sum_{k=1}^d p_k=1\), propensities satisfying \(\epsilon\le\pi_k\le1-\epsilon\) on every positive-mass category, arbitrary treated response means \(\mu_{1k}\in[0,1]\), and \(\mu_{0k}=0\) for every \(k\). Define \(\psi_1(P)=\sum_{k=1}^d p_k\mu_{1k}\). There are constants \(a_\epsilon,b_\epsilon>0\) and \(N_\epsilon<\infty\) such that, for all \(n\ge N_\epsilon\) and \(d\le b_\epsilon n\log n\),
\[
\inf_{\widehat\psi}\sup_{P^{\otimes n}\in\mathcal D_0^{(n,d)}(\epsilon)}E_P[(\widehat\psi-\psi_1(P))^2]\ge a_\epsilon\left\{\frac1n+\frac{d^2}{n^2\log^2n}\right\},
\]
where the infimum is over measurable estimators based on the fixed sample of size \(n\).
\end{lemma}
\par\smallskip\noindent \textit{Justification.} The cited v3 source supplies the fixed-sample one-arm scale; its control-zero construction and Poisson-to-fixed-sample transfer are mapped at theorem-and-appendix granularity. \textit{Gap.} This is a cited lower-bound substrate, not a new proof in this note. \textit{Consumer.} The ATE lower-bound transfer.

\begin{lemma}[lem:ate-lower-bound-transfer]\label{lem:ate-lower-bound-transfer}
For fixed \(0<\epsilon<1/2\), there are \(a_\epsilon,b_\epsilon>0\) such that, for all sufficiently large \(n\) and \(d\le b_\epsilon n\log n\),
\[
\mathsf R_{n,d,\epsilon}\ge a_\epsilon\left\{\frac1n+\frac{d^2}{n^2\log^2n}\right\}.
\]
\end{lemma}
\begin{proof}
Restrict to \(\mathcal D_0^{(n,d)}(\epsilon)\). There \(\mu_{0k}=0\), so
\[
\tau(P)=\sum_kp_k(\mu_{1k}-\mu_{0k})=\sum_kp_k\mu_{1k}=\psi_1(P).
\tag{1}
\]
Hence every estimator satisfies
\[
\sup_{P^{\otimes n}\in\mathcal E_{n,d,\epsilon}}E_P[(\widehat\tau-\tau(P))^2]\ge\sup_{P^{\otimes n}\in\mathcal D_0^{(n,d)}(\epsilon)}E_P[(\widehat\tau-\psi_1(P))^2].
\tag{2}
\]
Taking the infimum over the same measurable functions and applying Lemma~\ref{lem:zeng-one-arm-minimax-lower} proves the claim.
\end{proof}
\par\smallskip\noindent \textit{Justification.} On the emitted control-zero subclass, \(\mu_{0k}=0\) makes \(\tau(P)=\psi_1(P)\), so restriction of the ATE minimax problem yields the cited lower bound. \textit{Gap.} No fixed-interior lower-transfer obligation remains. \textit{Consumer.} The matched minimax theorem.

\begin{lemma}[lem:near-randomization-linear-upper]\label{lem:near-randomization-linear-upper}
Define the centered estimator
\[
\widehat\tau_{\mathrm{ctr}}=\frac2n\sum_{i=1}^n(2A_i-1)\left(Y_i-\frac12\right).
\]
For every \(n,d\in\mathbb N\), every \(0<\epsilon\le1/2\), and every \(P^{\otimes n}\in\mathcal E_{n,d,\epsilon}\),
\[
E_P[(\widehat\tau_{\mathrm{ctr}}-\tau(P))^2]\le\frac1n+4(1/2-\epsilon)^2.
\]
\end{lemma}
\begin{proof}
Put \(W_i=2(2A_i-1)(Y_i-1/2)\), so that \(\widehat\tau_{\mathrm{ctr}}=n^{-1}\sum_{i=1}^nW_i\). Because \(A_i,Y_i\in\{0,1\}\), one has \(|W_i|=1\). Conditional on \(X=k\),
\[
E_P[W_i\mid X=k]=2\left[\pi_k\left(\mu_{1k}-\frac12\right)-(1-\pi_k)\left(\mu_{0k}-\frac12\right)\right].
\tag{1}
\]
Subtracting the conditional ATE contribution gives the exact identity
\[
E_P[W_i\mid X=k]-(\mu_{1k}-\mu_{0k})
=2\left(\pi_k-\frac12\right)(\mu_{1k}+\mu_{0k}-1).
\tag{2}
\]
Since \(\epsilon\le\pi_k\le1-\epsilon\) on every positive-mass category and \(|\mu_{1k}+\mu_{0k}-1|\le1\), equations (1)--(2) and Definition~\ref{def:ate-functional} imply
\[
\left|E_P[\widehat\tau_{\mathrm{ctr}}]-\tau(P)\right|
\le2(1/2-\epsilon)\sum_{k=1}^dp_k
=2(1/2-\epsilon).
\tag{3}
\]
Independence and \(|W_i|=1\) give
\[
\operatorname{Var}_P(\widehat\tau_{\mathrm{ctr}})=\frac{\operatorname{Var}_P(W_1)}n\le\frac1n.
\tag{4}
\]
The bias--variance identity and (3)--(4) prove the displayed bound.
\end{proof}
\par\smallskip\noindent \textit{Justification.} Centering at \(1/2\) makes the score satisfy \(|W_i|=1\) and yields the exact conditional bias identity \(2(\pi_k-1/2)(\mu_{1k}+\mu_{0k}-1)\), giving variance at most \(1/n\) and squared bias at most \(4(1/2-\epsilon)^2\). \textit{Gap.} This is not a triangular-array converse. \textit{Consumer.} Endpoint and combined-envelope results.

\begin{lemma}[lem:one-category-bernoulli-lower]\label{lem:one-category-bernoulli-lower}
For every \(n,d\in\mathbb N\), \(\mathsf R_{n,d,1/2}\ge1/(100n)\).
\end{lemma}
\begin{proof}
Fix \(h=2/5\). For \(j\in\{0,1\}\), let \(P_j\) put all mass on \(X=1\), take \(A\sim\mathsf{Bernoulli}(1/2)\), set \(Y=0\) when \(A=0\), and take \(Y\mid A=1\sim\mathsf{Bernoulli}(1/2+jh/\sqrt n)\). Since \(h<1/2\), both response probabilities lie in \([0,1]\) for every \(n\ge1\). Both laws therefore lie in \(\mathcal E_{n,d,1/2}\), and Definition~\ref{def:ate-functional} gives
\[
\Delta_n:=|\tau(P_1)-\tau(P_0)|=\frac h{\sqrt n}.
\tag{1}
\]
Only the treated conditional response law differs. Direct calculation and \(-\log(1-x)\le x/(1-x)\) for \(0\le x<1\) give
\[
\begin{aligned}
D_{\mathrm{KL}}(P_0^{\otimes n}\|P_1^{\otimes n})
&=-\frac n4\log(1-4h^2/n)\\
&\le\frac{h^2}{1-4h^2/n}\le\frac{h^2}{1-4h^2}=\frac49.
\end{aligned}
\tag{2}
\]
Let \(Q_j=P_j^{\otimes n}\), let \(E=\{dQ_0\ge dQ_1\}\), and put \(t=Q_0(E)-Q_1(E)=\|Q_0-Q_1\|_{\mathrm{TV}}\). The log-sum inequality gives \(D_{\mathrm{KL}}(Q_0\|Q_1)\ge d(q+t\|q)\), where \(q=Q_1(E)\) and \(d(\cdot\|\cdot)\) is binary relative entropy. Its second derivative in the first argument is \(1/\{p(1-p)\}\ge4\), while its value and first derivative vanish at \(p=q\). Hence \(d(q+t\|q)\ge2t^2\), so the Pinsker bound and (2) yield
\[
t\le\sqrt{\frac{2}{9}}=\frac{\sqrt2}{3}.
\tag{3}
\]
Squaring (1) gives \(\Delta_n^2=4/(25n)\). Classify an estimate by the nearer of \(\tau(P_0)\) and \(\tau(P_1)\). Every classification error entails squared loss at least \(\Delta_n^2/4\), and the minimum sum of the two testing errors is \(1-t\). Therefore, for every estimator \(\widehat\tau\),
\[
\begin{aligned}
\max_{j\in\{0,1\}}E_{P_j}[(\widehat\tau-\tau(P_j))^2]
&\ge\frac{\Delta_n^2}{8}(1-t)\\
&\ge\frac1{50n}\left(1-\frac{\sqrt2}{3}\right)>\frac1{100n},
\end{aligned}
\tag{4}
\]
because \(2\sqrt2<3\). Taking the infimum over estimators and using \(\{P_0,P_1\}\subseteq\mathcal E_{n,d,1/2}\) proves the claim.
\end{proof}
\par\smallskip\noindent \textit{Justification.} The explicit one-category Bernoulli pair with \(h=2/5\) has Kullback--Leibler divergence at most \(4/9\) and gives the uniform constant \(c_0=1/100\). \textit{Gap.} The construction intentionally removes dimension dependence at exact randomization. \textit{Consumer.} The randomized-endpoint theorem.

\begin{lemma}[lem:randomized-endpoint-minimax]\label{lem:randomized-endpoint-minimax}
For every \(n,d\in\mathbb N\),
\[
\frac{1}{100n}\le\mathsf R_{n,d,1/2}\le\frac1n.
\]
\end{lemma}
\begin{proof}
At \(\epsilon=1/2\), overlap forces \(\pi_k=1/2\) on every positive-mass category. Lemma~\ref{lem:near-randomization-linear-upper} therefore gives risk at most \(1/n\) for \(\widehat\tau_{\mathrm{ctr}}\), proving the upper bound. Lemma~\ref{lem:one-category-bernoulli-lower} proves the lower bound with the explicit constant \(1/100\).
\end{proof}
\par\smallskip\noindent \textit{Justification.} The centered estimator gives the upper constant one, while the explicit Bernoulli two-point experiment gives the lower constant \(1/100\). \textit{Gap.} No dimension dependence remains. \textit{Consumer.} The endpoint clause.

\begin{lemma}[lem:combined-upper-envelope]\label{lem:combined-upper-envelope}
Fix \(0<\epsilon<1/2\) and constants \(C_\epsilon,\rho_\epsilon,N_\epsilon\) for which the upper bound in Theorem~\ref{thm:sharp-minimax-fixed-interior} holds. For \(n\ge N_\epsilon\) and \(d\le\rho_\epsilon n\log n\), choose the hybrid when
\[
C_\epsilon\left\{\frac1n+\frac{d^2}{n^2\log^2n}\right\}\le\frac1n+4(1/2-\epsilon)^2,
\]
and otherwise choose \(\widehat\tau_{\mathrm{ctr}}\). The selected estimator has worst-case risk at most
\[
K_\epsilon\left[\frac1n+\min\left\{\frac{d^2}{n^2\log^2n},(1/2-\epsilon)^2\right\}\right],\qquad K_\epsilon=\max\{C_\epsilon,4\}.
\]
\end{lemma}
\begin{proof}
Write \(a=n^{-1}\), \(u=d^2/(n^2\log^2n)\), and \(v=(1/2-\epsilon)^2\). Theorem~\ref{thm:sharp-minimax-fixed-interior} and Lemma~\ref{lem:near-randomization-linear-upper} show that the selected estimator has worst-case risk at most
\[
\min\{C_\epsilon(a+u),a+4v\}.
\tag{1}
\]
For \(K_\epsilon=\max\{C_\epsilon,4\}\), the first entry in (1) is at most \(K_\epsilon(a+u)\), and the second is at most \(K_\epsilon(a+v)\). Therefore
\[
\min\{C_\epsilon(a+u),a+4v\}
\le K_\epsilon\min\{a+u,a+v\}
=K_\epsilon\{a+\min(u,v)\}.
\tag{2}
\]
\end{proof}
\par\smallskip\noindent \textit{Justification.} The smaller of \(C_\epsilon(a+u)\) and \(a+4v\) is chosen deterministically once \(n\ge N_\epsilon\), yielding \(K_\epsilon=\max\{C_\epsilon,4\}\). \textit{Gap.} No matching triangular-array lower envelope is proved. \textit{Consumer.} Near-endpoint method choice.

\begin{theorem}[thm:sharp-minimax-fixed-interior]\label{thm:sharp-minimax-fixed-interior}
For every fixed \(0<\epsilon<1/2\), there are constants \(a_\epsilon,\rho_\epsilon,C_\epsilon>0\) and \(N_\epsilon<\infty\) such that, for every \(n\ge N_\epsilon\) and uniformly over \(d\le\rho_\epsilon n\log n\), the balanced ratio-polynomial hybrid in \(def:hybrid-estimator-handle\) satisfies
\[
a_\epsilon\left\{\frac1n+\frac{d^2}{n^2\log^2n}\right\}\le\mathsf R_{n,d,\epsilon}\le\sup_{P^{\otimes n}\in\mathcal E_{n,d,\epsilon}}E_P[(\widehat\tau_n^{\mathrm{hyb}}-\tau(P))^2]\le C_\epsilon\left\{\frac1n+\frac{d^2}{n^2\log^2n}\right\}.
\]
Consequently, the minimax MSE has parametric order \(n^{-1}\) whenever \(d=O(\sqrt n\log n)\), and, along sequences in the displayed range, it converges to zero if and only if \(d=o(n\log n)\).
\end{theorem}
\begin{proof}
Put \(r_{n,d}=n^{-1}+d^2/(n^2\log^2n)\). Lemma~\ref{lem:light-cell-polynomial} and Lemma~\ref{lem:universal-heavy-cell-rate} give, after decreasing the common range constant and increasing the common finite cutoff if necessary, the light- and heavy-branch bounds
\[
E_P\left[\left\{\widehat T_{\mathcal L}-\sum_{k\in\widehat{\mathcal L}_n}\phi(q_k)\right\}^2\right]\le C_\epsilon r_{n,d},\qquad E_P\left[\left\{\widehat T_{\mathcal H}-\sum_{k\in\widehat{\mathcal H}_n}\phi(q_k)\right\}^2\right]\le C_\epsilon r_{n,d}.
\tag{1}
\]
The pilot sets partition \(\{1,\ldots,d\}\), and Definition~\ref{def:ate-functional} gives
\[
\sum_{k\in\widehat{\mathcal H}_n}\phi(q_k)+\sum_{k\in\widehat{\mathcal L}_n}\phi(q_k)=\sum_{k=1}^d\phi(q_k)=\tau(P).
\tag{2}
\]
Thus \((x+y)^2\le2x^2+2y^2\) and (1) bound the MSE of the untruncated sum by \(4C_\epsilon r_{n,d}\). Because \(\tau(P)\in[-1,1]\), projection of that sum onto \([-1,1]\) cannot increase its distance from \(\tau(P)\). Hence the same order holds for \(\widehat\tau_n^{\mathrm{hyb}}\). The hybrid is a single measurable estimator in the infimum defining \(\mathsf R_{n,d,\epsilon}\), which proves the middle and upper inequalities.

By Lemma~\ref{lem:ate-lower-bound-transfer}, there are \(a_\epsilon,b_\epsilon>0\) and a finite cutoff such that
\[
\mathsf R_{n,d,\epsilon}\ge a_\epsilon r_{n,d}
\tag{3}
\]
whenever \(d\le b_\epsilon n\log n\). Intersecting the upper and lower ranges, taking \(\rho_\epsilon\) to be the smaller range constant, and taking \(N_\epsilon\) to be the larger finite cutoff proves the three-sided display. If \(d=O(\sqrt n\log n)\), then \(r_{n,d}=O(n^{-1})\), whereas (3) retains the term \(a_\epsilon/n\); hence the minimax MSE has exact parametric order. Finally, \(r_{n,d}\to0\) if and only if \(d/(n\log n)\to0\), proving the asserted consistency equivalence along sequences in the displayed range.
\end{proof}
\par\smallskip\noindent \textit{Justification.} The computable hybrid upper bound matches the source-backed control-zero converse. \textit{Gap.} The fixed-interior log-squared gap is closed. \textit{Consumer.} Minimax risk, parametric elbow, and consistency.

\begin{theorem}[thm:overlap-adaptive-universal-hybrid]\label{thm:overlap-adaptive-universal-hybrid}
The following conclusions hold.

\(\mathrm{(i)}\) The single computable hybrid \(\widehat\tau_n^{\mathrm{hyb}}\), with \(A=6\), \(\lambda_0=256\), \(b_0=4096\), and numerical \(\alpha_0\), does not take \(\epsilon\) as input. For every fixed \(0<\epsilon<1/2\), there are \(C_\epsilon,\rho_\epsilon>0\) and \(N_\epsilon<\infty\) such that, for every \(n\ge N_\epsilon\) and \(d\le\rho_\epsilon n\log n\),
\[
\sup_{P^{\otimes n}\in\mathcal E_{n,d,\epsilon}}E_P[(\widehat\tau_n^{\mathrm{hyb}}-\tau(P))^2]\le C_\epsilon\left\{\frac1n+\frac{d^2}{n^2\log^2n}\right\}.
\]
Thus fixed-class overlap adaptation requires no selector.

\(\mathrm{(ii)}\) For every \(n,d\in\mathbb N\) and \(0<\epsilon\le1/2\), the centered estimator
\[
\widehat\tau_{\mathrm{ctr}}=\frac2n\sum_{i=1}^n(2A_i-1)\left(Y_i-\frac12\right)
\]
satisfies
\[
\sup_{P^{\otimes n}\in\mathcal E_{n,d,\epsilon}}E_P[(\widehat\tau_{\mathrm{ctr}}-\tau(P))^2]\le\frac1n+4(1/2-\epsilon)^2.
\]

\(\mathrm{(iii)}\) For every \(n,d\in\mathbb N\),
\[
\frac{1}{100n}\le\mathsf R_{n,d,1/2}\le\frac1n.
\]

\(\mathrm{(iv)}\) Fix \(0<\epsilon<1/2\) and constants \(C_\epsilon,\rho_\epsilon,N_\epsilon\) from the fixed-interior upper theorem. For \(n\ge N_\epsilon\) and \(d\le\rho_\epsilon n\log n\), choose the hybrid when
\[
C_\epsilon\left\{\frac1n+\frac{d^2}{n^2\log^2n}\right\}\le\frac1n+4(1/2-\epsilon)^2,
\]
and otherwise choose \(\widehat\tau_{\mathrm{ctr}}\). This deterministic choice satisfies
\[
\mathsf R_{n,d,\epsilon}\le K_\epsilon\left[\frac1n+\min\left\{\frac{d^2}{n^2\log^2n},(1/2-\epsilon)^2\right\}\right],\qquad K_\epsilon=\max\{C_\epsilon,4\}.
\]

\(\mathrm{(v)}\) These fixed-class and endpoint conclusions do not assert a matching lower envelope for triangular arrays \(\epsilon=\epsilon_n\).
\end{theorem}
\begin{proof}
All tuning quantities in Definition~\ref{def:hybrid-estimator-handle} are numerical and independent of \(\epsilon\), and all operations use observed counts. A four-variable polynomial of degree at most \(M\) has \(\binom{M+4}{4}=O(M^4)\) monomials, so the estimator is computable in \(O(dM^4)\) arithmetic operations. The upper inequality in Theorem~\ref{thm:sharp-minimax-fixed-interior} proves clause \(\mathrm{(i)}\).

Clause \(\mathrm{(ii)}\) is Lemma~\ref{lem:near-randomization-linear-upper}. Lemma~\ref{lem:one-category-bernoulli-lower} gives the explicit lower constant \(1/100\), and Lemma~\ref{lem:randomized-endpoint-minimax} combines it with the centered upper bound to prove clause \(\mathrm{(iii)}\) for every \(n,d\). Lemma~\ref{lem:combined-upper-envelope}, with its explicit requirement \(n\ge N_\epsilon\), proves clause \(\mathrm{(iv)}\). Clause \(\mathrm{(v)}\) follows from the quantifiers: the hybrid constants are pointwise in fixed \(\epsilon\), and no displayed lower bound supplies a uniform triangular-array transition.
\end{proof}
\par\smallskip\noindent \textit{Justification.} The theorem collects universal \(A=6\) fixed-class tuning, the centered near-randomization bound, the explicit endpoint bracket, and the sharpened combined upper envelope. \textit{Gap.} Only the matching triangular-array phase remains. \textit{Consumer.} Implementation and overlap-phase interpretation.
