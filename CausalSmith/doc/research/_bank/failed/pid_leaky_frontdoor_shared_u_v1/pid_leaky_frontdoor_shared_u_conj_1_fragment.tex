\paragraph{\S9-fragment-statement: Refutation note for the labelled solve target \texttt{nondecomp}.}\label{prop:nondecomp-refuted}
\textbf{Conjecture restated verbatim (Conjecture~\ref{conj:nondecomp}).}
\textnormal{(NOT YET PROVED)} Let $\kappa(P)$ be the observable signature \eqref{eq:kappa}.
\begin{enumerate}[label=(\alph*),leftmargin=2em,itemsep=1pt]
  \item For all $\lambda\ge0$, $[L(\lambda),U(\lambda)]\subseteq B_{\mathrm{box}}(\lambda)\cap B_{\mathrm{sens}}(\lambda)$.
  \item Under Assumption~\ref{ass:mediator-shift}, on the relatively open observable set
        $\{P:\kappa(P)\neq0,\ q_{\cdot\mid1}\neq q_{\cdot\mid0}\}$, there is $\lambda_0(P)>0$ such that
        for $0<\lambda<\lambda_0(P)$ the inclusion in (a) is \emph{strict} in both factors.
  \item At the exclusion corner the per-mediator box closes:
        $B_{\mathrm{box}}(0)=I(0)=\{\theta_{\mathrm{FD}}\}$, so the strict inclusion in (b) against
        $B_{\mathrm{box}}$ is necessarily a strictly-positive-$\lambda$ phenomenon. In mediator-stationary
        regimes $q_{\cdot\mid1}=q_{\cdot\mid0}$ the strict-inclusion assertion is not made.
\end{enumerate}

\textbf{Verdict.} Refuted: the allowed observable law with \(p_0=p_1=1/2\), \(q_{1\mid0}=1/4\), \(q_{1\mid1}=3/4\), \(r_{0,0}=2/5\), \(r_{1,0}=3/5\), \(r_{0,1}=3/10\), and \(r_{1,1}=7/10\) satisfies \(\kappa(P)=3/10\neq0\) and \(q_{\cdot\mid1}\neq q_{\cdot\mid0}\), but for every \(0\le\lambda\le3/10\) the shared-\(U\) interval and the per-mediator box coincide, \(I(\lambda)=B_{\mathrm{box}}(\lambda)=[-\lambda,\lambda]\); hence no positive \(\lambda_0\) can make the inclusion strict against \(B_{\mathrm{box}}\).

\paragraph{\S13-refutation-attempt and counterexample.}
\textbf{Primitive executed.} I used the two-strata response-type search requested for
partial-identification refutations. The search space has
\(\mathcal U=\{a,b\}\), binary \(X\), binary \(M\), and bounded outcome means.
The active candidate is the smallest shared-\(U\) completion in which the two
treatment arms load on different latent strata, the mediator distribution shifts
with treatment, and all four observed conditional means stay away from the box
boundary. This is the two-vertex face on which Pearl's front-door corner
\cite{Pearl1995CausalDiagrams}, already recovered in Theorem~\ref{thm:collapse},
can be perturbed by a common direct-leak increment without violating the clean
mediator equations.

\textbf{Witness law.} Let
\[
  p_0=p_1=\frac12,\qquad
  q_{1\mid0}=\frac14,\quad q_{0\mid0}=\frac34,\qquad
  q_{1\mid1}=\frac34,\quad q_{0\mid1}=\frac14,
\]
and set
\[
  r_{0,0}=\frac25,\qquad r_{1,0}=\frac35,\qquad
  r_{0,1}=\frac3{10},\qquad r_{1,1}=\frac7{10}.
\]
The observed support and overlap assumption holds because all treatment and
mediator probabilities are positive. The outcome means lie in \([0,1]\), and
indeed the distance from every displayed \(r_{x,m}\) to the nearest outcome
boundary is at least \(3/10\). The mediator-shift assumption holds because
\(q_{\cdot\mid1}\neq q_{\cdot\mid0}\).

At the exclusion corner, use the two-point support \(\mathcal U=\{a,b\}\) with
\[
  w_0(a)=1,\qquad w_1(b)=1,\qquad \pi(a)=\pi(b)=\frac12.
\]
For each mediator value \(m\), set
\[
  \varphi(m,0,a)=\varphi(m,1,a)=r_{0,m},\qquad
  \varphi(m,0,b)=\varphi(m,1,b)=r_{1,m}.
\]
Then Assumption~\ref{ass:clean} gives the compatibility equations
\[
  \sum_u\varphi(m,0,u)w_0(u)=r_{0,m},\qquad
  \sum_u\varphi(m,1,u)w_1(u)=r_{1,m},
\]
and the direct leak is zero. The finite support, bounded outcome, clean mediator,
leak, and overlap assumptions are therefore all satisfied.

\textbf{Observable signature.} The front-door functional is
\[
  \theta_{\mathrm{FD}}
  =\sum_m(q_{m\mid1}-q_{m\mid0})(p_1r_{1,m}+p_0r_{0,m})
  =-\frac12\cdot\frac12+\frac12\cdot\frac12=0.
\]
The observed treatment contrast is
\[
  \E[Y\mid X=1]-\E[Y\mid X=0]
  =\left(\frac14\cdot\frac35+\frac34\cdot\frac7{10}\right)
   -\left(\frac34\cdot\frac25+\frac14\cdot\frac3{10}\right)
  =\frac3{10}.
\]
Hence \(\kappa(P)=3/10\neq0\). This witness lies exactly in the current
correction-round strictness regime:
\[
  \kappa(P)\neq0,\qquad q_{\cdot\mid1}\neq q_{\cdot\mid0}.
\]

\textbf{Evaluation of the two-strata vertices.} Fix \(0\le\lambda\le3/10\). For
any feasible shared-\(U\) completion, write
\[
  s_m(u)=\varphi(m,0,u),\qquad d_m(u)=\varphi(m,1,u)-\varphi(m,0,u).
\]
Using Bayes consistency \(\pi=p_0w_0+p_1w_1\), the compatibility equations
\eqref{eq:compat}, and the algebraic identity
\(\varphi(m,1,u)=s_m(u)+d_m(u)\), the causal objective satisfies
\[
\begin{aligned}
  \theta-\theta_{\mathrm{FD}}
  &=\sum_m\left\{q_{m\mid1}\sum_u d_m(u)\pi(u)
       -p_1\bigl(q_{m\mid1}-q_{m\mid0}\bigr)\sum_u d_m(u)w_1(u)\right\} \\
  &=\sum_m\left\{
       p_0q_{m\mid1}\sum_u d_m(u)w_0(u)
       +p_1q_{m\mid0}\sum_u d_m(u)w_1(u)\right\}.
\end{aligned}
\]
The leak restriction gives \(|d_m(u)|\le\lambda\), and the nonnegative weights
\(\{p_0q_{m\mid1},p_1q_{m\mid0}\}_{m\in\{0,1\}}\) sum to one. Therefore every
shared-\(U\) feasible value satisfies
\[
  -\lambda\le\theta-\theta_{\mathrm{FD}}\le\lambda.
\]
Since \(\theta_{\mathrm{FD}}=0\), this proves \(I(\lambda)\subseteq[-\lambda,\lambda]\).

Conversely, every \(t\in[-\lambda,\lambda]\) is attained by a shared-\(U\)
completion. Keep the same \(w_0,w_1,\pi\) as above and define, for both mediator
values,
\[
  \varphi(m,0,a)=r_{0,m},\qquad
  \varphi(m,1,a)=r_{0,m}+t,\qquad
  \varphi(m,1,b)=r_{1,m},\qquad
  \varphi(m,0,b)=r_{1,m}-t.
\]
Because \(|t|\le\lambda\le3/10\), all four structural means stay in \([0,1]\).
The compatibility equations are unchanged: \(w_0\) observes only stratum \(a\)
and \(w_1\) observes only stratum \(b\). The direct leak equals \(t\) at both
latent states and both mediator values, so \(|d_m(u)|\le\lambda\). Substituting
the displayed completion into \eqref{eq:ace} gives
\[
  \theta=t.
\]
Thus
\[
  I(\lambda)=[-\lambda,\lambda]\qquad(0\le\lambda\le3/10).
\]

The same calculation applies to the per-mediator box. In
\(B_{\mathrm{box}}(\lambda)\) the shared pair \((w_0,w_1)\) is relaxed to
mediator-specific pairs \((w_0^{(m)},w_1^{(m)})\), but for each mediator value
the identity above remains valid with \(w_x\) replaced by \(w_x^{(m)}\). Hence
\[
  \theta-\theta_{\mathrm{FD}}
  =\sum_m\left\{
       p_0q_{m\mid1}\sum_u d_m(u)w_0^{(m)}(u)
       +p_1q_{m\mid0}\sum_u d_m(u)w_1^{(m)}(u)\right\},
\]
so the leak bound again implies \(B_{\mathrm{box}}(\lambda)\subseteq[-\lambda,\lambda]\).
Because the shared-\(U\) completions just constructed are also admissible for the
relaxed per-mediator problem, the reverse inclusion follows from
\(I(\lambda)\subseteq B_{\mathrm{box}}(\lambda)\). Therefore
\[
  B_{\mathrm{box}}(\lambda)=[-\lambda,\lambda]=I(\lambda)
  \qquad(0\le\lambda\le3/10).
\]

This equality refutes part (b) of Conjecture~\ref{conj:nondecomp} as corrected in
this round. The witness satisfies Assumption~\ref{ass:mediator-shift}, has
\(\kappa(P)\neq0\), and has \(q_{\cdot\mid1}\neq q_{\cdot\mid0}\), but for every
candidate \(\lambda_0(P)>0\) one can choose
\[
  0<\lambda<\min\{\lambda_0(P),3/10\}
\]
and obtain equality rather than strict inclusion against \(B_{\mathrm{box}}(\lambda)\).
The conjecture requires strict inclusion in both factors, so failure in the
per-mediator box factor is enough to refute the stated claim. The obstruction is
not the mediator-stationary face excluded by the approved assumption; it is the
two-strata disjoint-support face on which a common leak increment saturates both
the shared-\(U\) problem and the per-mediator relaxation.
